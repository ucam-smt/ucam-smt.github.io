34 template<class Data, class KenLMModelT = lm::ngram::Model, class Arc = fst::StdArc >
35 class DisambigTask : public ucam::util::TaskInterface<Data> {
36 
37  typedef typename Arc::Weight Weight;
38  typedef typename Arc::Label Label;
39 
40  private:
41 
43  fst::VectorFst<Arc> *unimap_;
45  uint shp_;
46 
48  float prune_;
49 
51  const ucam::util::RegistryPO& rg_;
52 
54  const std::string lmkey_;
56  const std::string inputkey_;
58  const std::string outputkey_;
60  const std::string unimapkey_;
61 
63  Data *d_;
64 
66  fst::MakeWeight2<Arc> mw_;
67 
69  fst::VectorFst<Arc> olattice_;
70 
71  public:
73  DisambigTask ( const ucam::util::RegistryPO& rg,
74  const std::string& inputkey = HifstConstants::kRecaserInput,
75  const std::string& outputkey = HifstConstants::kRecaserOutput,
76  const std::string& lmkey = HifstConstants::kRecaserLmLoad,
77  const std::string& unimapkey = HifstConstants::kRecaserUnimapLoad,
78  bool forceloading = false ) :
79  rg_ ( rg ),
80  lmkey_ ( lmkey ),
81  inputkey_ ( inputkey ),
82  outputkey_ ( outputkey ),
83  unimapkey_ ( unimapkey ),
84  unimap_ ( NULL ) {
85  using ucam::util::toNumber;
86  // Prune or shortest path...
87  std::vector<std::string> pstrat = rg_.getVectorString (
88  HifstConstants::kRecaserPrune );
89  USER_CHECK ( pstrat.size() == 2,
90  "prune parameter must be byshortestpath/byweight,number" );
91  if ( pstrat[0] == "byshortestpath" ) {
92  LINFO ( "Shortest Path n=" << pstrat[1] );
93  shp_ = toNumber<uint> ( pstrat[1] );
94  prune_ = std::numeric_limits<float>::max();
95  } else if ( pstrat[0] == "byweight" ) {
96  LINFO ( "Prune by weight b=" << pstrat[1] );
97  shp_ = std::numeric_limits<uint>::max();;
98  prune_ = toNumber<float> ( pstrat[1] );
99  } else {
100  USER_CHECK ( false,
101  "prune parameter incorrectly set: first parameter is byshortestpath or byweight" );
102  }
103  if ( rg.get<std::string> ( lmkey_) == ""
104  && rg.get<std::string> ( unimapkey_ ) == "" ) return;
105  if ( ! USER_CHECK ( rg.get<std::string> ( lmkey_ ) != ""
106  && rg.get<std::string> ( unimapkey_ ) != "" ,
107  "recaser.lm and recaser.unimap must either be both defined or both left to empty string " ) )
108  return;
109  };
110 
112  bool run ( Data& d ) {
113  d_ = &d;
114  USER_CHECK ( d.fsts.find ( inputkey_ ) != d.fsts.end(),
115  "No input fst to recase?" );
116  fst::ShortestPath<Arc> ( * ( static_cast< fst::VectorFst<Arc> * >
117  (d.fsts[inputkey_] ) ), &olattice_, 1 );
118  fst::Map<Arc> ( &olattice_, fst::RmWeightMapper<Arc>() );
119  run ( &olattice_ );
120  LINFO ( "(Recased) lattice available at key=" << outputkey_ );
121  d.fsts[outputkey_] = &olattice_;
122  return false;
123  };
124 
126  virtual ~DisambigTask() { }
127 
128  private:
129 
131  void run ( fst::VectorFst<Arc> *fst ) {
132  if ( d_->fsts.find ( unimapkey_ ) == d_->fsts.end() ) {
133  LINFO ( "No recasing step (key=" << unimapkey_ << " not found)" );
134  return;
135  } else if ( d_->fsts[unimapkey_] == NULL ) {
136  LINFO ( "No recasing step (NULL) " );
137  return;
138  }
139  unimap_ = static_cast<fst::VectorFst<Arc> *> ( d_->fsts[unimapkey_] );
140  LINFO ( "Apply Unigram Model to 1-best" );
141  fst::VectorFst<Arc> mappedinput ( fst::RRhoCompose<Arc> ( *fst, *unimap_ ) );
142  LINFO ( "Tag OOVs" );
143  tagOOVs<Arc> ( &mappedinput, *d_->recasingvcblm );
144  LDBG_EXECUTE ( mappedinput.Write ( "mappedinput.fst" ) );
145  USER_CHECK ( d_->klm.find ( lmkey_ ) != d_->klm.end(),
146  "Language Model not loaded!" );
147  USER_CHECK ( d_->klm[lmkey_].size(), "Language Model not loaded!" );
148  KenLMModelT& model = * ( d_->klm[lmkey_][0]->model );
149 #ifndef USE_GOOGLE_SPARSE_HASH
150  unordered_set<Label> epsilons;
151 #else
152  google::dense_hash_set<Label> epsilons;
153  epsilons.set_empty_key ( std::numeric_limits<Label>::max() );
154 #endif
155 
156  epsilons.insert ( DR );
157  epsilons.insert ( OOV );
158  epsilons.insert ( EPSILON );
159  epsilons.insert ( SEP );
160  fst::ApplyLanguageModelOnTheFly<Arc, fst::MakeWeight<Arc>, KenLMModelT> *f =
161  new fst::ApplyLanguageModelOnTheFly<Arc, fst::MakeWeight<Arc>, KenLMModelT>
162  ( mappedinput
163  , model
164  , epsilons
165  , true
166  , d_->klm[lmkey_][0]->lmscale
167  , d_->klm[lmkey_][0]->lmwp
168  , d_->klm[lmkey_][0]->idb );
169  fst::VectorFst<Arc> output = * ( ( *f ) () );
170  delete f;
171  LINFO ( "Recover OOVs" );
172  recoverOOVs<Arc> ( &output );
173  if ( shp_ < std::numeric_limits<uint>::max() ) {
174  fst::VectorFst<Arc> aux;
175  LINFO ( "Shortest Path n=" << shp_ );
176  fst::ShortestPath<Arc> ( output, &aux, shp_ );
177  output = aux;
178  fst::TopSort<Arc> ( &output );
179  } else if ( prune_ < std::numeric_limits<float>::max() ) {
180  LINFO ( "Prune by weight=" << prune_ );
181  fst::Prune<Arc> ( &output, mw_ ( prune_ ) );
182  } else {
183  USER_CHECK ( false,
184  "prune parameter incorrectly set: first parameter is byshortestpath or byweight" );
185  }
186  *fst = output;
187  fst::Project ( fst, fst::PROJECT_OUTPUT ); //Recased symbols on output language.
188  }
189 
190  ZDISALLOW_COPY_AND_ASSIGN ( DisambigTask );
191 
192 };
