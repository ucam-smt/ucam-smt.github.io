59 TEST ( HifstReferenceFilter, basic_test ) {
60  //Create test fst file expecto.fst.
61  typedef fst::LexicographicArc< fst::StdArc::Weight, fst::StdArc::Weight> Arc;
62  typedef fst::LexicographicWeight<fst::StdArc::Weight, fst::StdArc::Weight>
63  Weight;
64  fst::VectorFst<Arc> aux;
65  fst::MakeWeight<Arc> mw;
66  aux.AddState();
67  aux.AddState();
68  aux.AddState();
69  aux.SetStart ( 0 );
70  aux.SetFinal ( 2, Arc::Weight::One() );
71  aux.AddArc ( 0, Arc ( 10, 10, mw ( 0 ), 1 ) );
72  aux.AddArc ( 1, Arc ( 100, 100, mw ( 0 ), 2 ) );
73  fst::FstWrite ( aux, "expecto.fst" );
74  //Prepare RegistryPO object.
75  unordered_map<std::string, boost::any> v;
76  v[HifstConstants::kReferencefilterLoad] = std::string ( "expecto.fst" );
77  v[HifstConstants::kReferencefilterWrite] = std::string ( "" );
78  v[HifstConstants::kReferencefilterSubstring] = std::string ("yes");
79  v[HifstConstants::kReferencefilterPrunereferenceweight] = float (
80  std::numeric_limits<float>::max() );
81  v[HifstConstants::kReferencefilterPrunereferenceshortestpath] = unsigned (
82  std::numeric_limits<unsigned>::max() );
83  DataForReferenceFilter d;
84  d.sidx = 0;
85  const uu::RegistryPO rg ( v );
86  uh::ReferenceFilterTask<DataForReferenceFilter> rft ( rg );
87  EXPECT_EQ ( rft.getBuilt(), false );
88  EXPECT_EQ ( rft.getDisableSubString(), false );
89  EXPECT_EQ ( rft.getWeight(), std::numeric_limits<float>::max() );
90  EXPECT_EQ ( rft.getShortestPath(), std::numeric_limits<unsigned>::max() );
91  EXPECT_EQ ( rft.getVocabulary().size(), 0 );
92  EXPECT_EQ ( rft.getTranslationLatticeFile(), "expecto.fst" );
93  rft.run ( d );
94  EXPECT_EQ ( rft.getBuilt(), true );
95  ASSERT_EQ ( d.tvcb.size(), 2 );
96  EXPECT_TRUE ( d.tvcb.find ( "10" ) != d.tvcb.end() );
97  EXPECT_TRUE ( d.tvcb.find ( "100" ) != d.tvcb.end() );
98  ASSERT_EQ ( d.filters.size(), 1 );
99  EXPECT_EQ ( d.filters[0]->NumStates(), 3 );
100  std::stringstream ss;
101  fst::PrintFst ( *d.filters[0], &ss );
102  EXPECT_EQ ( ss.str(),
103  "0\t1\t10\t10\n0\t2\t100\t100\n0\n1\t2\t100\t100\n1\n2\n" );
104  bfs::remove ( bfs::path ( "expecto.fst" ) );
105 };
