117 void buildSubstringTransducer ( fst::VectorFst<Arc> *myfst ) {
118  USER_CHECK ( myfst, "NULL pointer not accepted" );
119  USER_CHECK ( myfst->NumStates(), "Number of states is zero!" );
120  typedef typename Arc::StateId StateId;
121  fst::TopSort ( myfst );
122  fst::Map ( myfst, fst::RmWeightMapper<Arc>() );
123  for ( fst::StateIterator< fst::VectorFst<Arc> > si ( *myfst ); !si.Done();
124  si.Next() ) {
125  StateId state_id = si.Value();
126  if ( state_id ) {
127  myfst->AddArc ( 0, Arc ( 0, 0, Arc::Weight::One(), si.Value() ) );
128  myfst->SetFinal ( state_id, Arc::Weight::One() );
129  }
130  }
131  fst::RmEpsilon ( myfst );

147 template <class Arc>
148 inline fst::VectorFst<Arc> *EncodeDeterminizeMinimizeDecode (
149  fst::VectorFst<Arc> *myfst ) {
150  fst::EncodeMapper<Arc> em ( fst::kEncodeLabels, fst::ENCODE );
151  fst::Encode ( myfst, &em ); //note that this modifies fst1
152  LDBG_EXECUTE ( myfst->Write ( "fsts/encoded.fst" ) );
153  fst::VectorFst<Arc> *fst2 = new fst::VectorFst<Arc>;
154  fst::Determinize ( *myfst, fst2 );
155  fst::Minimize ( fst2 );
156  fst::EncodeMapper<Arc> em2 ( em,
157  fst::DECODE ); //create em2 by copying em but now with decode option.
158  fst::Decode ( fst2, em2 );
159  return fst2;

216 template<class Arc>
217 inline void printstrings ( const fst::VectorFst<Arc>& pcostslat,
218  std::ostream *hyps, unsigned s = 0 ) {
219  static std::basic_string<unsigned> ihyp;
220  static std::basic_string<unsigned> ohyp;
221  static std::basic_string<typename Arc::Weight> cost;
222  if ( !USER_CHECK ( s < pcostslat.NumStates(),
223  "The state most surely doesn't exist! Topsort this lattice and make sure you use a valid s" ) )
224  return;
225  typename Arc::Weight fvalue = Arc::Weight::One();
226  fst::ArcIterator< fst::VectorFst<Arc> > i ( pcostslat, s );
227  if ( !i.Done() ) {
228  for ( ; !i.Done(); i.Next() ) {
229  Arc a = i.Value();
230  ihyp.push_back ( a.ilabel );
231  ohyp.push_back ( a.olabel );
232  cost.push_back ( a.weight );
233  printstrings ( pcostslat, hyps, a.nextstate ); // recursive call to next state.
234  ihyp.resize ( ihyp.size() - 1 );
235  ohyp.resize ( ohyp.size() - 1 );
236  cost.resize ( cost.size() - 1 );
237  };
238  fvalue = pcostslat.Final ( s );
239  if ( fvalue == Arc::Weight::Zero() ) return;
240  }
241  for ( unsigned k = 0; k < ihyp.size();
242  ++k ) if ( ihyp[k] != 0 ) *hyps << ihyp[k] << " ";
243  *hyps << "|| ";
244  for ( unsigned k = 0; k < ihyp.size();
245  ++k ) if ( ohyp[k] != 0 ) *hyps << ohyp[k] << " ";
246  *hyps << "|| ";
247  typename Arc::Weight c = fvalue;
248  for ( unsigned k = 0; k < cost.size(); ++k ) {
249  c = Times ( c, cost[k] );
250  }
251  *hyps << c << endl;
252 };

271 template<class Arc>
272 inline void printstrings ( const fst::VectorFst<Arc>& fst
273  , unordered_map<std::string, float>& finalhyps
274  , bool input = true ) {
275  fst::VectorFst<Arc> ofst ( fst );
276  fst::TopSort ( &ofst ); //topological order is needed for the algorithm to work.
277  std::vector <std::vector<struct hypcost> > partialhyps;
278  partialhyps.clear();
279  finalhyps.clear();
280  struct hypcost hcempty;
281  hcempty.cost = 0;
282  std::vector<struct hypcost> vhcempty;
283  partialhyps.resize ( ofst.NumStates() );
284  partialhyps[0].push_back ( hcempty );
285  for ( fst::StateIterator< fst::MutableFst<Arc> > si ( ofst ); !si.Done();
286  si.Next() ) {
287  typename Arc::Weight value = ofst.Final ( si.Value() );
288  std::vector<struct hypcost>& hypc = partialhyps[ ( unsigned ) si.Value()];
289  for ( unsigned k = 0; k < hypc.size(); ++k ) {
290  if ( value != Arc::Weight::Zero() ) {
291  finalhyps[partialhyps[ ( unsigned ) si.Value()][k].hyp] =
292  partialhyps[ ( unsigned ) si.Value()][k].cost + ( float ) value.Value();
293  }
294  for ( fst::MutableArcIterator< fst::MutableFst<Arc> > ai ( &ofst, si.Value() );
295  !ai.Done(); ai.Next() ) {
296  struct hypcost hcnew = hypc[k];
297  Arc arc = ai.Value();
298  if ( input
299  && arc.ilabel != 0 ) hcnew.hyp += " " + ucam::util::toString ( arc.ilabel );
300  else if ( !input
301  && arc.olabel != 0 ) hcnew.hyp += " " + ucam::util::toString ( arc.olabel );
302  hcnew.cost += ( float ) arc.weight.Value();
303  partialhyps[arc.nextstate].push_back ( hcnew );
304  }
305  }
306  partialhyps[ ( unsigned ) si.Value()] =
307  vhcempty; //won't need these partial hyps any more
308  }
309 };

311 inline unsigned ShortestPathLength (const fst::VectorFst<fst::StdArc>* fst) {
312  using fst::StdArc;
313  using fst::VectorFst;
314  VectorFst<StdArc> tmp;
315  fst::ShortestPath (*fst, &tmp);
316  fst::RmEpsilon (&tmp);

318  for (fst::StateIterator< VectorFst<StdArc> > si (tmp); !si.Done(); si.Next() ) {

351 inline fst::VectorFst<fst::StdArc>* FstScaleWeights (fst::VectorFst<fst::StdArc>*
352  fst
353  , const double scale) {
354  using fst::StdArc;
355  using fst::VectorFst;
356  VectorFst<StdArc>* fstscaled = fst->Copy();
357  for (fst::StateIterator< VectorFst<StdArc> > si (*fstscaled); !si.Done();
358  si.Next() ) {
359  for (fst::MutableArcIterator< VectorFst<StdArc> > ai (fstscaled, si.Value() );
360  !ai.Done(); ai.Next() ) {
361  StdArc arc = ai.Value();
362  arc.weight = static_cast<StdArc::Weight> (arc.weight.Value() * scale);
363  ai.SetValue (arc);
364  }
365  StdArc::Weight final = fstscaled->Final (si.Value() );
366  if (final != ZPosInfinity() ) {
367  fstscaled->SetFinal (si.Value(),
368  static_cast<StdArc::Weight> (final.Value() * scale) );
369  }
370  }
371  return fstscaled;
372 }

377 template<class Arc>
378 inline void GetMinAndMaxHypothesisLength (const fst::VectorFst<Arc>* fst,
379  unsigned& jMin, unsigned& jMax) {
380  using fst::Map;
381  using fst::RmWeightMapper;
382  using fst::TimesMapper;
383  fst::VectorFst<Arc>* tmp = fst->Copy();
384  Map (tmp, RmWeightMapper<Arc, Arc>() );
385  Map (tmp, TimesMapper<Arc> ( UNIT_COST_POSITIVE ) );
386  jMin = ShortestPathLength (tmp);
387  Map (tmp, RmWeightMapper<Arc, Arc>() );
388  Map (tmp, TimesMapper<Arc> ( UNIT_COST_NEGATIVE ) );
389  jMax = ShortestPathLength (tmp);
390  delete tmp;
391 }

426 template<class Arc>
427 inline void string2fst (const std::string& sidxwords,
428  fst::VectorFst<Arc> *fst,
429  const std::string& tidxwords = "",
430  typename Arc::Weight finalweight = Arc::Weight::One()
431  ) {
432  assert (sidxwords != "");
433  std::vector<std::string> swords;
434  boost::algorithm::split (swords, sidxwords,
435  boost::algorithm::is_any_of ( " " ) );
436  fst->AddState();
437  fst->SetStart (0);
438  for (unsigned k = 0; k < swords.size(); ++k) {
439  typename Arc::Label swidx = ucam::util::toNumber<unsigned> (swords[k]);
440  fst->AddState();
441  fst->AddArc (k, Arc (swidx, 0, Arc::Weight::One(), k + 1) );
442  }
443  if (tidxwords == "") {
444  fst->SetFinal (swords.size(), Arc::Weight::One() );
445  fst::Project<Arc> (fst,
446  fst::PROJECT_INPUT); //if only source provided, we assume an automaton.
447  fst->SetFinal (swords.size(), finalweight);
448  return;
449  }
450  std::vector<std::string> twords;
451  boost::algorithm::split (twords, tidxwords,
452  boost::algorithm::is_any_of ( " " ) );
453  for (unsigned k = swords.size(); k < swords.size() + twords.size(); ++k) {
454  typename Arc::Label twidx = ucam::util::toNumber<unsigned>
455  (twords[k - swords.size()]);
456  fst->AddState();
457  fst->AddArc (k, Arc (0, twidx, Arc::Weight::One(), k + 1) );
458  }
459  fst->SetFinal (swords.size() + twords.size(), finalweight);
460  fst::Determinize (fst::RmEpsilonFst<Arc> (*fst), fst);
461  fst::Minimize (fst);
462  fst::RmEpsilon (fst);
463 };
