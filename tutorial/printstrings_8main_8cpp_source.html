187  PatternAddress<unsigned> input (rg.get<std::string>

189  PatternAddress<unsigned> output (rg.get<std::string>
190  (HifstConstants::kOutput.c_str() ) );
191  unsigned n = rg.get<unsigned> (HifstConstants::kNbest.c_str() );
192  boost::scoped_ptr<oszfstream> out;
193  bool unique = rg.exists (HifstConstants::kUnique.c_str() );
194  bool printOutputLabels = rg.exists(HifstConstants::kPrintOutputLabels.c_str());
195  std::string old;
196  for ( ucam::util::IntRangePtr ir (ucam::util::IntRangeFactory ( rg,
197  HifstConstants::kRangeOne ) );
198  !ir->done();
199  ir->next() ) {
200  FORCELINFO ("Processing file " << input ( ir->get() ) );
201  boost::scoped_ptr<fst::VectorFst<Arc> > ifst (fst::VectorFstRead<Arc> (input (
202  ir->get() ) ) );
203  fst::VectorFst<Arc> nfst;
204  Connect (&*ifst);
205  if (old != output (ir->get() ) ) {
206  out.reset (new oszfstream (output (ir->get() ) ) );
207  old = output (ir->get() );
208  }
209  if (!ifst->NumStates() ) {
210  *out << "[EMPTY]" << std::endl;
211  continue;
212  }
213  // Projecting allows unique to work for all cases.
214  if (printOutputLabels)
215  fst::Project(&*ifst, PROJECT_OUTPUT);
216  else
217  fst::Project(&*ifst, PROJECT_INPUT);
218  ShortestPath (*ifst, &nfst, n, unique );
219  std::vector<HypT> hyps;
220  fst::printStrings<Arc> (nfst, &hyps);
221  for (unsigned k = 0; k < hyps.size(); ++k) {
222  *out->getStream() << hyps[k] << std::endl;
223  }
224  }
