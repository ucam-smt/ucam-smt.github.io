82  bool run ( Data& d ) {
83  myfst_.DeleteStates();
84  d_ = &d;
85  fst::VectorFst<Arc> *lattice = static_cast<fst::VectorFst<Arc> *>
86  ( d.fsts[alilatskey_] );
87  USER_CHECK ( lattice->NumStates(), "Empty alignment lattice?" );
88  Invert ( lattice );
89  fst::VectorFst<TupleArc32> *lxr
90  = applyFlowerLattice (*lattice
91  , * ( static_cast<fst::VectorFst<TupleArc32> *>
92  ( d.fsts[ruleflowerlatticekey_] ) ) );
93  LDBG_EXECUTE ( lattice->Write ( "fsts/aplats/aplats+flower.fst" ) );
94  if (stripHifstEpsilons_) {
95  LINFO ("Remove hifst epsilons");
96  ru_ (lxr);
97  }
98  LINFO ( "Project, RmEpsilon, Determinize and Minimize" );
99  fst::Project ( lxr, fst::PROJECT_INPUT );
100  LDBG_EXECUTE ( lxr->Write ( "fsts/aplats/aplats+flower+p.fst.gz" ) );
101  fst::RmEpsilon<TupleArc32> ( lxr );
102  fst::Determinize<TupleArc32> ( *lxr, &myfst_ );
103  LDBG_EXECUTE ( myfst_.Write ( "fsts/aplats/aplats+flower+p+re+d.fst" ) );
104  delete lxr;
105  fst::Minimize<TupleArc32> ( &myfst_ );
106  LDBG_EXECUTE ( myfst_.Write ( "fsts/aplats/aplats+flower+p+re+d+m.fst" ) );
107  d.fsts[sparseweightvectorlatticekey_] = &myfst_;
108  LINFO ( "Ready! Stored with key=" << sparseweightvectorlatticekey_ << "; NS=" <<
109  myfst_.NumStates() );
110  return false;
111  };
112 
113  ~SparseWeightVectorLatticesTask() {

116  }

121  fst::VectorFst<TupleArc32> * applyFlowerLattice ( const fst::VectorFst<Arc>
122  & hypfst,
123  const fst::VectorFst<TupleArc32>& grammarflowerlattice ) {
124  LINFO ( "Removing Weights of Target Lattice" );
125  fst::VectorFst<Arc> prvfst ( hypfst );
126  fst::Map ( &prvfst, fst::RmWeightMapper<Arc>() );
127  fst::VectorFst<TupleArc32> *vwfst = new fst::VectorFst<TupleArc32>;
128  LINFO ( "Mapping Arc Target Lattice to TupleArc32" );
129  fst::MakeSparseVectorWeight<Arc> mwcopy ( 1 );
130  fst::Map<Arc> ( prvfst, vwfst,
131  fst::GenericWeightMapper<Arc, TupleArc32, fst::MakeSparseVectorWeight<Arc> >
132  ( mwcopy ) );
133  fst::VectorFst<TupleArc32> *output = new fst::VectorFst<TupleArc32>;
134  LDBG_EXECUTE ( vwfst->Write ( "fsts/aplats/vw-beforecomposing.fst" ) );
135  LINFO ( "Compose (TupleArc32)" );
136  fst::Compose ( *vwfst, grammarflowerlattice, output );
137  delete vwfst;
138  return output;
139  };
