74 TEST (range, floatrange) {

76  std::vector <float> result;

78  uu::NumberRange<float> ir (x);
79  for ( ir.start(); !ir.done(); ir.next() ) {
80  result.push_back ( ir.get() );
81  }
82  ASSERT_EQ ( result.size(), 2 );
