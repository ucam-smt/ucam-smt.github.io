28 // Applies scale factor to grammar at each cost in the arc.
29 template <class Arc>
30 inline void SetGsf ( fst::VectorFst<Arc> *grmfst, const float gsf ) {
31  if ( gsf == 1.0f ) return;
32  fst::ScaleWeight<Arc> sw ( gsf );
33  fst::Map<Arc> ( grmfst,
34  fst::GenericWeightAutoMapper<Arc, fst::ScaleWeight<Arc> > ( sw ) );
35 };

38 template < class Arc >
39 inline void loadflowerfst ( ucam::util::iszfstream& umf,
40  fst::VectorFst<Arc>& flowerlattice ) {
41  std::string line;
42  typedef typename Arc::Weight Weight;
43  flowerlattice.AddState();
44  flowerlattice.SetStart ( 0 );
45  flowerlattice.SetFinal ( 0, Weight::One() );
46  fst::MakeWeight<Arc> mw;
47  while ( umf.getline ( line ) ) {
48  using ucam::util::toNumber;
49  std::vector<std::string> aux;
50  boost::algorithm::split ( aux, line, boost::algorithm::is_any_of ( " " ) );
51  USER_CHECK ( aux.size() % 2, "Wrong unimap input file format" );
52  if ( aux[0] == "<unk>" ) continue; //skip this line
53  if ( aux[0] == "<s>" ) aux[0] = "1";
54  if ( aux[1] == "<s>" ) aux[1] = "1";
55  if ( aux[0] == "</s>" ) aux[0] = "2";
56  if ( aux[1] == "</s>" ) aux[1] = "2";
57  for ( unsigned k = 1; k < aux.size(); k += 2 ) {
58  float w = -std::log ( toNumber<float> ( aux[k + 1] ) ) ;
59  if (w != std::numeric_limits<float>::infinity() )
60  flowerlattice.AddArc ( 0, Arc ( toNumber<unsigned> ( aux[0] ),
61  toNumber<unsigned> ( aux[k] ), mw ( w ), 0 ) );
62  else
63  LWARN ("Skipping 0 probability at line:" << line);
64  }
65  }
66  flowerlattice.AddArc ( 0, Arc ( RHO, RHO, Weight::One(), 0 ) );
67  flowerlattice.AddArc ( 0, Arc ( OOV, OOV, Weight::One(), 0 ) ); //for OOVs...
68  ArcSort ( &flowerlattice, fst::ILabelCompare<Arc>() );
69 };

73 template<class Arc>
74 inline void tagOOVs ( fst::VectorFst<Arc> *myfst,
75  unordered_set<std::string>& vcb ) {
76  fst::MakeWeight<Arc> mw;
77  typedef typename Arc::StateId StateId;
78  for ( fst::StateIterator< fst::VectorFst<Arc> > si ( *myfst ); !si.Done();
79  si.Next() ) {
80  StateId state_id = si.Value();
81  for ( fst::MutableArcIterator< fst::MutableFst<Arc> > ai ( myfst, si.Value() );
82  !ai.Done(); ai.Next() ) {
83  Arc arc = ai.Value();
84  if ( vcb.find ( ucam::util::toString<unsigned> ( arc.olabel ) ) == vcb.end() ) {
85  arc.ilabel = arc.olabel;
86  arc.olabel = OOV;
87  arc.weight = mw ( 0 );
88  ai.SetValue ( arc );
89  }
90  }
91  }
92 };

96 template<class Arc>
97 inline void recoverOOVs ( fst::VectorFst<Arc> *myfst ) {
98  typedef typename Arc::StateId StateId;
99  for ( fst::StateIterator< fst::VectorFst<Arc> > si ( *myfst ); !si.Done();
100  si.Next() ) {
101  StateId state_id = si.Value();
102  for ( fst::MutableArcIterator< fst::MutableFst<Arc> > ai ( myfst, si.Value() );
103  !ai.Done(); ai.Next() ) {
104  Arc arc = ai.Value();
105  if ( arc.olabel == OOV ) {
106  arc.olabel = arc.ilabel;
107  ai.SetValue ( arc );
108  }
109  }
110  }
