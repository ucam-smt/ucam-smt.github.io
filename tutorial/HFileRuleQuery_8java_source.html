98  private void drainQueue() throws IOException {
99  s2tClient.queryRules(queue);
100  t2sClient.queryRules(queue);
101  for (Pair<RuleWritable, AlignmentAndFeatureMap> entry : queue) {
102  RuleWritable rule = entry.getFirst();
103  AlignmentAndFeatureMap rawFeatures = entry.getSecond();
104  if (retriever.asciiConstraints.contains(rule)) {
105  RuleWritable asciiRule = new RuleWritable(rule);
106  asciiRule.setLeftHandSide(new Text(
107  EnumRuleType.ASCII_OOV_DELETE.getLhs()));
108  synchronized (retriever.foundAsciiConstraints) {
109  retriever.foundAsciiConstraints.add(asciiRule);
110  }
111  features.writeRule(rule, rawFeatures,
112  EnumRuleType.ASCII_OOV_DELETE, out);
113  } else {
114  features.writeRule(rule, rawFeatures,
115  EnumRuleType.EXTRACTED, out);
116  }
117 
118  }
119  queue.clear();
120  }

122  @SuppressWarnings("unchecked")
123  @Override
124  public void run() {
125  List<Text> sortedQuery = new ArrayList<>(query);
126  query.clear();
127  StopWatch stopWatch = new StopWatch();
128  System.out.println("Sorting query");
129  stopWatch.start();
130  Collections.sort(sortedQuery, new MergeComparator());
131  System.out.printf("Query sort took %d seconds\n",
132  stopWatch.getTime() / 1000);
133  stopWatch.reset();
134  stopWatch.start();
135  try {
136  for (Text source : sortedQuery) {
137  SidePattern sourcePattern = SidePattern.getPattern(source
138  .toString());
139 
140  tempOut.reset();
141  source.write(tempOut);
142  if (!bf.contains(tempOut.getData(), 0, tempOut.getLength(),
143  null)) {
144  continue;
145  }
146  if (reader.seek(source)) {
147  if (retriever.testVocab.contains(source)) {
148  synchronized (retriever.foundTestVocab) {
149  retriever.foundTestVocab.add(source);
150  }
151  }
152  Set<RuleWritable> existingRules = new HashSet<>();
153  List<Pair<RuleWritable, AlignmentAndFeatureMap>> allFiltered = new ArrayList<>();
154  List<String> provenances = new ArrayList<>();
155  provenances.add("");
156  provenances.addAll(conf
157  .getStringCollection(ProvenanceCountMap.PROV));
158  for (String provenance : provenances) {
159  if (!retriever.filter.isProvenanceUnion()
160  && !provenance.equals("")) {
161  continue;
162  }
163  SortedSet<Pair<RuleWritable, AlignmentAndFeatureMap>> rules = new TreeSet<Pair<RuleWritable, AlignmentAndFeatureMap>>(
164  retriever.filter.getComparator(provenance));
165  for (Pair<RuleWritable, AlignmentAndFeatureMap> entry : reader
166  .getRulesForSource()) {
167  RuleWritable rule = entry.getFirst();
168  AlignmentAndFeatureMap rawFeatures = entry
169  .getSecond();
170  if (retriever.filter.filterRule(sourcePattern,
171  rule, rawFeatures.getSecond(),
172  provenance)) {
173  continue;
174  }
175  rules.add(Pair.createPair(new RuleWritable(rule),
176  rawFeatures));
177  }
178  List<Pair<RuleWritable, AlignmentAndFeatureMap>> filtered = retriever.filter
179  .filterRulesBySource(sourcePattern, rules,
180  provenance);
181  for (Pair<RuleWritable, AlignmentAndFeatureMap> ruleFiltered : filtered) {
182  if (!existingRules
183  .contains(ruleFiltered.getFirst())) {
184  allFiltered.add(ruleFiltered);
185  existingRules.add(ruleFiltered.getFirst());
186  }
187  }
188  }
189  queue.addAll(allFiltered);
190  if (queue.size() > BATCH_SIZE) {
191  drainQueue();
192  }
193 
194  }
195 
196  }
197  drainQueue();
198  } catch (IOException e) {
199  e.printStackTrace();
200  System.exit(1);
201  }
202  System.out
203  .printf("Query took %d seconds\n", stopWatch.getTime() / 1000);
204 
205  }
