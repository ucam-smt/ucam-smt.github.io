37 void setScales ( const ucam::util::RegistryPO& rg,
38  unsigned *offset,
39  const std::string& lmscales = HifstConstants::kLmFeatureweights,
40  const std::string& grammarscales =
41  HifstConstants::kRuleflowerlatticeFeatureweights,
42  const std::string& featureweights = HifstConstants::kFeatureweights,
43  const std::string& lmload = HifstConstants::kLmLoad
44  ) {
45  if (rg.get<std::string> (featureweights) != "" ) {
46  // overrides separate grammar feature weights + lm feature weights
47  fst::TropicalSparseTupleWeight<float>::Params()
48  = ucam::util::ParseParamString<float> (rg.getString (featureweights) );
49  *offset = rg.getVectorString (lmload).size();
50  LWARN ( HifstConstants::kFeatureweights << " overrides program options " <<
51  HifstConstants::kRuleflowerlatticeFeatureweights << " and " <<
52  HifstConstants::kLmFeatureweights);
53  return;
54  }
55  std::vector<float> fscales1, fscales2;
56  if ( rg.exists ( grammarscales ) )
57  if ( rg.get<std::string> ( grammarscales ) != "" )
58  fscales2 = ucam::util::ParseParamString<float> ( rg.getString (
59  grammarscales ) );
60  if ( rg.exists ( lmscales ) )
61  if ( rg.get<std::string> ( lmscales ) != "" )
62  fscales1 = ucam::util::ParseParamString<float> ( rg.getString ( lmscales ) );
63  std::vector<float>& fscales = fst::TropicalSparseTupleWeight<float>::Params();
64  *offset = fscales1.size();
65  if ( fscales1.size() + fscales2.size() ) {
66  LWARN ( "env parameter is overriden by " << lmscales << "," << grammarscales );
67  fscales.clear();
68  fscales = fscales1;
69  copy ( fscales2.begin(), fscales2.end(), std::back_inserter ( fscales ) );
70  }
71  USER_CHECK ( fscales.size(),
72  "Number of scaling factors must be greater than 0" );
74  std::string x = ucam::util::toString<float> ( fscales[0] );
75  for ( unsigned k = 1; k < fscales.size(); ++k )
76  x += "," + ucam::util::toString<float> ( fscales[k] );
77  LINFO ( "Number of language models =" << *offset << ". Scales=" << x );
78 };

84 template <class Data = AlilatsToSparseWeightLatsData<lm::ngram::Model>
85  , class KenLMModelT = lm::ngram::Model >
86 class SingleThreadedAliLatsToSparseVecLatsTask: public
87  ucam::util::TaskInterface<Data> {
88  private:
89  typedef ucam::fsttools::LoadWordMapTask< Data > LoadWordMap;
90  typedef ucam::fsttools::WriteFstTask < Data , TupleArc32 > WriteFst;
91  typedef ucam::fsttools::ReadFstTask < Data , fst::LexStdArc > ReadFst;
92  typedef ucam::fsttools::LoadLanguageModelTask < Data, KenLMModelT >
93  LoadLanguageModel;
94  typedef LoadSparseWeightFlowerLatticeTask < Data >
95  LoadSparseWeightFlowerLattice;
96  typedef SparseWeightVectorLatticesTask <Data, fst::LexStdArc >
97  SparseWeightVectorLattices;
98  typedef ucam::fsttools::ApplyLanguageModelTask<Data, KenLMModelT, TupleArc32 >
99  ApplyLanguageModel;
100  typedef DumpNbestFeaturesTask < Data > DumpNbestFeatures;
101 
102  const ucam::util::RegistryPO& rg_;
103  public:
108  SingleThreadedAliLatsToSparseVecLatsTask ( const ucam::util::RegistryPO& rg ) :
109  rg_ ( rg ) {
110  };
111 
115  bool run ( Data& d ) {
116  unsigned numlms;
117  setScales ( rg_ , &numlms);
118  boost::scoped_ptr < LoadSparseWeightFlowerLattice>
119  mytask (new LoadSparseWeightFlowerLattice ( rg_ , numlms,
120  HifstConstants::kSparseweightvectorlatticeLoadalilats ) );
121  mytask->appendTask
122  ( WriteFst::init ( rg_, HifstConstants::kRuleflowerlatticeStore ) )
123  ( LoadWordMap::init ( rg_, HifstConstants::kLmWordmap, true ) )
124  ( new LoadLanguageModel ( rg_, HifstConstants::kLmLoad,
125  "" ) ) //Here, language model weights always to 1
126  ( new ReadFst ( rg_
127  , HifstConstants::kSparseweightvectorlatticeLoadalilats ) )
128  ( new SparseWeightVectorLattices ( rg_
129  , HifstConstants::kSparseweightvectorlatticeLoadalilats
130  , HifstConstants::kRuleflowerlatticeStore
131  , HifstConstants::kSparseweightvectorlatticeStorenolm ) )
132  ( new ApplyLanguageModel ( rg_
133  , HifstConstants::kLmLoad
134  , HifstConstants::kSparseweightvectorlatticeStorenolm
135  , HifstConstants::kSparseweightvectorlatticeStore ) )
136  ( WriteFst::init ( rg_
137  , HifstConstants::kSparseweightvectorlatticeStore ) )
138  ( LoadWordMap::init ( rg_
139  , HifstConstants::kSparseweightvectorlatticeWordmap ) )
140  ( DumpNbestFeatures::init ( rg_
141  , numlms
142  , HifstConstants::kSparseweightvectorlatticeStore ) )
143  ;
144  for ( ucam::util::IntRangePtr ir (ucam::util::IntRangeFactory ( rg_ ,
145  HifstConstants::kRangeOne ) );
146  !ir->done ();
147  ir->next () ) {
148  d.sidx = ir->get ();
149  mytask->chainrun ( d ); // Run!
150  }
151  return false;
152  };
153 
154  inline bool operator() () {
155  Data d;
156  return run ( d );
157  };
158 
159  private:
160

168 template <class Data = AlilatsToSparseWeightLatsData<lm::ngram::Model> , class KenLMModelT = lm::ngram::Model >
169 class MultiThreadedAliLatsToSparseVecLatsTask: public
170  ucam::util::TaskInterface<Data> {
171 
172  private:
173  typedef ucam::fsttools::LoadWordMapTask< Data > LoadWordMap;
174  typedef ucam::fsttools::WriteFstTask < Data , TupleArc32 > WriteFst;
175  typedef ucam::fsttools::ReadFstTask < Data , fst::LexStdArc > ReadFst;
176  typedef ucam::fsttools::LoadLanguageModelTask < Data, KenLMModelT >
177  LoadLanguageModel;
178  typedef LoadSparseWeightFlowerLatticeTask < Data >
179  LoadSparseWeightFlowerLattice;
180  typedef SparseWeightVectorLatticesTask <Data, fst::LexStdArc >
181  SparseWeightVectorLattices;
182  typedef ucam::fsttools::ApplyLanguageModelTask<Data, KenLMModelT, TupleArc32 >
183  ApplyLanguageModel;
184  typedef DumpNbestFeaturesTask < Data > DumpNbestFeatures;
185 
187  const ucam::util::RegistryPO& rg_;
189  unsigned threadcount_;
190  public:
191  MultiThreadedAliLatsToSparseVecLatsTask ( const ucam::util::RegistryPO& rg ) :
192  threadcount_ ( rg.get<unsigned> ( HifstConstants::kNThreads ) ),
193  rg_ (rg) {
194  }
195 
196  bool run ( Data& original_data ) {
197  unsigned numlms;
198  setScales ( rg_ , &numlms);
199  boost::scoped_ptr < LoadSparseWeightFlowerLattice>
200  loadtask ( new LoadSparseWeightFlowerLattice ( rg_
201  , numlms
202  , HifstConstants::kSparseweightvectorlatticeLoadalilats ) );
203  loadtask->appendTask
204  ( WriteFst::init ( rg_ , HifstConstants::kRuleflowerlatticeStore ) )
205  ( new LoadLanguageModel ( rg_ , HifstConstants::kLmLoad,
206  "" ) ) //Forcing language model scales always to 1
207  ( LoadWordMap::init ( rg_ , HifstConstants::kLmWordmap , true ) )
208  ( LoadWordMap::init ( rg_ ,
209  HifstConstants::kSparseweightvectorlatticeWordmap ) )
210  ;
211  loadtask->chainrun ( original_data ); // Load grammar and language model;
212  {
213  ucam::util::TrivialThreadPool tp ( threadcount_ );
214  bool finished = false;
215  for ( ucam::util::IntRangePtr ir (ucam::util::IntRangeFactory ( rg_ ,
216  HifstConstants::kRangeOne ) );
217  !ir->done ();
218  ir->next () ) {
219  Data *d = new Data; //( original_data ); // reset.
220  d->sidx = ir->get();
221  d->klm = original_data.klm;
222  d->fsts = original_data.fsts;
223  d->wm = original_data.wm;
224  FORCELINFO ( "=====Extract features for sentence " << d->sidx << ":" );
225  ReadFst *runtask = new ReadFst ( rg_ ,
226  HifstConstants::kSparseweightvectorlatticeLoadalilats );
227  runtask->appendTask
228  ( new SparseWeightVectorLattices ( rg_
229  , HifstConstants::kSparseweightvectorlatticeLoadalilats
230  , HifstConstants::kRuleflowerlatticeStore
231  , HifstConstants::kSparseweightvectorlatticeStorenolm ) )
232  ( new ApplyLanguageModel ( rg_
233  , HifstConstants::kLmLoad
234  , HifstConstants::kSparseweightvectorlatticeStorenolm
235  , HifstConstants::kSparseweightvectorlatticeStore ) )
236  ( WriteFst::init ( rg_ , HifstConstants::kSparseweightvectorlatticeStore ) )
237  ( DumpNbestFeatures::init ( rg_ , numlms,
238  HifstConstants::kSparseweightvectorlatticeStore ) )
239  ;
240  tp ( ucam::util::TaskFunctor<Data> ( runtask,
241  d ) ); //tp takes ownership of runtask and d
242  }
243  }
244  return false;
245  };
246 
247  inline bool operator() () {
248  Data d;
249  return run ( d );
250  };
251 
252  private:
253  DISALLOW_COPY_AND_ASSIGN ( MultiThreadedAliLatsToSparseVecLatsTask );
254 };
