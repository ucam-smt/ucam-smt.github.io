79  private static class SwappingMapper
80  extends
81  Mapper<RuleWritable, RuleInfoWritable, RuleWritable, ProvenanceCountMap> {
82 
83  @Override
84  protected void map(RuleWritable key, RuleInfoWritable value,
85  Context context) throws IOException, InterruptedException {
86  RuleWritable newKey = key;
87  Rule r = new Rule(key);
88  if (r.isSwapping()) {
89  newKey = new RuleWritable(r.invertNonTerminals());
90  }
91  context.write(newKey, value.getProvenanceCountMap());
92  }
93  }

95  public static Job getJob(Configuration conf) throws IOException {
96  conf.set("mapred.map.child.java.opts", "-Xmx200m");
97  conf.set("mapred.reduce.child.java.opts", "-Xmx5128m");
98  conf.setBoolean(MarginalReducer.SOURCE_TO_TARGET, false);
99  Job job = new Job(conf);
100  job.setJarByClass(Target2SourceJob.class);
101  job.setJobName("Target2Source");
102  job.setSortComparatorClass(Target2SourceComparator.class);
103  job.setPartitionerClass(Target2SourcePartitioner.class);
104  job.setMapperClass(SwappingMapper.class);
105  job.setReducerClass(MarginalReducer.class);
106  job.setMapOutputKeyClass(RuleWritable.class);
107  job.setMapOutputValueClass(ProvenanceCountMap.class);
108  job.setOutputKeyClass(RuleWritable.class);
109  job.setOutputValueClass(FeatureMap.class);
110  job.setInputFormatClass(SequenceFileInputFormat.class);
111  job.setOutputFormatClass(SequenceFileOutputFormat.class);
112  return job;
