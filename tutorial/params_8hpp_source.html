32 template<typename T>
33 inline std::vector<T> ParseParamString ( const std::string& stringparams ,
34  size_t pos = 0 ) {
35  std::vector<T> result;
36  std::stringstream strm ( std::stringstream::in | std::stringstream::out );
37  if ( pos ) strm << stringparams.substr ( pos ) << std::noskipws;
38  else strm << stringparams << std::noskipws;
39  char separator;
40  while ( strm.good() ) {
41  if ( result.size() > 0 ) {
42  strm >> separator;
43  }
44  T w;
45  strm >> w;
46  result.push_back ( w );
47  }
48  if ( strm.fail() || strm.bad() || strm.fail() ) {
49  std::cerr << "ERROR: Unable to parse params : " << stringparams.substr (
50  pos ) << std::endl;
51  for ( uint k = 0; k < result.size(); ++k )
52  std::cerr << result[k] << std::endl;
53  exit ( 1 );
54  }
55  return result;
56 }
