147  private void findSplits(byte[] b, int s, int l,
148  RuleWritableSplit split, DataInputBuffer inBytes) {
149 
150  try {
151  split.lhsStart = WritableUtils.decodeVIntSize(b[s]) + s;
152  inBytes.reset(b, s, l);
153  split.lhsLength = WritableUtils.readVInt(inBytes);
154  int sourceN = split.lhsLength + split.lhsStart;
155  split.sourceStart = WritableUtils.decodeVIntSize(b[sourceN])
156  + sourceN;
157  inBytes.reset(b, sourceN, l - (sourceN - s));
158  split.sourceLength = WritableUtils.readVInt(inBytes);
159  int targetN = split.sourceStart + split.sourceLength;
160  split.targetStart = WritableUtils.decodeVIntSize(b[targetN])
161  + targetN;
162  split.targetLength = l - (split.targetStart - s);
163  } catch (IOException e) {

270  private void marginalReduce(Iterable<RuleCount> rules,
271  ProvenanceCountMap totals, Context context) throws IOException,
272  InterruptedException {
273  for (RuleCount rw : rules) {
274  features.clear();
275  for (Entry<ByteWritable, IntWritable> entry : rw.counts.entrySet()) {
276  double probability = (double) entry.getValue().get()
277  / (double) totals.get(entry.getKey()).get();
278  int featureIndex = mappings[(int) entry.getKey().get()];
279  features.put(featureIndex, probability);
280  ++featureIndex;
281  features.put(featureIndex, entry.getValue().get());
282  }
283  RuleWritable outKey = rw.rule;
284  if (!source2Target) {
285  Rule r = new Rule(outKey);
286  if (r.isSwapping()) {
287  r.invertNonTerminals();
288  outKey = new RuleWritable(r.invertNonTerminals());
289  }
290  }
291  context.write(outKey, features);
292  }

296  @Override
297  public void run(Context context) throws IOException, InterruptedException {
298  setup(context);
299 
300  while (context.nextKey()) {
301  RuleWritable key = context.getCurrentKey();
302  // First Key!
303  if (marginal.getLength() == 0) {
304  marginal.set(getMarginal(key));
305  }
306  Iterator<ProvenanceCountMap> it = context.getValues().iterator();
307  ProvenanceCountMap counts = it.next();
308 
309  if (it.hasNext()) {
310  throw new RuntimeException("Non-unique rule! " + key);
311  }
312  if (!marginal.equals(getMarginal(key))) {
313  marginalReduce(ruleCounts, totals, context);
314  totals.clear();
315  ruleCounts.clear();
316  }
317  marginal.set(getMarginal(key));
318  ruleCounts.add(new RuleCount(new RuleWritable(key),
319  new ProvenanceCountMap(counts)));
320  totals.increment(counts);
321  }
322  marginalReduce(ruleCounts, totals, context);
323  cleanup(context);
324  }
