25 BleuStats::Error BleuStats::ComputeError (const unsigned int order) const {
26  double logBleu = 0.0;
27  double logBrev = 0.0;
28  for (unsigned int n = 0; n < order; ++n) {
29  logBleu += log ( (double) this->hits[n] / (double) this->tots[n]);
30  }
31  logBleu *= 1 / (double) BleuStats::MAX_BLEU_ORDER;
32  logBrev = min (0.0, 1 - this->refLength / (double) (this->tots[0]) );
33  return Error (exp (logBleu + logBrev), exp (logBrev) );
34 }
