70  bool run ( Data& d ) {
71  mylmfst_.DeleteStates();
72  if ( !USER_CHECK ( d.klm.size() ,
73  "No language models available" ) ) return true;
74  if ( !USER_CHECK ( d.klm.find ( lmkey_ ) != d.klm.end() ,
75  "No language models available (key not initialized) " ) ) return true;
76  if ( !USER_CHECK ( d.fsts.find ( latticeloadkey_ ) != d.fsts.end() ,
77  " Input fst not available!" ) ) return true;
78  mylmfst_ = * (static_cast<fst::VectorFst<Arc> * > ( d.fsts[latticeloadkey_] ) );
79  if (deletelmscores_) {
80  LINFO ( "Delete old LM scores first" );
81  //Deletes LM scores if using lexstdarc. Note -- will copy through on stdarc and ignore on tuplearc!
82  fst::MakeWeight2<Arc> mwcopy;
83  fst::Map<Arc> ( &mylmfst_,
84  fst::GenericWeightAutoMapper<Arc, fst::MakeWeight2<Arc> > ( mwcopy ) );
85  }
86  LINFO ( "Input lattice loaded with key=" << latticeloadkey_ << ", NS=" <<
87  mylmfst_.NumStates() );
88  fst::MakeWeight<Arc> mw;
89  for ( uint k = 0; k < d.klm[lmkey_].size(); ++k ) {
90  if ( !USER_CHECK ( d.klm[lmkey_][k]->model != NULL,
91  "Language model " << k << " not available!" ) ) return true;
92  KenLMModelT& model = *d.klm[lmkey_][k]->model;
93 #ifndef USE_GOOGLE_SPARSE_HASH
94  unordered_set<Label> epsilons;
95 #else
96  google::dense_hash_set<Label> epsilons;
97  epsilons.set_empty_key ( numeric_limits<Label>::max() );
98 #endif
99  epsilons.insert ( DR );
101  epsilons.insert ( OOV );
102  epsilons.insert ( EPSILON );
103  epsilons.insert ( SEP );
104  LINFO ( "Applying language model " << k
105  << " with lmkey=" << lmkey_
106  << ", using lmscale=" << d.klm[lmkey_][k]->lmscale );
107  LDEBUG ( "lattice NS=" << mylmfst_.NumStates() );
108  fst::ApplyLanguageModelOnTheFly<Arc, fst::MakeWeight<Arc>, KenLMModelT> *f
109  = new fst::ApplyLanguageModelOnTheFly<Arc, fst::MakeWeight<Arc>, KenLMModelT >
110  ( mylmfst_
111  , model
112  , epsilons
113  , natlog_
114  , d.klm[lmkey_][k]->lmscale
115  , d.klm[lmkey_][k]->lmwp
116  , d.klm[lmkey_][k]->idb);
117  f->setMakeWeight ( mw );
118  d.stats->setTimeStart ( "on-the-fly-composition " + ucam::util::toString (
119  k ) );
120  mylmfst_ = * ( ( *f ) () );
121  delete f;
122  d.stats->setTimeEnd ( "on-the-fly-composition " + ucam::util::toString ( k ) );
123  LDEBUG ( mylmfst_.NumStates() );
124  mw.update();
125  }
126  d.fsts[latticestorekey_] = &mylmfst_;
127  LINFO ( "Done!" );
128  return false;
129  };
