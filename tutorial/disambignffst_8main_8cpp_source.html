50  void operator()() {
51  using namespace fst;
52  using ucam::util::toString;
53  using namespace ucam::fsttools;
54 
55  boost::scoped_ptr<fst::VectorFst<ArcT> >mfst
56  (fst::VectorFstRead<ArcT> ( in_ ) );
57  if (mfst->NumStates()) {
58  if (detOut_) // invert fst.
59  Invert(&*mfst);
60  RmEpsilon(&*mfst);
61  TopSort(&*mfst);
62 
63  std::string ns = toString(mfst->NumStates());
64  ssd_.setTimeStart("disambig-" + out_ + ", NS=" + ns);
65  if (!minimize_) {
66  FORCELINFO("Only determinize");
67  TopoFeaturesHelper<ProjectDeterminizeAction> tfh(exitOnFirstPassFailure_);
68  tfh(&*mfst);
69  } else { // Experimental option to play with.
70  FORCELINFO("Determinize, Minimize and Push!");
71  TopoFeaturesHelper<ProjectDeterminizeMinimizePushAction> tfh(exitOnFirstPassFailure_);
72  tfh(&*mfst);
73  }
74  RmEpsilon(&*mfst); // take out epsilons created by reversals etc.
75  ssd_.setTimeEnd("disambig-" + out_ + ", NS=" + ns);
76  std::string nsd = toString(mfst->NumStates());
77  FORCELINFO(out_ << ": NS=" << ns << ",NSD=" << nsd);
78  LDBG_EXECUTE(mfst->Write("08-det-topo-final-rm.fst"));
79  if (detOut_) // invert back
80  Invert(&*mfst);
81  } else {
82  ssd_.setTimeStart("disambig-" + out_ + ", NS=0");
83  ssd_.setTimeEnd("disambig-" + out_ + ", NS=0");
84  }
85  FstWrite<StdArc> (*mfst, out_ );
86  ssd_.write(std::cerr);
87  }
88 };

92 // loads, runs disambiguation and stores.
93 template<class ArcT, class ThreadPoolT >
94 inline void run(ucam::util::RegistryPO const &rg
95  , ThreadPoolT &tp) {
96  using namespace HifstConstants;
97  using namespace ucam::util;
98 
99  PatternAddress<unsigned> pi (rg.get<std::string>(kInput) );
100  PatternAddress<unsigned> po (rg.get<std::string>(kOutput) );
101  bool detOut = rg.getBool(kDeterminizeOutput);
102  bool minimize = rg.getBool(kMinimize);
103  bool exitOnFirstPassFailure = rg.getBool(kExitOnFirstPassFailure);
104  using namespace fst;
105  for ( IntRangePtr ir (IntRangeFactory ( rg, kRangeOne ) );
106  !ir->done();
107  ir->next() ) {
108  DisambigFunctor<ArcT> df(pi(ir->get()), po(ir->get())
109  , detOut, minimize, exitOnFirstPassFailure);
110  tp(df);
111  }
112 };

114 // Main function call this overloaded method (see main.hpp for details)
115 // Determine semiring, multithreading and kick off disambiguation of
116 // (potentially non-funcitonal) FSTs using topological features
117 void ucam::util::MainClass::run() {
118  using namespace HifstConstants;
119  using namespace ucam::util;
120  std::string const arctype =rg_->get<std::string>(kHifstSemiring);
121  unsigned nthreads = (rg_->exists( kNThreads) )
122  ? rg_->get<unsigned>(kNThreads) : 0;
123 
124  if (arctype == kHifstSemiringStdArc) {
125  if (nthreads) { // even the single thread is in the trivial pool
126  FORCELINFO("Multithreading with " << nthreads << " threads");
127  TrivialThreadPool tp(nthreads);
128  ::run<fst::StdArc, TrivialThreadPool>(*rg_, tp);
129  } else { // not defined, will run single thread without the trivial pool
130  NoThreadPool ntp;
131  ::run<fst::StdArc, NoThreadPool>(*rg_, ntp);
132  }
133  } else {
134  LERROR("Unknown semiring!");
135  exit(EXIT_FAILURE);
136  }
137 }
