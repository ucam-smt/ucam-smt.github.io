74 inline void checkCreateSSGrammarOptions (po::variables_map *vm) {
75  if ( vm->count ( HifstConstants::kGrammarLoad.c_str() ) ) {
76  LDEBUG ( HifstConstants::kGrammarLoad << "=" <<
77  ( *vm ) [HifstConstants::kGrammarLoad.c_str()].as<std::string>() );
78  } else {
79  LERROR ( HifstConstants::kGrammarLoad << " not defined" );
80  exit ( EXIT_FAILURE );
81  }
82  if ( vm->count ( HifstConstants::kSourceLoad.c_str() ) ) {
83  LDEBUG ( HifstConstants::kSourceLoad << "=" <<
84  ( *vm ) [HifstConstants::kSourceLoad.c_str()].as<std::string>() );
85  } else if ( vm->count ( HifstConstants::kServerEnable.c_str() ) ) {
86  LERROR ( HifstConstants::kSourceLoad <<
87  " not defined, mandatory unless running on server mode " );
