27 void run (ucam::util::RegistryPO const& rg) {
28  using ucam::fsttools::DisambigData;
29  using ucam::fsttools::DisambigTask;
30  using ucam::fsttools::ReadFstTask;
31  using ucam::fsttools::WriteFstTask;
32  using ucam::fsttools::LoadUnimapTask;
33  using ucam::fsttools::LoadLanguageModelTask;
34  DisambigData data;
35  //Our pipeline is defined by the following tasks:
36  //Load Recasing language model
37  boost::scoped_ptr< LoadLanguageModelTask<DisambigData > > mytask (
38  new LoadLanguageModelTask<DisambigData> ( rg ,
39  HifstConstants::kRecaserLmLoad , HifstConstants::kRecaserLmFeatureweight ) );
40  mytask->appendTask
41  //Load unigram transduction model
42  ( new LoadUnimapTask< DisambigData, ArcT > ( rg ,
43  HifstConstants::kRecaserUnimapLoad, HifstConstants::kRecaserLmLoad ) )
44  //Read input lattice
45  ( new ReadFstTask<DisambigData, ArcT > ( rg ,
46  HifstConstants::kRecaserInput ) )
47  //Apply both models and prune
48  ( new DisambigTask<DisambigData, ArcT > ( rg ,
49  HifstConstants::kRecaserInput , HifstConstants::kRecaserOutput ,
50  HifstConstants::kRecaserLmLoad , HifstConstants::kRecaserUnimapLoad ) )
51  //Write output lattice
52  ( new WriteFstTask<DisambigData, ArcT > ( rg ,
53  HifstConstants::kRecaserOutput ) )
54  ;
55  //For a given range of lattices, proceed to run the task
56  for ( ucam::util::IntRangePtr ir (ucam::util::IntRangeFactory ( rg,
57  HifstConstants::kRangeOne ) )
58  ; !ir->done()
59  ; ir->next() ) {
60  data.sidx = ir->get();
61  mytask->chainrun ( data );
62  }
63 }
