59  while ((line = br.readLine()) != null) {
60  String[] parts = line.split(" ");
61  if (parts.length != 4) {
62  throw new RuntimeException("Malformed rule: " + line);
63  }
64  int lhs = Integer.parseInt(parts[0]);
65  String[] sourceWords = parts[1].split("_");
66  String[] targetWords = parts[2].split("_");
67  List<Integer> sourceIds = new ArrayList<>();
68  List<Integer> targetIds = new ArrayList<>();
69  for (String sourceWord : sourceWords) {
70  sourceIds.add(Integer.parseInt(sourceWord));
71  }
72  for (String targetWord : targetWords) {
73  targetIds.add(Integer.parseInt(targetWord));
74  }
75  String[] alignmentLinks = parts[3].split("_");
76  List<AlignmentLink> alignment = new ArrayList<>();
77  for (String alignmentLink : alignmentLinks) {
78  String[] srcTrg = alignmentLink.split("-");
79  if (srcTrg.length != 2) {
80  throw new RuntimeException(
81  "Malformed rule (alignment): " + line);
82  }
83  alignment.add(new AlignmentLink(
84  Integer.parseInt(srcTrg[0]),
85  Integer.parseInt(srcTrg[1])));
86  }
87  Rule r = new Rule(lhs, sourceIds, targetIds);
88  if (!res.containsKey(r)) {
89  Map<List<AlignmentLink>, Integer> alignments =
90  new HashMap<>();
91  alignments.put(alignment, 1);
92  res.put(r, new Pair<>(1, alignments));
93  } else {
94  Pair<Integer, Map<List<AlignmentLink>, Integer>> countAlignments =
95  res.get(r);
96  countAlignments.setFirst(countAlignments.getFirst() + 1);
97  Map<List<AlignmentLink>, Integer> alignments =
98  countAlignments.getSecond();
99  if (!alignments.containsKey(alignment)) {
100  alignments.put(alignment, 1);
101  } else {
102  alignments.put(alignment, alignments.get(alignment) + 1);
103  }
104  }
105  }
106  }
107  return res;
108  }
