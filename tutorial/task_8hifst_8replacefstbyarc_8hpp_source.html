41  public:

55  inline fst::VectorFst<Arc> *operator() ( fst::VectorFst<Arc> const& fst,
56  Label const& hieroindex ) const {
57  if ( fst.NumStates() <= minns_ ) return NULL;
58  fst::VectorFst<Arc>* outfst = new fst::VectorFst<Arc>;
59  outfst->AddState();
60  outfst->SetStart ( 0 );
61  outfst->AddState();
62  outfst->SetFinal ( 1, Weight::One() );
63  if ( aligner_ ) outfst->AddArc ( 0, Arc ( NORULE, hieroindex, Weight::One() ,
64  1 ) );
65  else outfst->AddArc ( 0, Arc ( hieroindex, hieroindex, Weight::One() , 1 ) );
66  return outfst;
67  };

70  DISALLOW_COPY_AND_ASSIGN ( ReplaceFstByArc );
