242 // We need sparse tuple weight to do the same work as the lexicographic semiring
243 // this test simply shows the way these weights work.
244 // Although there are notes in the documentation suggesting
245 // that the list is sorted, this is not actually true.
246 // Also, indices can be repeated.
247 TEST(tropicalsparseweight, removeKthWeight) {
248  using namespace fst;
249  TupleArc32::Weight w,w2;
250  w.Push(6,6);
251  w.Push(5,5);
252  w.Push(8,0);
253  w.Push(5,-2);
254  w.SetDefaultValue(300);
255  std::stringstream ss; ss << w;
256  EXPECT_EQ ( ss.str(), "300,3,6,6,5,5,5,-2,");
257  // remove weight 5.
258 
259  // this does not copy default value
260  for (SparseTupleWeightIterator<StdArc::Weight, int> it(w); !it.Done(); it.Next()) {
261  if (it.Value().first != 5) // you cannot break;
262  w2.Push(it.Value());
263  }
264  std::stringstream ss2; ss2 << w2;
265  EXPECT_EQ ( ss2.str(), "0,1,6,6,");
266  w2.SetDefaultValue(w.DefaultValue());
267  std::stringstream ss3; ss3 << w2;
268  EXPECT_EQ ( ss3.str(), "300,1,6,6,");
269 
270  // This is implemented in SparseTupleWeightMap (sparse-tuple-weight.h)
271  // The default value is used for Times operation if not available.
272  TupleArc32::Weight w3=Times(w,w2);
273  // std::cerr << w3 << std::endl;
274  // 600,3,6,12,5,305,5,298
275 }
