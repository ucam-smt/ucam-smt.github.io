93 template <class Arc>
94 inline void FstWrite ( const Fst<Arc>& fst
95  , const std::string& filename
96  , const std::string& txtname = "txt" ) {
97  if ( filename == "/dev/null" ) return;
98  LDEBUG ("Started..." << filename);
99  //boost::scoped_ptr<oszfstream> file ( new oszfstream ( filename ) );
100  ucam::util::oszfstream file (filename);
101  if (!file.is_open() ) {
102  LERROR ("Error opening " << filename);
103  exit (EXIT_FAILURE);
104  }
105  if ( filename != "-"
106  && filename != "/dev/stdout"
107  && filename != "/dev/stderr"
108  && (ucam::util::ends_with ( filename, "." + txtname + ".gz" )
109  || ucam::util::ends_with ( filename, "." + txtname )
110  )
111  ) {
112  PrintFst<Arc> ( fst, file.getStream() );
113  file.close();
114  return;
115  }
116  fst.Write ( *file.getStream(), FstWriteOptions (filename) );
117  file.close();
118  LDEBUG ("Finished...");
119 };

148 template <class Arc, class HypT>
149 void printStrings (const VectorFst<Arc>& fst, std::vector<HypT>* hyps) {
150  typename Arc::StateId start = fst.Start();
151  for (ArcIterator<VectorFst<Arc> > ai (fst, start); !ai.Done(); ai.Next() ) {
152  std::basic_string<unsigned> hyp;
153  Arc a = ai.Value();
154  hyp.push_back (a.ilabel);
155  typename Arc::Weight w = a.weight;
156  typename Arc::StateId nextState = a.nextstate;
157  ArcIterator<VectorFst<Arc> >* ai2 =
158  new ArcIterator<VectorFst<Arc> > (fst, nextState);
159  while (!ai2->Done() ) {
160  Arc a2 = ai2->Value();
161  hyp.push_back (a2.ilabel);
162  w = Times (w, a2.weight);
163  nextState = a2.nextstate;
164  delete ai2;
165  ai2 = new ArcIterator<VectorFst<Arc> > (fst, nextState);
166  }
167  delete ai2;
168  hyps->push_back (HypT (hyp, w) );
169  }
170 }
