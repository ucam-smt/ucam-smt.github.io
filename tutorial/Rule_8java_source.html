198  private void buildAlignment(int sourceIndex, int sourceStartIndex,
199  int minTargetIndex, int maxTargetIndex, int minTargetIndexFirstNT,
200  int maxTargetIndexFirstNT, int minTargetIndexSecondNT,
201  int maxTargetIndexSecondNT, int srcXSpanMinusOne,
202  int srcX2SpanMinusOne, int firstTrgNTSpanMinusOne,
203  int secondTrgNTSpanMinusOne, StartMiddleEnd sourcePosition,
204  Alignment a) {
205  if (!a.isSourceAligned(sourceIndex)) {
206  return;
207  }
208  for (int trgIdx : a.getS2t().get(sourceIndex)) {
209  if (trgIdx < minTargetIndex
210  || trgIdx > maxTargetIndex
211  || (minTargetIndexFirstNT <= trgIdx && trgIdx <= maxTargetIndexFirstNT)
212  || (minTargetIndexSecondNT <= trgIdx && trgIdx <= maxTargetIndexSecondNT)) {
213  continue;
214  }
215  int ruleTrgIdx = -1;
216  int ruleSrcIdx = -1;
217  switch (sourcePosition) {
218  case START:
219  ruleSrcIdx = sourceIndex - sourceStartIndex;
220  break;
221  case MIDDLE:
222  ruleSrcIdx = sourceIndex - sourceStartIndex - srcXSpanMinusOne;
223  break;
224  case END:
225  ruleSrcIdx = sourceIndex - sourceStartIndex - srcXSpanMinusOne - srcX2SpanMinusOne;
226  break;
227  }
228  if (trgIdx < minTargetIndexFirstNT) {
229  ruleTrgIdx = trgIdx - minTargetIndex;
230  } else if (maxTargetIndexFirstNT < trgIdx && trgIdx < minTargetIndexSecondNT) {
231  ruleTrgIdx = trgIdx - minTargetIndex - firstTrgNTSpanMinusOne;
232  } else if (trgIdx > maxTargetIndexSecondNT) {
233  ruleTrgIdx = trgIdx - minTargetIndex - firstTrgNTSpanMinusOne
234  - secondTrgNTSpanMinusOne;
235  }
236  alignment.add(new AlignmentLink(ruleSrcIdx, ruleTrgIdx));
237  }
238  }
