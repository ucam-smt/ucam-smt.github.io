142  // TODO use Properties instead
143  private void loadConfig(String configFile) throws FileNotFoundException,
144  IOException {
145  try (BufferedReader br = new BufferedReader(new FileReader(configFile))) {
146  String line;
147  String[] parts;
148  while ((line = br.readLine()) != null) {
149  if (line.startsWith("#")) {
150  continue;
151  }
152  if (line.isEmpty()) {
153  continue;
154  }
155  parts = line.split("\\s+");
156  if (parts.length == 0) {
157  System.err.println("Malformed config file");
158  System.exit(1);
159  }
160  String[] featureValue = parts[0].split("=", 2);
161  if (featureValue[0].equals("min_source2target_phrase")) {
162  minSource2TargetPhrase = Double
163  .parseDouble(featureValue[1]);
164  } else if (featureValue[0].equals("min_target2source_phrase")) {
165  minTarget2SourcePhrase = Double
166  .parseDouble(featureValue[1]);
167  } else if (featureValue[0].equals("min_source2target_rule")) {
168  minSource2TargetRule = Double.parseDouble(featureValue[1]);
169  } else if (featureValue[0].equals("min_target2source_rule")) {
170  minTarget2SourceRule = Double.parseDouble(featureValue[1]);
171  } else if (featureValue[0].equals("allowed_source_pattern")) {
172  if (sourcePatternConstraints == null) {
173  sourcePatternConstraints = new HashMap<SidePattern, Map<String, Double>>();
174  }
175  Map<String, Double> constraints = new HashMap<String, Double>();
176  for (int i = 1; i < parts.length; i++) {
177  String[] constraintValue = parts[i].split("=", 2);
178  constraints.put(constraintValue[0],
179  Double.parseDouble(constraintValue[1]));
180  }
181  sourcePatternConstraints.put(
182  SidePattern.parsePattern(featureValue[1]),
183  constraints);
184  } else if (featureValue[0].equals("allowed_pattern")) {
185  if (allowedPatterns == null) {
186  allowedPatterns = new HashSet<RulePattern>();
187  }
188  allowedPatterns.add(RulePattern
189  .parsePattern(featureValue[1]));
190  } else if (featureValue[0].equals("skip_pattern")) {
191  if (skipPatterns == null) {
192  skipPatterns = new HashSet<>();
193  }
194  skipPatterns.add(RulePattern.parsePattern(featureValue[1]));
195  } else if (featureValue[0].equals("provenance_union")) {
196  provenanceUnion = Boolean.parseBoolean(featureValue[1]);
197  }
198  }
199  }

217  public List<Pair<RuleWritable, AlignmentAndFeatureMap>> filterRulesBySource(
218  SidePattern sourcePattern,
219  SortedSet<Pair<RuleWritable, AlignmentAndFeatureMap>> rules,
220  String provenance) {
221  List<Pair<RuleWritable, AlignmentAndFeatureMap>> results = new ArrayList<>();
222  int numberTranslations = 0;
223  int numberTranslationsMonotone = 0; // case with more than 1 NT
224  int numberTranslationsInvert = 0;
225  int prevCount = -1;
226  IntWritable countIndex = IntWritableCache.createIntWritable(s2tIndices
227  .get(provenance) + 1);
228  for (Pair<RuleWritable, AlignmentAndFeatureMap> entry : rules) {
229  // number of translations per source threshold
230  // in case of ties we either keep or don't keep the ties
231  // depending on the config
232 
233  RulePattern rulePattern = RulePattern.getPattern(entry.getFirst(),
234  entry.getFirst());
235  boolean doNotSkip = skipPatterns == null
236  || !skipPatterns.contains(rulePattern);
237  int count = (int) entry.getSecond().getSecond().get(countIndex)
238  .get();
239 
240  if (doNotSkip && !sourcePattern.isPhrase()) {
241  if (sourcePattern.hasMoreThan1NT()) {
242  if (sourcePatternConstraints.get(sourcePattern).get(
243  "ntrans") <= numberTranslationsMonotone
244  && sourcePatternConstraints.get(sourcePattern).get(
245  "ntrans") <= numberTranslationsInvert
246  && count != prevCount) {
247  break;
248  }
249  } else if (sourcePatternConstraints.get(sourcePattern).get(
250  "ntrans") <= numberTranslations
251  && count != prevCount) {
252  break;
253  }
254  }
255 
256  if (sourcePattern.isPhrase()) {
257  results.add(entry);
258  } else if (doNotSkip) {
259  if (sourcePattern.hasMoreThan1NT()) {
260 
261  if (rulePattern.isSwappingNT()
262  && (count == prevCount || sourcePatternConstraints
263  .get(sourcePattern).get("ntrans") > numberTranslationsInvert)) {
264  results.add(entry);
265  } else if (count == prevCount
266  || sourcePatternConstraints.get(sourcePattern).get(
267  "ntrans") > numberTranslationsMonotone) {
268  results.add(entry);
269  }
270  } else {
271  results.add(entry);
272  }
273  }
274  if (sourcePattern.hasMoreThan1NT()) {
275  if (rulePattern.isSwappingNT()) {
276  numberTranslationsInvert++;
277  } else {
278  numberTranslationsMonotone++;
279  }
280  }
281  numberTranslations++;
282  prevCount = count;
283 
284  }
285  return results;
286  }

299  public boolean filterRule(SidePattern sourcePattern, RuleWritable rule,
300  FeatureMap features, String provenance) {
301  int s2tIndex = s2tIndices.get(provenance);
302  IntWritable source2targetProbabilityIndex = IntWritableCache
303  .createIntWritable(s2tIndex);
304  // if the rule does not have that provenance, we automatically filter it
305  if (!features.containsKey(source2targetProbabilityIndex)) {
306  return true;
307  }
308  int t2sIndex = t2sIndices.get(provenance);
309  IntWritable target2sourceProbabilityIndex = IntWritableCache
310  .createIntWritable(t2sIndex);
311  IntWritable countIndex = IntWritableCache
312  .createIntWritable(s2tIndex + 1);
313 
314  double source2targetProbability = features.get(
315  source2targetProbabilityIndex).get();
316  double target2sourceProbability = features.get(
317  target2sourceProbabilityIndex).get();
318  int numberOfOccurrences = (int) features.get(countIndex).get();
319 
320  RulePattern rulePattern = RulePattern.getPattern(rule, rule);
321  if (!sourcePattern.isPhrase()
322  && !allowedPatterns.contains(rulePattern)
323  && (skipPatterns == null || !skipPatterns.contains(rulePattern))) {
324  return true;
325  }
326  if (sourcePattern.isPhrase()) {
327  // source-to-target threshold
328  if (source2targetProbability <= minSource2TargetPhrase) {
329  return true;
330  }
331  // target-to-source threshold
332  if (target2sourceProbability <= minTarget2SourcePhrase) {
333  return true;
334  }
335  } else {
336  // source-to-target threshold
337  if (source2targetProbability <= minSource2TargetRule) {
338  return true;
339  }
340  // target-to-source threshold
341  if (target2sourceProbability <= minTarget2SourceRule) {
342  return true;
343  }
344  // minimum number of occurrence threshold
345  if (sourcePatternConstraints.get(sourcePattern).containsKey("nocc")) {
346  if (numberOfOccurrences < sourcePatternConstraints.get(
347  sourcePattern).get("nocc")) {
348  return true;
349  }
350  }
351  }
352  return false;
353  }
