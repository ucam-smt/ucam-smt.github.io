32 template <class Data = ApplyLMData<lm::ngram::Model>
33  , class KenLMModelT = lm::ngram::Model
34  , class Arc = fst::StdArc >
35 class SingleThreadedApplyLanguageModelTask: public
36  ucam::util::TaskInterface<Data> {
37  private:
38  typedef LoadWordMapTask< Data > LoadWordMap;
39  typedef LoadLanguageModelTask < Data, KenLMModelT > LoadLanguageModel;
40  typedef ApplyLanguageModelTask<Data , KenLMModelT, Arc > ApplyLanguageModel;
41  typedef ReadFstTask<Data, Arc> ReadFst;
42  typedef WriteFstTask< Data , Arc> WriteFst;
43  typedef SpeedStatsTask<Data> SpeedStats;
44 
45  //Command-line/config file options
46  const ucam::util::RegistryPO& rg_;
47 
48  public:
53  SingleThreadedApplyLanguageModelTask ( const ucam::util::RegistryPO& rg ) :
54  rg_ ( rg ) {
55  };
56 
59  void operator() () {
60  Data d;
61  run ( d );
62  }
63 
69  bool run ( Data& original_data ) {
70  boost::scoped_ptr< LoadWordMap > tasks (new LoadWordMap (rg_,
71  HifstConstants::kLmWordmap, true) );
72  tasks->appendTask
73  ( new LoadLanguageModel ( rg_ ) )
74  ( new ReadFst ( rg_ , HifstConstants::kLatticeLoad ) )
75  ( new ApplyLanguageModel ( rg_
76  , HifstConstants::kLmLoad
77  , HifstConstants::kLatticeLoad
78  , HifstConstants::kLatticeStore
79  , rg_.exists (HifstConstants::kLatticeLoadDeleteLmCost) ) )
80  ( new WriteFst ( rg_ , HifstConstants::kLatticeStore ) )
81  ( new SpeedStats ( rg_ ) )
82  ;
83  Data d;
84  for ( ucam::util::IntRangePtr ir (ucam::util::IntRangeFactory ( rg_,
85  HifstConstants::kRangeOne ) )
86  ; !ir->done()
87  ; ir->next() ) {
88  d.sidx = ir->get();
89  LINFO ( "Running sentence " << d.sidx );
90  tasks->chainrun ( d ); //Run!
91  }
92  return false;
93  }
94 };

101 template <class Data = ApplyLMData<lm::ngram::Model>
102  , class KenLMModelT = lm::ngram::Model, class Arc = fst::StdArc >
103 class MultiThreadedApplyLanguageModelTask: public
104  ucam::util::TaskInterface<Data> {
105  private:
106  typedef LoadWordMapTask< Data > LoadWordMap;
107  typedef LoadLanguageModelTask < Data, KenLMModelT > LoadLanguageModel;
108  typedef ApplyLanguageModelTask<Data , KenLMModelT, Arc > ApplyLanguageModel;
109  typedef ReadFstTask<Data, Arc> ReadFst;
110  typedef WriteFstTask< Data , Arc> WriteFst;
111  typedef SpeedStatsTask<Data> SpeedStats;
112 
114  const ucam::util::RegistryPO& rg_;
116  uint threadcount_;
117  public:
122  MultiThreadedApplyLanguageModelTask ( const ucam::util::RegistryPO& rg ) :
123  threadcount_ ( rg.get<uint> ( HifstConstants::kNThreads ) ),
124  rg_ ( rg ) {
125  };
126 
128  inline bool operator() () {
129  Data d;
130  return run ( d );
131  }
132 
139  bool run ( Data& original_data ) {
140  boost::scoped_ptr< LoadWordMap > mylm (new LoadWordMap (rg_,
141  HifstConstants::kLmWordmap, true) );
142  mylm->appendTask ( new LoadLanguageModel ( rg_ ) );
143  mylm->run ( original_data ); //Loading language model only once
144  {
145  ucam::util::TrivialThreadPool tp ( threadcount_ );
146  for ( ucam::util::IntRangePtr ir (ucam::util::IntRangeFactory ( rg_,
147  HifstConstants::kRangeOne ) )
148  ; !ir->done(); ir->next() ) {
149  ReadFst *applylm = new ReadFst ( rg_ , HifstConstants::kLatticeLoad ) ;
150  applylm->appendTask
151  ( new ApplyLanguageModel ( rg_
152  , HifstConstants::kLmLoad
153  , HifstConstants::kLatticeLoad
154  , HifstConstants::kLatticeStore
155  , rg_.exists (HifstConstants::kLatticeLoadDeleteLmCost) ) )
156  ( new WriteFst ( rg_ , HifstConstants::kLatticeStore ) )
157  ( new SpeedStats ( rg_ ) )
158  ;
159  Data *d = new Data; //( original_data );
160  d->klm = original_data.klm;
161  d->sidx = ir->get();
162  LINFO ( "Running sentence " << d->sidx );
163  tp ( ucam::util::TaskFunctor<Data> ( applylm, d ) );
164  }
165  }
166  return false;
167  };
