127  private void loadDir(String dirString, Configuration conf)
128  throws IOException {
129  File dir = new File(dirString);
130  CacheConfig cacheConf = new CacheConfig(conf);
131  if (!dir.isDirectory()) {
132  throw new IOException(dirString + " is not a directory!");
133  }
134  File[] names = dir.listFiles(new FilenameFilter() {
135  @Override
136  public boolean accept(File dir, String name) {
137  return name.endsWith("hfile");
138  }
139  });
140  if (names.length == 0) {
141  throw new IOException("No hfiles in " + dirString);
142  }
143  bfs = new BloomFilter[names.length];
144  readers = new HFileRuleReader[names.length];
145  for (File file : names) {
146  String name = file.getName();
147  // Files are generated by a hadoop reducer have the part
148  // number between the 7 and 12th characters, eg part-r-00001
149  int i = Integer.parseInt(name.substring(7, 12));
150  System.out.println("File has bucket number " +i);
151  HFile.Reader hfReader = HFile.createReader(
152  FileSystem.getLocal(conf), new Path(file.getPath()),
153  cacheConf);
154  bfs[i] = BloomFilterFactory.createFromMeta(
155  hfReader.getBloomFilterMetadata(), hfReader);
156  readers[i] = new HFileRuleReader(hfReader);
157  }
158  }

160  private Set<RuleWritable> getAsciiConstraints() throws IOException {
161  Set<RuleWritable> res = new HashSet<>();
162  try (BufferedReader br = new BufferedReader(new FileReader(
163  asciiConstraintsFileName))) {
164  String line;
165  Pattern regex = Pattern.compile(".*: (.*) # (.*)");
166  Matcher matcher;
167  while ((line = br.readLine()) != null) {
168  matcher = regex.matcher(line);
169  if (matcher.matches()) {
170  String[] sourceString = matcher.group(1).split(" ");
171  String[] targetString = matcher.group(2).split(" ");
172  if (sourceString.length != targetString.length) {
173  System.err.println("Malformed ascii constraint file: "
174  + asciiConstraintsFileName);
175  System.exit(1);
176  }
177  List<Integer> source = new ArrayList<Integer>();
178  List<Integer> target = new ArrayList<Integer>();
179  int i = 0;
180  while (i < sourceString.length) {
181  if (i % MAX_SOURCE_PHRASE == 0 && i > 0) {
182  Rule rule = new Rule(-1, source, target);
183  res.add(new RuleWritable(rule));
184  source.clear();
185  target.clear();
186  }
187  source.add(Integer.parseInt(sourceString[i]));
188  target.add(Integer.parseInt(targetString[i]));
189  i++;
190  }
191  Rule rule = new Rule(-1, source, target);
192  res.add(new RuleWritable(rule));
193  } else {
194  System.err.println("Malformed ascii constraint file: "
195  + asciiConstraintsFileName);
196  System.exit(1);
197  }
198  }
199  }
200  return res;
201  }

271  private List<Set<Text>> generateQueries(String testFileName,
272  Configuration conf) throws IOException {
273  PatternInstanceCreator patternInstanceCreator = new PatternInstanceCreator(
274  conf);
275  patternInstanceCreator.createSourcePatterns();
276  List<Set<Text>> queries = new ArrayList<>(readers.length);
277  for (int i = 0; i < readers.length; ++i) {
278  queries.add(new HashSet<Text>());
279  }
280  try (BufferedReader reader = new BufferedReader(new FileReader(
281  testFileName))) {
282  int count = 0;
283  for (String line = reader.readLine(); line != null; line = reader
284  .readLine()) {
285  StopWatch stopWatch = new StopWatch();
286  stopWatch.start();
287  Set<Rule> rules = patternInstanceCreator
288  .createSourcePatternInstances(line);
289  Collection<Text> sources = new ArrayList<>(rules.size());
290  for (Rule rule : rules) {
291  Text source = (new RuleWritable(rule)).getSource();
292  sources.add(source);
293  }
294  for (Text source : sources) {
295  if (filter.filterSource(source)) {
296  continue;
297  }
298  int partition = partitioner.getPartition(source, null,
299  queries.size());
300  queries.get(partition).add(source);
301  }
302  System.out.println("Creating patterns for line " + ++count
303  + " took " + (double) stopWatch.getTime() / 1000d
304  + " seconds");
305  }
306  }
307  return queries;
308  }

381  public int run(String[] args) throws FileNotFoundException, IOException,
382  InterruptedException, IllegalArgumentException,
383  IllegalAccessException {
384  RuleRetrieverParameters params = new RuleRetrieverParameters();
385  JCommander cmd = new JCommander(params);
386 
387  try {
388  cmd.parse(args);
389  Configuration conf = getConf();
390  Util.ApplyConf(cmd, FeatureCreator.MAPRED_SUFFIX, conf);
391  RuleRetriever retriever = new RuleRetriever();
392  retriever.loadDir(params.hfile, conf);
393  retriever.setup(params.test_file, conf);
394  StopWatch stopWatch = new StopWatch();
395  stopWatch.start();
396  System.err.println("Generating query");
397  List<Set<Text>> queries = retriever.generateQueries(
398  params.test_file, conf);
399  System.err.printf("Query took %d seconds to generate\n",
400  stopWatch.getTime() / 1000);
401  System.err.println("Executing queries");
402  try (BufferedWriter out = new BufferedWriter(
403  new OutputStreamWriter(new GZIPOutputStream(
404  new FileOutputStream(params.rules))))) {
405  FeatureCreator features = new FeatureCreator(conf);
406  ExecutorService threadPool = Executors.newFixedThreadPool(Integer.parseInt(params.retrieval_threads));
407 
408  for (int i = 0; i < queries.size(); ++i) {
409  HFileRuleQuery query = new HFileRuleQuery(
410  retriever.readers[i], retriever.bfs[i], out,
411  queries.get(i), features, retriever, conf);
412  threadPool.execute(query);
413  }
414  threadPool.shutdown();
415  threadPool.awaitTermination(1, TimeUnit.DAYS);
416  // Add ascii constraints not already found in query
417  for (RuleWritable asciiConstraint : retriever.asciiConstraints) {
418  if (!retriever.foundAsciiConstraints
419  .contains(asciiConstraint)) {
420  features.writeRule(asciiConstraint,
421  AlignmentAndFeatureMap.EMPTY,
422  EnumRuleType.ASCII_OOV_DELETE, out);
423  }
424  }
425  // Add Deletetion and OOV rules
426  RuleWritable deletionRuleWritable = new RuleWritable();
427  deletionRuleWritable.setLeftHandSide(new Text(
428  EnumRuleType.ASCII_OOV_DELETE.getLhs()));
429  deletionRuleWritable.setTarget(new Text("0"));
430  RuleWritable oovRuleWritable = new RuleWritable();
431  oovRuleWritable.setLeftHandSide(new Text(
432  EnumRuleType.ASCII_OOV_DELETE.getLhs()));
433  oovRuleWritable.setTarget(new Text(""));
434  for (Text source : retriever.testVocab) {
435  if (retriever.foundTestVocab.contains(source)) {
436  deletionRuleWritable.setSource(source);
437  features.writeRule(deletionRuleWritable,
438  AlignmentAndFeatureMap.EMPTY,
439  EnumRuleType.ASCII_OOV_DELETE, out);
440  } else {
441  oovRuleWritable.setSource(source);
442  features.writeRule(oovRuleWritable,
443  AlignmentAndFeatureMap.EMPTY,
444  EnumRuleType.ASCII_OOV_DELETE, out);
445  }
446  }
447  // Glue rules
448  for (RuleWritable glueRule : retriever.getGlueRules()) {
449  features.writeRule(glueRule, AlignmentAndFeatureMap.EMPTY,
450  EnumRuleType.GLUE, out);
451  }
452  }
453  System.out.println(retriever.foundAsciiConstraints);
454  System.out.println(retriever.foundTestVocab);
455  } catch (ParameterException e) {
456  System.err.println(e.getMessage());
457  cmd.usage();
458  }
459 
460  return 1;
461  }
