43  public static void ApplyConf(JCommander cmd, String suffix,
44  Configuration conf) throws IllegalArgumentException,
45  IllegalAccessException {
46  Object params = cmd.getObjects().get(0);
47  for (Field field : params.getClass().getDeclaredFields()) {
48  String name = field.getName();
49  String value = (String) field.get(params);
50  conf.set(name, value);
51  }
52  String mapreduceFeatures = conf.get("mapreduce_features");
53  if (mapreduceFeatures != null) {
54  // initial feature index is zero, then increments with the number of
55  // features of each feature type. nextFeatureIndex is used to
56  // prevent
57  // conf to be overwritten before being used.
58  int featureIndex = 0, nextFeatureIndex = 0;
59  for (String featureString : mapreduceFeatures.split(",")) {
60  MapReduceFeature feature =
61  MapReduceFeature.findFromConf(featureString);
62  if (feature.isProvenanceFeature()) {
63  for (String provenance : conf.get("provenance").split(",")) {
64  featureIndex = nextFeatureIndex;
65  // the next feature index is the current plus the number
66  // of
67  // features
68  // of the current feature class.
69  nextFeatureIndex += feature.getNumberOfFeatures();
70  conf.setInt(feature.getConfName() + "-" + provenance
71  + suffix, featureIndex);
72  }
73  } else {
74  featureIndex = nextFeatureIndex;
75  // the next feature index is the current plus the number of
76  // features
77  // of the current feature class.
78  nextFeatureIndex += feature.getNumberOfFeatures();
79  conf.setInt(feature.getConfName() + suffix, featureIndex);
