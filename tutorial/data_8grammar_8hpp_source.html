102  inline const std::string getRHSSource ( std::size_t idx, uint rulepos ) const {
103  std::size_t pos = filecontents.find_first_of ( " ", vpos[idx].p );
104  std::size_t j = vpos[idx].p - 1, jold;
105  for ( uint k = 0; k <= rulepos; ++k ) {
106  jold = j;
107  j = filecontents.find_first_of ( "_ ", jold + 1 );
108  if ( j == std::string::npos )
109  if ( rulepos ) return "";
110  }
111  return filecontents.substr ( jold + 1, j - jold - 1 );
112  }

199  inline const bool isAcceptedByVocabulary ( const std::size_t idx,
200  const unordered_set<std::string>& vcb ) const {
201  if ( !vcb.size() ) return true;
202  std::vector<std::string> tx = getRHSSplitTranslation ( idx );
203  for ( uint k = 0; k < tx.size(); ++k ) {
204  if ( tx[k] == "<dr>" || tx[k] == "<oov>" || tx[k] == "<s>" || tx[k] == "</s>"
205  || tx[k] == "<sep>") continue;
206  if ( !isTerminal ( tx[k] ) ) continue;
207  if ( vcb.find ( tx[k] ) == vcb.end() ) return false;
208  }
209  return true;
