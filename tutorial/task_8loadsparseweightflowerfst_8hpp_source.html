94  bool directload ( const std::string& filename ) {
95  bool directload = false;
96  std::vector<std::string> vgn;
97  boost::algorithm::split ( vgn, filename, boost::algorithm::is_any_of ( "." ) );
98  if ( vgn[vgn.size() - 1] == "fst" ) directload = true;
99  else if ( vgn.size() > 2 ) if ( vgn[vgn.size() - 1] == "gz"
100  && vgn[vgn.size() - 2] == "fst" ) directload = true;
101  if ( directload ) {
102  LINFO ( "Loading FST directly (assumes arcsorted flower) =" << filename );
103  fst::VectorFst<TupleArc32> *yupi = fst::VectorFstRead<TupleArc32> ( filename );
104  flowerlattice_ = *yupi;
105  delete yupi;
106  return true;
107  }
108  return false;
109  }

118  bool load ( const std::string& filename ) {
119  if ( !USER_CHECK ( filename != "" , "No grammar to load?" ) ) return false;
120  if ( filename == previousfile_ ) {
121  LINFO ( "Skipping grammar loading..." );
122  return false;
123  }
124  previousfile_ = filename;
125  FORCELINFO ( "building rule flower from " << filename );
126  if ( directload ( filename ) ) return true;
128  // unordered_set<TupleArc32::Label> idxrules;
129  unordered_set<unsigned> idxrules;
130  if ( filterbyalilats_ )
131  for ( ucam::util::IntRangePtr ir (ucam::util::IntRangeFactory ( rg_ ) );
132  !ir->done ();
133  ir->next () ) {
134  fst::VectorFst<fst::LexStdArc> *alilatsfst =
135  fst::VectorFstRead<fst::LexStdArc> ( alilats_ ( ir->get() ) );
136  fst::extractSourceVocabulary<fst::LexStdArc> ( *alilatsfst, &idxrules );
137  delete alilatsfst;
138  }
139  flowerlattice_.AddState();
140  flowerlattice_.SetStart ( 0 );
141  flowerlattice_.SetFinal ( 0, TupleArc32::Weight::One() );
142  unsigned lc = 0, llc = 0;
143  ucam::util::iszfstream myrulefile;
144  myrulefile.open ( filename.c_str() );
145  LINFO ( "Opening rule file " << filename );
146  if ( !myrulefile.is_open() ) {
147  LERROR ( "Failed to open " << filename );
148  return false;
149  }
150  do {
151  std::string line;
152  getline ( myrulefile, line );
153  // LDEBUG2(line);
154  ++lc;
155  if ( line.size() > 0 ) {
156  while ( line.at ( line.length() - 1 ) == ' ' )
157  line.erase ( line.length() - 1 );
158  while ( line.at ( 0 ) == ' ' )
159  line.erase ( 0, 1 );
160  if ( line.at ( 0 ) == '#' || line.at ( 0 ) == '%' ) line = "";
161  }
162  if ( line == "" ) {
163  LDEBUG ( lc << "is an empty line/comment." );
164  continue;
165  }
166  if ( ! ( lc % 100000 ) ) LINFO ( lc << " rules parsed..." );
167  if ( !idxrules.empty() )
168  if ( idxrules.find ( lc ) == idxrules.end() ) continue;
169  llc++;
170  TupleW32 vtcost;
171  std::vector<std::string> fields;
172  boost::algorithm::split ( fields, line, boost::algorithm::is_any_of ( " " ) );
173  for ( unsigned k = 3; k < fields.size(); ++k ) {
174  float prob;
175  unsigned pos = 0;
176  std::vector<std::string> splitfields;
177  boost::algorithm::split ( splitfields, fields[k],
178  boost::algorithm::is_any_of ( "@" ) );
179  using ucam::util::toNumber;
180  if ( splitfields.size() == 2 ) {
181  if ( splitfields[0] == "" ) prob = 1.000f;
182  else prob = toNumber<float> ( splitfields[0] );
183  pos = offset_ + toNumber<unsigned> ( splitfields[1] );
184  } else {
185  prob = toNumber<float> ( fields[k] );
186  pos = offset_ + k - 2;
187  LDEBUG ( pos );
188  }
189  vtcost.Push ( pos,
190  prob ); // It's sparse weight -- will take care of taking out zeros.
191  splitfields.clear();
192  }
193  unsigned label = lc;
194  flowerlattice_.AddArc ( 0, TupleArc32 ( label, label, vtcost, 0 ) );
195  } while ( myrulefile && !myrulefile.eof() );
196  LINFO ( "File: " << filename << " has been succesfully loaded" );
197  myrulefile.close();
198  LINFO ( "Number of rules actually loaded for this job: " << llc );
199  fst::ArcSort<TupleArc32> ( &flowerlattice_, fst::ILabelCompare<TupleArc32>() );
200  return built_ = true;
201  };
202 
203  private:

205 };
