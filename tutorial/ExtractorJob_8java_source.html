96  private static class ExtractorMapper
97  extends
98  Mapper<MapWritable, TextArrayWritable, RuleWritable, RuleInfoWritable> {
99 
100  private static final IntWritable ONE = new IntWritable(1);
101 
102  private RuleInfoWritable ruleInfo = new RuleInfoWritable();
103 
104  private Map<Text, ByteWritable> prov2Id = new HashMap<>();
105 
106  private static final ByteWritable ALL = new ByteWritable((byte) 0);
107 
108  @Override
109  protected void setup(Context context) throws IOException,
110  InterruptedException {
111  super.setup(context);
112  String provString = context.getConfiguration().get(
113  ProvenanceCountMap.PROV);
114  String[] provs = provString.split(",");
115  if (provs.length + 1 >= Byte.MAX_VALUE) {
116  throw new RuntimeException(
117  String.format(
118  "Number of provenances is %d which is greater than 128",
119  provs.length));
120  }
121  for (int i = 0; i < provs.length; ++i) {
122  prov2Id.put(new Text(provs[i]),
123  new ByteWritable((byte) (i + 1)));
124  }
125  }
126 
127  /*
128  * (non-Javadoc)
129  *
130  * @see org.apache.hadoop.mapreduce.Mapper#map(java.lang.Object,
131  * java.lang.Object, org.apache.hadoop.mapreduce.Mapper.Context)
132  */
133  @Override
134  protected void map(MapWritable key, TextArrayWritable value,
135  Context context) throws IOException, InterruptedException {
136  Configuration conf = context.getConfiguration();
137  String sourceSentence = ((Text) value.get()[0]).toString();
138  String targetSentence = ((Text) value.get()[1]).toString();
139  String wordAlign = ((Text) value.get()[2]).toString();
140  SentencePair sp = new SentencePair(sourceSentence, targetSentence);
141  Alignment a = new Alignment(wordAlign, sp);
142  RuleExtractor re = new RuleExtractor(conf);
143  for (Rule r : re.extract(a, sp)) {
144  RuleWritable rw = new RuleWritable(r);
145  AlignmentWritable aw = new AlignmentWritable(r.getAlignment());
146  ruleInfo.clear();
147  ruleInfo.putProvenanceCount(ALL, ONE);
148  for (Writable prov : key.keySet()) {
149  if (prov2Id.keySet().contains(prov)) {
150  ruleInfo.putProvenanceCount(prov2Id.get(prov), ONE);
151  }
152  }
153  ruleInfo.putAlignmentCount(aw, 1);
154  context.write(rw, ruleInfo);
155  }
156  }
157  }

159  private static class ExtractorReducer
160  extends
161  Reducer<RuleWritable, RuleInfoWritable, RuleWritable, RuleInfoWritable> {
162 
163  private RuleInfoWritable compressed = new RuleInfoWritable();
164 
165  @Override
166  protected void reduce(
167  RuleWritable key,
168  Iterable<RuleInfoWritable> values, Context context)
169  throws IOException, InterruptedException {
170  compressed.clear();
171  for (RuleInfoWritable value : values) {
172  compressed.increment(value);
173  }
174  context.write(key, compressed);
175  }
176  }
