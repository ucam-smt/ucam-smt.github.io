19 template <class Arc>
20 class TuneWpMain {
21  private:
22 
23  const ucam::util::RegistryPO& rg_;
24  fst::MakeWeight<Arc> mw_;
25  public:
26  TuneWpMain (ucam::util::RegistryPO const& rg) :
27  rg_ (rg) {
28  };
29  void operator () () {
30  using fst::WordPenaltyMapper;
31  using fst::VectorFstRead;
32  using fst::FstWrite;
33  using ucam::util::PatternAddress;
34  PatternAddress<unsigned> pi (rg_.get<std::string> (HifstConstants::kInput ) );
35  PatternAddress<unsigned> po (rg_.get<std::string> (HifstConstants::kOutput ) );
36  ucam::util::NumberRange<float> wp ( rg_.get<std::string>
37  (HifstConstants::kWordPenalty ) );
38  //Insert epsilons
39  unordered_set<typename Arc::Label> epsilons =
40  rg_.getSetNumber<typename Arc::Label> (HifstConstants::kEpsilonLabels);
41  unsigned shp = rg_.get<unsigned> (HifstConstants::kNbest);
42  for ( ucam::util::IntRangePtr ir (ucam::util::IntRangeFactory ( rg_,
43  HifstConstants::kRangeOne ) );
44  !ir->done();
45  ir->next() ) {
46  fst::VectorFst<Arc> *mfst = VectorFstRead<Arc> (pi (ir->get() ) );
47  for ( wp.start();
48  !wp.done();
49  wp.next() ) {
50  fst::Map<Arc, WordPenaltyMapper<Arc> > (mfst,
51  WordPenaltyMapper<Arc> (mw_ (wp.get() ), epsilons) );
52  if (shp < std::numeric_limits<unsigned>::max() ) {
53  fst::VectorFst<Arc> aux;
54  fst::ShortestPath<Arc> (*mfst, &aux, shp);
55  *mfst = aux;
56  }
57  std::string auxs = po (ir->get() );
58  ucam::util::find_and_replace (auxs, "%%wp%%",
59  ucam::util::toString<float> (wp() ) );
60  FstWrite<Arc> (*mfst, auxs);
61  }
62  }
63  };
64 
65 };
66 
67 int main (int argc, const char* argv[] ) {
68  ucam::util::initLogger ( argc, argv );
69  FORCELINFO ( argv[0] << " starts!" );
70  ucam::util::RegistryPO rg ( argc, argv );
71  FORCELINFO ( rg.dump ( "CONFIG parameters:\n=====================",
72  "=====================" ) ) ;
73  if (rg.get<std::string> (HifstConstants::kHifstSemiring.c_str() ) ==
74  HifstConstants::kHifstSemiringStdArc) {
75  TuneWpMain<fst::StdArc> twp (rg);
76  twp();
77  } else if (rg.get<std::string> (HifstConstants::kHifstSemiring.c_str() ) ==
78  HifstConstants::kHifstSemiringLexStdArc) {
79  TuneWpMain<fst::LexStdArc> twp (rg);
80  twp();
81  } else {
82  LERROR ("Unknown type");
83  }
84  FORCELINFO ( argv[0] << " ends!" );
85 }
