88  public List<Rule> extract(Alignment a, SentencePair sp) {
89  List<Rule> res = extractPhrasePairs(a, sp);
90  if (REMOVE_MONOTONIC_REPEATS) {

107  private List<Rule> extractPhrasePairs(Alignment a, SentencePair sp) {
108 
109  List<Rule> res = new ArrayList<Rule>();
110 
111  // loop over source index (beginning of phrase)
112  for (int sourceStartIndex = 0; sourceStartIndex < sp.getSource()
113  .getWords().length; sourceStartIndex++) {
114  // source phrase built on the fly
115  List<Integer> sourcePhrase = new ArrayList<Integer>();
116  // maintain the minimum and maximum target index aligned to the
117  // source phrase
118  int minTargetIndex = a.getMinTargetIndex(sourceStartIndex);
119  int maxTargetIndex = a.getMaxTargetIndex(sourceStartIndex);
120  // loop over source index (end of phrase)
121  for (int sourceEndIndex = sourceStartIndex; sourceEndIndex < Math
122  .min(sourceStartIndex + MAX_SOURCE_PHRASE, sp.getSource()
123  .getWords().length); sourceEndIndex++) {
124  // update the sourcePhrase
125  sourcePhrase.add(sp.getSource().getWords()[sourceEndIndex]);
126  // update minimum and maximum target index aligned to the source
127  // phrase
128  int minTargetIndexCandidate = a
129  .getMinTargetIndex(sourceEndIndex);
130  int maxTargetIndexCandidate = a
131  .getMaxTargetIndex(sourceEndIndex);
132  if (minTargetIndexCandidate < minTargetIndex)
133  minTargetIndex = minTargetIndexCandidate;
134  if (maxTargetIndexCandidate > maxTargetIndex)
135  maxTargetIndex = maxTargetIndexCandidate;
136  if (minTargetIndex > maxTargetIndex) // occurs when haven't
137  // found any aligned
138  // word
139  // yet
140  continue;
141 
142  // check if the target phrase between positions minTargetIndex
143  // and maxTargetIndex and the source phrase are consistent with
144  // the alignment
145  boolean consistent = true;
146  List<Integer> targetPhrase = new ArrayList<Integer>();
147  for (int targetIndex = minTargetIndex; targetIndex <= maxTargetIndex; targetIndex++) {
148  targetPhrase.add(sp.getTarget().getWords()[targetIndex]);
149  if (a.isTargetAligned(targetIndex)
150  && (a.getMinSourceIndex(targetIndex) < sourceStartIndex || a
151  .getMaxSourceIndex(targetIndex) > sourceEndIndex)) {
152  consistent = false;
153  break;
154  }
155  }
156 
157  // we found a phrase pair
158  if (consistent) {
159  // TODO the rule may be constructed on the fly
160  // to avoid duplicated logic as well as duplicated loops.
161  Rule r = new Rule(sourceStartIndex, sourceEndIndex,
162  minTargetIndex, maxTargetIndex, sp, a);
163  res.add(r);
164  List<Rule> extendedUnaligned = extendUnalignedBoundaryWord(
165  sourceStartIndex, sourceEndIndex, minTargetIndex,
166  maxTargetIndex, a, sp);
167  res.addAll(extendedUnaligned);
168  }
169  }
170  }
171  return res;
172  }

186  private List<Rule> extendUnalignedBoundaryWord(int sourceStartIndex,
187  int sourceEndIndex, int targetStartIndex, int targetEndIndex,
188  Alignment a, SentencePair sp) {
189  List<Rule> res = new ArrayList<Rule>();
190  int prev = 0; // number of previous target unaligned words
191  int targetExtendIndex = targetStartIndex - 1;
192  // i>0 because i=0 is a position reserved for NULL. In this version, we
193  // don't make
194  // use of the alignment to NULL because we use symmetrized alignments
195  // but that could
196  // be changed
197  while (targetExtendIndex >= 0 && !a.isTargetAligned(targetExtendIndex)) {
198  prev++;
199  Rule r = new Rule(sourceStartIndex, sourceEndIndex,
200  targetExtendIndex, targetEndIndex, sp, a);
201  res.add(r);
202  targetExtendIndex--;
203  }
204  int foll = 0; // number of following target unaligned words
205  targetExtendIndex = targetEndIndex + 1;
206  while (targetExtendIndex < sp.getTarget().getWords().length
207  && !a.isTargetAligned(targetExtendIndex)) {
208  foll++;
209  Rule r = new Rule(sourceStartIndex, sourceEndIndex,
210  targetStartIndex, targetExtendIndex, sp, a);
211  res.add(r);
212  targetExtendIndex++;
213  }
214 
215  if (prev > 0 && foll > 0) { // if there are unaligned words in both
216  // sides:
217  for (targetExtendIndex = 1; targetExtendIndex <= prev; targetExtendIndex++) { // for
218  // each
219  // start
220  // (including
221  // at
222  // least
223  // one
224  // previous)
225  int start = targetStartIndex - targetExtendIndex;
226  for (int k = 1; k <= foll; k++) { // for each end (including at
227  // least one following)
228  int end = targetEndIndex + k;
229  Rule r = new Rule(sourceStartIndex, sourceEndIndex, start,
230  end, sp, a);
231  res.add(r);
232  }
233  }
234  }
235  return res;
236  }

240  private List<Rule> extractHieroRule(Alignment a, SentencePair sp) {
241  List<Rule> res = new ArrayList<Rule>();
242 
243  // first get the regular blocks, that is the boundaries for the phrase
244  // pairs that would be extracted
245  // these are maximum width blocks, they cannot be extended
246  List<Block> regularBlocks = getRegularBlocks(a, sp);
247  // for (Block b: regularBlocks) {
248  // System.err.println(b);
249  // }
250 
251  // something here about monotonic repetitions
252  res.addAll(extractInternalBlockRules(regularBlocks, a, sp));
253 
254  // finally extract the rules "inter-blocks"
255  res.addAll(extractInterBlockRules(regularBlocks, a, sp));
256  return res;
257  }

259  private List<Block> getRegularBlocks(Alignment a, SentencePair sp) {
260  List<Block> res = new ArrayList<Block>();
261  int sourceStartIndex = 0;
262  int sourceEndIndex = 0;
263  int targetStartIndex = 0;
264  int targetEndIndex = 0;
265  while (sourceStartIndex <= sp.getSource().getWords().length
266  && targetStartIndex <= sp.getTarget().getWords().length) {
267  Block next = findNextBlock(sourceStartIndex, sourceEndIndex,
268  targetStartIndex, targetEndIndex, a, sp);
269 
270  sourceStartIndex = next.sourceStartIndex;
271  sourceEndIndex = next.sourceEndIndex;
272  targetStartIndex = next.targetStartIndex;
273  targetEndIndex = next.targetEndIndex;
274 
275  if (targetStartIndex >= sp.getTarget().getWords().length
276  && sourceStartIndex >= sp.getSource().getWords().length) {
277  // do nothing
278  } else if (targetStartIndex < sp.getTarget().getWords().length
279  && !a.isTargetAligned(targetStartIndex)) {
280  sourceEndIndex--;
281  } else if (sourceStartIndex < sp.getSource().getWords().length
282  && !a.isSourceAligned(sourceStartIndex)) {
283  targetEndIndex--;
284  }
285 
286  if (targetStartIndex >= sp.getTarget().getWords().length
287  || sourceStartIndex >= sp.getSource().getWords().length) {
288  break;
289  }
290  if (sourceStartIndex <= sourceEndIndex
291  && targetStartIndex <= targetEndIndex) {
292  res.add(new Block(sourceStartIndex, sourceEndIndex,
293  targetStartIndex, targetEndIndex));
294  }
295  sourceStartIndex = sourceEndIndex + 1;
296  targetStartIndex = targetEndIndex + 1;
297  sourceEndIndex = sourceStartIndex;
298  targetEndIndex = targetStartIndex;
299  }
300  return res;
301  }

303  private Block findNextBlock(int sourceStartIndex, int sourceEndIndex,
304  int targetStartIndex, int targetEndIndex, Alignment a,
305  SentencePair sp) {
306  if (targetStartIndex >= sp.getTarget().getWords().length
307  || sourceStartIndex >= sp.getSource().getWords().length
308  || !a.isTargetAligned(targetStartIndex)
309  || !a.isSourceAligned(sourceStartIndex)) {
310  return new Block(sourceStartIndex, sourceEndIndex,
311  targetStartIndex, targetEndIndex);
312  }
313  boolean cont = true;
314  while (cont) {
315  cont = false;
316  for (int sourceIndex = sourceStartIndex; sourceIndex <= sourceEndIndex; sourceIndex++) {
317  if (a.getMaxTargetIndex(sourceIndex) > targetEndIndex) {
318  targetEndIndex = a.getMaxTargetIndex(sourceIndex);
319  cont = true;
320  }
321  }
322  for (int targetIndex = targetStartIndex; targetIndex <= targetEndIndex; targetIndex++) {
323  if (a.getMaxSourceIndex(targetIndex) > sourceEndIndex) {
324  sourceEndIndex = a.getMaxSourceIndex(targetIndex);
325  cont = true;
326  }
327  }
328  }
329  return new Block(sourceStartIndex, sourceEndIndex, targetStartIndex,
330  targetEndIndex);
331  }

342  private Pair<Integer, Integer> updateTargetLimit(Alignment a,
343  int sourceStartIndex, int sourceEndIndex) {
344  Pair<Integer, Integer> res = new Pair<Integer, Integer>();
345  int minTargetIndex = a.getMinTargetIndex(sourceStartIndex);
346  int maxTargetIndex = a.getMaxTargetIndex(sourceStartIndex);
347  for (int sourceIndex = sourceStartIndex; sourceIndex <= sourceEndIndex; sourceIndex++) {
348  int minTargetIndexCandidate = a.getMinTargetIndex(sourceIndex);
349  int maxTargetIndexCandidate = a.getMaxTargetIndex(sourceIndex);
350  if (minTargetIndexCandidate < minTargetIndex)
351  minTargetIndex = minTargetIndexCandidate;
352  if (maxTargetIndexCandidate > maxTargetIndex)
353  maxTargetIndex = maxTargetIndexCandidate;
354  }
355  res.setFirst(minTargetIndex);
356  res.setSecond(maxTargetIndex);
357  return res;
358  }

360  private List<Rule> extractInternalBlockRules(Block regularBlock,
361  Alignment a, SentencePair sp) {
362  List<Rule> res = new ArrayList<Rule>();
363  for (int sourceStartIndex = regularBlock.sourceStartIndex; sourceStartIndex <= regularBlock.sourceEndIndex; sourceStartIndex++) {
364  for (int sourceEndIndex = sourceStartIndex + 1; sourceEndIndex <= regularBlock.sourceEndIndex; sourceEndIndex++) {
365  // check links and update target limit
366  Pair<Integer, Integer> targetLimit = updateTargetLimit(a,
367  sourceStartIndex, sourceEndIndex);
368  int offset = 0;
369  for (int targetIndex = targetLimit.getFirst(); targetIndex <= targetLimit
370  .getSecond(); targetIndex++) {
371  if (a.getMinSourceIndex(targetIndex) < sourceStartIndex) {
372  offset = a.getMinSourceIndex(targetIndex)
373  - sourceStartIndex;
374  break;
375  }
376  if (a.getMaxSourceIndex(targetIndex) > sourceEndIndex) {
377  offset = a.getMaxSourceIndex(targetIndex)
378  - sourceEndIndex;
379  break;
380  }
381  }
382  if (offset < 0) {
383  break; // if negative offset, jump to another
384  // sourceStartIndex (end the sourceEndIndex for
385  // loop)
386  } else if (offset > 0) {
387  sourceEndIndex = sourceEndIndex + offset - 1; // if
388  // positive,
389  // add
390  // offset
391  // to jump
392  // to
393  // the
394  // adequate
395  // sourceEndIndex
396  } else { // zero offset, found a plausible subregion
397  // System.err.println("Extracting rules one nonterminal "
398  // +
399  // sourceStartIndex + " " + sourceEndIndex + " " +
400  // minTargetIndex + " " + maxTargetIndex);
401  res.addAll(extractRulesOneNonTerminal(sourceStartIndex,
402  sourceEndIndex, targetLimit.getFirst(),
403  targetLimit.getSecond(), a, sp));
404  res.addAll(extractRulesTwoNonTerminal(sourceStartIndex,
405  sourceEndIndex, targetLimit.getFirst(),
406  targetLimit.getSecond(), a, sp));
407  }
408  }
409  }
410  return res;
411  }

413  private List<Rule> extractRulesOneNonTerminal(int sourceStartIndex,
414  int sourceEndIndex, int minTargetIndex, int maxTargetIndex,
415  Alignment a, SentencePair sp) {
416  List<Rule> res = new ArrayList<Rule>();
417  int offset = 0;
418  for (int sourceStartIndexX = sourceStartIndex; sourceStartIndexX <= sourceEndIndex; sourceStartIndexX++) {
419  for (int sourceEndIndexX = sourceStartIndexX; sourceEndIndexX <= sourceEndIndex; sourceEndIndexX++) {
420  if (!a.isSourceAligned(sourceStartIndexX)
421  || !a.isSourceAligned(sourceEndIndexX)) {
422  break;
423  }
424  Pair<Integer, Integer> targetLimit = updateTargetLimit(a,
425  sourceStartIndexX, sourceEndIndexX);
426  offset = 0;
427  for (int targetIndex = targetLimit.getFirst(); targetIndex <= targetLimit
428  .getSecond(); targetIndex++) {
429  if (a.getMinSourceIndex(targetIndex) < sourceStartIndexX) {
430  offset = a.getMinSourceIndex(targetIndex)
431  - sourceStartIndexX;
432  break;
433  }
434  if (a.getMaxSourceIndex(targetIndex) > sourceEndIndexX) {
435  offset = a.getMaxSourceIndex(targetIndex)
436  - sourceEndIndexX;
437  break;
438  }
439  }
440  if (targetLimit.getFirst() == minTargetIndex
441  && targetLimit.getSecond() == maxTargetIndex) {
442  break;
443  }
444  if (offset < 0) {
445  break;
446  }
447  if (offset > 0) {
448  sourceEndIndexX = sourceEndIndexX + offset - 1;
449  } else if (sourceStartIndexX != sourceStartIndex
450  || sourceEndIndexX != sourceEndIndex) {
451  if (filterPassOneNonTerminalRule(sourceStartIndex,
452  sourceEndIndex, sourceStartIndexX, sourceEndIndexX)) {
453  Rule r = new Rule(sourceStartIndex, sourceEndIndex,
454  minTargetIndex, maxTargetIndex,
455  sourceStartIndexX, sourceEndIndexX,
456  targetLimit.getFirst(),
457  targetLimit.getSecond(), sp, a);
458  res.add(r);
459  }
460  }
461  }
462  }
463  return res;
464  }

466  private List<Rule> extractRulesTwoNonTerminal(int sourceStartIndex,
467  int sourceEndIndex, int minTargetIndex, int maxTargetIndex,
468  Alignment a, SentencePair sp) {
469  List<Rule> res = new ArrayList<Rule>();
470  if (sourceEndIndex - sourceStartIndex < 2) { // we want at least 2
471  // source words to
472  // extract
473  // a rule with two
474  // nonterminals
475  return res;
476  }
477  for (int sourceStartIndexX = sourceStartIndex; sourceStartIndexX < sourceEndIndex - 1; sourceStartIndexX++) {
478  for (int sourceEndIndexX = sourceStartIndexX; sourceEndIndexX < sourceEndIndex - 1; sourceEndIndexX++) {
479  if (!a.isSourceAligned(sourceStartIndexX)
480  || !a.isSourceAligned(sourceEndIndexX))
481  break;
482  Pair<Integer, Integer> targetLimit = updateTargetLimit(a,
483  sourceStartIndexX, sourceEndIndexX);
484  int offset = 0;
485  for (int targetIndex = targetLimit.getFirst(); targetIndex <= targetLimit
486  .getSecond(); targetIndex++) {
487  if (a.getMinSourceIndex(targetIndex) < sourceStartIndexX) {
488  offset = a.getMinSourceIndex(targetIndex)
489  - sourceStartIndexX;
490  break;
491  }
492  if (a.getMaxSourceIndex(targetIndex) > sourceEndIndexX) {
493  offset = a.getMaxSourceIndex(targetIndex)
494  - sourceEndIndexX;
495  break;
496  }
497  }
498  if (targetLimit.getFirst() == minTargetIndex
499  && targetLimit.getSecond() == maxTargetIndex) {
500  break;
501  }
502  if (offset < 0) {
503  break;
504  } else if (offset > 0) {
505  sourceEndIndexX = sourceEndIndexX + offset - 1;
506  } else {
507  for (int sourceStartIndexX2 = sourceEndIndexX + 2; sourceStartIndexX2 <= sourceEndIndex; sourceStartIndexX2++) {
508  for (int sourceEndIndexX2 = sourceStartIndexX2; sourceEndIndexX2 <= sourceEndIndex; sourceEndIndexX2++) {
509  if (!a.isSourceAligned(sourceStartIndexX2)
510  || !a.isSourceAligned(sourceEndIndexX2)) {
511  break;
512  }
513  Pair<Integer, Integer> targetLimitX2 = updateTargetLimit(
514  a, sourceStartIndexX2, sourceEndIndexX2);
515  int offset2 = 0;
516  for (int targetIndex = targetLimitX2.getFirst(); targetIndex <= targetLimitX2
517  .getSecond(); targetIndex++) {
518  if (a.getMinSourceIndex(targetIndex) < sourceStartIndexX2) {
519  offset2 = a.getMinSourceIndex(targetIndex)
520  - sourceStartIndexX2;
521  break;
522  }
523  if (a.getMaxSourceIndex(targetIndex) > sourceEndIndexX2) {
524  offset2 = a.getMaxSourceIndex(targetIndex)
525  - sourceEndIndexX2;
526  break;
527  }
528  }
529 
530  if (offset2 < 0)
531  break;
532  else if (offset2 > 0) {
533  sourceEndIndexX2 = sourceEndIndexX2 + offset2
534  - 1;
535  } else {
536  if (filterPassTwoNonTerminalRule(
537  sourceStartIndex, sourceEndIndex,
538  sourceStartIndexX, sourceEndIndexX,
539  sourceStartIndexX2, sourceEndIndexX2, a)) {
540  Rule r = new Rule(sourceStartIndex,
541  sourceEndIndex, minTargetIndex,
542  maxTargetIndex, sourceStartIndexX,
543  sourceEndIndexX,
544  targetLimit.getFirst(),
545  targetLimit.getSecond(),
546  sourceStartIndexX2,
547  sourceEndIndexX2,
548  targetLimitX2.getFirst(),
549  targetLimitX2.getSecond(), sp, a);
550  res.add(r);
551  }
552  }
553  }
554  }
555  }
556  }
557  }
558  return res;
559  }

588  private boolean filterPassOneNonTerminalRule(int sourceStartIndex,
589  int sourceEndIndex, int sourceStartIndexX, int sourceEndIndexX) {
590  boolean res = (((sourceEndIndex - sourceStartIndex + 1)
591  - (sourceEndIndexX - sourceStartIndexX + 1) + 1) <= MAX_SOURCE_ELEMENTS
592  && sourceStartIndexX - sourceStartIndex <= MAX_TERMINAL_LENGTH
593  && sourceEndIndex - sourceEndIndexX <= MAX_TERMINAL_LENGTH && sourceEndIndexX
594  - sourceStartIndexX + 1 <= MAX_NONTERMINAL_SPAN);
595  if (!res) {
596  return false;
597  }
598  // TODO something about monotonic repetitions
599  if (REMOVE_MONOTONIC_REPEATS
600  && isMonotonicRepeatOneNonterminal(sourceStartIndex,
601  sourceEndIndex, sourceStartIndexX, sourceEndIndexX)) {
602  return false;
603  }
604  return true;
605  }

616  private boolean filterPassTwoNonTerminalRule(int sourceStartIndex,
617  int sourceEndIndex, int sourceStartIndexX, int sourceEndIndexX,
618  int sourceStartIndexX2, int sourceEndIndexX2, Alignment a) {
619  boolean middleTerminalAligned = false;
620  for (int sourceIndex = sourceEndIndexX + 1; sourceIndex < sourceStartIndexX2; sourceIndex++) {
621  if (a.isSourceAligned(sourceIndex)) {
622  middleTerminalAligned = true;
623  break;
624  }
625  }
626  boolean res = (middleTerminalAligned
627  && ((sourceEndIndex - sourceStartIndex + 1)
628  - (sourceEndIndexX - sourceStartIndexX + 1)
629  - (sourceEndIndexX2 - sourceStartIndexX2 + 1) + 2) <= MAX_SOURCE_ELEMENTS
630  && sourceStartIndexX - sourceStartIndex <= MAX_TERMINAL_LENGTH
631  && sourceStartIndexX2 - sourceEndIndexX <= MAX_TERMINAL_LENGTH
632  && sourceEndIndex - sourceEndIndexX2 <= MAX_TERMINAL_LENGTH
633  && sourceEndIndexX - sourceStartIndexX + 1 <= MAX_NONTERMINAL_SPAN && sourceEndIndexX2
634  - sourceStartIndexX2 + 1 <= MAX_NONTERMINAL_SPAN);
635  if (!res) {
636  return res;
637  }
638  if (REMOVE_MONOTONIC_REPEATS
639  && isMonotonicRepeatTwoNonterminal(sourceStartIndex,
640  sourceEndIndex, sourceStartIndexX, sourceEndIndexX,
641  sourceStartIndexX2, sourceEndIndexX2)) {
642  return false;
643  }
644  return true;
645  }

647  private boolean isMonotonicRepeatOneNonterminal(int sourceStartIndex,
648  int sourceEndIndex, int sourceStartIndexX, int sourceEndIndexX) {
649  if (sourceStartIndexX == sourceStartIndex) {
650  if (Xterm.containsKey(sourceEndIndexX)
651  && Xterm.get(sourceEndIndexX).containsKey(sourceEndIndex)) {
652  return true;
653  }
654  if (!Xterm.containsKey(sourceEndIndexX)) {
655  Map<Integer, Boolean> newElt = new HashMap<Integer, Boolean>();
656  newElt.put(sourceEndIndex, true);
657  Xterm.put(sourceEndIndexX, newElt);
658  } else {
659  Xterm.get(sourceEndIndexX).put(sourceEndIndex, true);
660  }
661  }
662  if (sourceEndIndexX == sourceEndIndex) {
663  if (termX.containsKey(sourceStartIndex)
664  && termX.get(sourceStartIndex).containsKey(
665  sourceStartIndexX)) {
666  return true;
667  }
668  if (!termX.containsKey(sourceStartIndex)) {
669  Map<Integer, Boolean> newElt = new HashMap<Integer, Boolean>();
670  newElt.put(sourceStartIndexX, true);
671  termX.put(sourceStartIndex, newElt);
672  } else {
673  termX.get(sourceStartIndex).put(sourceStartIndexX, true);
674  }
675  }
676  return false;
