84  bool run ( Data& d ) {
85  // Assigns keys to fetch feature weights appropriately
86  // This keeps all current options available
87  // User can work with lm.featureweights + grammar.featureweights
88  // or only featureweights, as a concatenation of the two previous
89  // (language model features first)
90  // Probably lm.featureweights and grammar.featureweights will be deprecated
91  // at some point.
92  std::string lmFeatureweights = HifstConstants::kLmFeatureweights;
93  std::string grammarFeatureweights = HifstConstants::kGrammarFeatureweights;
94  unsigned grammarFeatureweightOffset = 0;
95  if (rg_.get<std::string> (HifstConstants::kFeatureweights) != "") {
96  grammarFeatureweights = lmFeatureweights = HifstConstants::kFeatureweights;
97  grammarFeatureweightOffset = rg_.getVectorString (
98  HifstConstants::kLmLoad).size();
99  }
100  boost::scoped_ptr < LoadGrammar> grammartask
101  ( new LoadGrammar ( rg_, grammarFeatureweights, grammarFeatureweightOffset ) );
102  grammartask->appendTask
103  ( LoadWordMap::init ( rg_ , HifstConstants::kPreproWordmapLoad , true ) )
104  ( LoadWordMap::init ( rg_ , HifstConstants::kPostproWordmapLoad ) )
105  //TODO: add an option to ensure that postpro.wordmap uses this one
106  //, but then reversed wordmap is forced
107  ( LoadWordMap::init ( rg_ , HifstConstants::kLmWordmap , true) )
108  ( new LoadUnimap ( rg_ , HifstConstants::kRecaserUnimapLoad ) )
109  ( new PrePro ( rg_ ) )
110  ( new LoadLanguageModel ( rg_
111  , HifstConstants::kLmLoad
112  , lmFeatureweights ) )
113  ( new LoadLanguageModel ( rg_
114  , HifstConstants::kHifstLocalpruneLmLoad
115  , HifstConstants::kHifstLocalpruneLmFeatureweights
116  , HifstConstants::kHifstLocalpruneLmWordpenalty) )
117  ( new LoadLanguageModel ( rg_
118  , HifstConstants::kRecaserLmLoad
119  , HifstConstants::kRecaserLmFeatureweight
120  , HifstConstants::kRecaserLmWps
121  , HifstConstants::kRecaserLmWordmap ) )
122  ( new PatternsToInstances ( rg_ ) )
123  ( ReferenceFilter::init ( rg_ ) )
124  ( new SentenceSpecificGrammar ( rg_ ) )
125  ( new Parse ( rg_ ) )
126  ( new HiFST ( rg_ ) )
127  ( WriteFst::init ( rg_ , HifstConstants::kHifstLatticeStore ) )
128  ( new Recase ( rg_
129  , HifstConstants::kHifstLatticeStore
130  , HifstConstants::kPostproInput
131  , HifstConstants::kRecaserLmLoad ) )
132  ( WriteFst::init ( rg_ , HifstConstants::kRecaserOutput,
133  HifstConstants::kPostproInput ) )
134  ( new PostPro ( rg_ ) )
135  ( new HifstStats ( rg_ ) )
136  ;
137  bool finished = false;
138  oszfstream *fileoutput = NULL;
139  if ( textoutput_ != "" ) {
140  fileoutput = new oszfstream ( textoutput_ );
141  }
142  for ( ucam::util::IntRangePtr ir (ucam::util::IntRangeFactory ( rg_ ) );
143  !ir->done ();
144  ir->next () ) {
145  d.sidx = ir->get ();
146  d.filters.clear();
147  boost::scoped_ptr<std::string> aux ( new std::string ( "" ) );
148  d.translation = aux.get();
149  //Move to whichever next sentence and read
150  finished = fastforwardread_ ( d.sidx , &d.originalsentence );
151  boost::algorithm::trim (d.originalsentence);
152  if (finished && d.originalsentence == "" ) break;
153  FORCELINFO ( "=====Translate sentence " << d.sidx << ":" <<
154  d.originalsentence );
155  grammartask->chainrun ( d ); //Run translation!
156  if ( fileoutput != NULL )
157  *fileoutput << *d.translation << endl;
158  if ( finished ) break;
159  }
160  if ( fileoutput != NULL )
161  delete fileoutput;
162  return false;
163  };
164 
166  inline bool operator() () {
167  Data d;
168  return run ( d );
169  }
170 
171  private:
172 
173  DISALLOW_COPY_AND_ASSIGN ( SingleThreadedHifstTask );

300  ( new Recase ( rg_ ,
301  HifstConstants::kHifstLatticeStore,

390  bool operator () ( socket_ptr sock, Data& d ) {
392  LINFO ( "Init new taskdata..." );
393  boost::scoped_ptr<Data> mydata ( new Data );
394  mydata->grammar = d.grammar;
395  mydata->klm = d.klm;
396  mydata->filters.clear();
397  if ( d.fsts.find ( HifstConstants::kRecaserUnimapLoad ) != d.fsts.end() )
398  mydata->fsts[HifstConstants::kRecaserUnimapLoad] =
399  d.fsts[HifstConstants::kRecaserUnimapLoad];
400  mydata->recasingvcblm = d.recasingvcblm;
401  mydata->wm = d.wm;
402  LINFO ("Number of wordmaps... " << mydata->wm.size() );
403  try {
404  char data[max_length + 1];
405  std::size_t query_length = 0;
406  std::size_t query_length1 = boost::asio::read ( *sock,
407  boost::asio::buffer ( &query_length, sizeof ( std::size_t ) ) );
408  std::size_t query_length2 = boost::asio::read ( *sock,
409  boost::asio::buffer ( data, query_length ) );
410  data[query_length2] = 0;
411  mydata->originalsentence = data;
412  FORCELINFO ( "Query to translate: " << mydata->originalsentence );
413  boost::scoped_ptr<std::string> translation ( new std::string );
414  mydata->translation = translation.get();
415  this->operator() ( *mydata );
416  char datasend[max_length + 1];
417  //Send data
418  strcpy ( datasend, ( char * ) translation.get()->c_str() );
419  std::size_t length = strlen ( datasend );
420  boost::asio::write ( *sock, boost::asio::buffer ( &length,
421  sizeof ( std::size_t ) ) );
422  FORCELINFO ( "Sending:" << datasend );
423  boost::asio::write ( *sock, boost::asio::buffer ( datasend, length ) );
424  sock->close();
425  } catch ( std::exception& e ) {
426  std::cerr << "Exception in thread! " << e.what() << "\n";
427  }
428  return true;
429  };
