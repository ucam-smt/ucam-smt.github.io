102  private static class MergeReducer extends
103  Reducer<RuleWritable, AlignmentAndFeatureMap, Text, TargetFeatureList> {
104 
105  private TargetFeatureList list = new TargetFeatureList();
106 
107  private Text source = new Text();
108 
109  @Override
110  protected void reduce(RuleWritable key,
111  Iterable<AlignmentAndFeatureMap> values,
112  Context context) throws IOException, InterruptedException {
113  // First rule!
114  if (source.getLength() == 0) {
115  source.set(key.getSource());
116  }
117  if (!source.equals(key.getSource())) {
118  context.write(source, list);
119  list.clear();
120  source.set(key.getSource());
121  }
122  AlignmentAndFeatureMap alignmentAndFeatures = new AlignmentAndFeatureMap();
123  for (AlignmentAndFeatureMap value : values) {
124  alignmentAndFeatures.merge(value);
125  }
126  list.add(Pair.createPair(new Text(key.getTarget()),
127  alignmentAndFeatures));
128  }
129 
130  @Override
131  protected void cleanup(Context context) throws IOException,
132  InterruptedException {
133  super.cleanup(context);
134  context.write(source, list);
135  }
136  }

174  public int run(String[] args) throws IllegalArgumentException,
175  IllegalAccessException, IOException, ClassNotFoundException,
176  InterruptedException {
177  MergeJobParameters params = new MergeJobParameters();
178  JCommander cmd = new JCommander(params);
179 
180  try {
181  cmd.parse(args);
182  Configuration conf = getConf();
183  Util.ApplyConf(cmd, "", conf);
184  Job job = getJob(conf);
185 
186  String[] featurePathNames = params.input_features.split(",");
187  Path[] featurePaths = StringUtils.stringToPath(featurePathNames);
188  for (Path featurePath : featurePaths) {
189  MultipleInputs.addInputPath(job, featurePath, SequenceFileInputFormat.class, MergeFeatureMapper.class);
190  }
191  Path rulePath = new Path(params.input_rules);
192  MultipleInputs.addInputPath(job, rulePath,
193  SequenceFileInputFormat.class, MergeRuleMapper.class);
194 
195  FileOutputFormat.setOutputPath(job, new Path(params.output));
196 
197  return job.waitForCompletion(true) ? 0 : 1;
198  } catch (ParameterException e) {
199  System.err.println(e.getMessage());
200  cmd.usage();
201  }
202 
203  return 1;
204  }
