67 template<class FromArc, class ToArc, class WeightMakerFunctorT >
68 class GenericWeightMapper {
69  public:
70  explicit GenericWeightMapper ( const WeightMakerFunctorT& mw ) : mw_ ( mw ) {};
71 
72  ToArc operator() ( const FromArc& arc ) const {
73  if ( arc.weight == FromArc::Weight::Zero() )
74  return ToArc ( arc.ilabel, arc.olabel, ToArc::Weight::Zero(), arc.nextstate );
75  if ( arc.weight == FromArc::Weight::One() )
76  return ToArc ( arc.ilabel, arc.olabel, ToArc::Weight::One(), arc.nextstate );
77  return ToArc ( arc.ilabel, arc.olabel, mw_ ( arc.weight ), arc.nextstate );
78  }
79 
80  MapSymbolsAction InputSymbolsAction() const {
81  return MAP_COPY_SYMBOLS;
82  }
83  MapSymbolsAction OutputSymbolsAction() const {
84  return MAP_COPY_SYMBOLS;
85  }
86  MapFinalAction FinalAction() const {
87  return MAP_NO_SUPERFINAL;
88  }
89  uint Properties ( uint props ) const {
90  return ( props & kWeightInvariantProperties ) | kUnweighted;
91  }
92 
93  private:
95  WeightMakerFunctorT mw_;
96 };

108  epsilons_ (epsilons),
109  wp_ ( wp ) {

139 struct StdToLexStdMapper {
140  typedef StdArc FromArc;
141  typedef LexStdArc ToArc;
142  typedef ToArc::Weight W;
143 
144  //If i==0, copies to both weights. If i=1 or 2, only copies stdweight to first or second weight respectively, leaving a Weight::One() on the other.
145  explicit StdToLexStdMapper (int i = 0) : i_ (i) {
146  CHECK (!i || i == 1 || i == 2);
147  }
148 
149  LexStdArc operator() (const StdArc& arc) const {
150  FromArc::Weight w1 = (i_ == 1
151  || !i_) ? arc.weight.Value() : FromArc::Weight::One();
152  FromArc::Weight w2 = (i_ == 2
153  || !i_) ? arc.weight.Value() : FromArc::Weight::One();
154  W w (w1, w2);
155  return LexStdArc (arc.ilabel, arc.olabel, w, arc.nextstate);
156  }

174 struct LexStdToStdMapper {
175  typedef LexStdArc FromArc;
176  typedef StdArc ToArc;
177  typedef ToArc::Weight W;
178 
179  explicit LexStdToStdMapper() : i_ (0) {}
180 
181  explicit LexStdToStdMapper (int i) : i_ (i) {
182  CHECK (i == 1 || i == 2 || i == 0);
183  }
184 
185  StdArc operator() (const LexStdArc& arc) const {
186  W w;
187  if (i_ == 0)
188  w = Times ( arc.weight.Value1(), arc.weight.Value2() );
189  if (i_ == 1)
190  w = arc.weight.Value1();
191  if (i_ == 2)
192  w = arc.weight.Value2();
193  return StdArc (arc.ilabel, arc.olabel, w, arc.nextstate);
194  }
