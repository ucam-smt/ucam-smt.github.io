79  if ( isTerminal ( source[k] ) ) continue;

82  nt = 0;

85  USER_CHECK ( partial_mappings.find ( translation[k] ) != partial_mappings.end(),
86  "RHS source and RHS target do not match!" );
87  ( *mappings ) [partial_mappings[translation[k]]] = nt++;
88  }
