28 /*
29  * \brief Main function.
30  * \param argc: Number of command-line program options.
31  * \param argv: Actual program options.
32  * \remarks Main function. Runs applylm tool -- apply language models using kenlm
33  * Multithreaded version included, triggered by nthread option
34  */
35 
36 int main ( int argc, const char* argv[] ) {
37  using ucam::util::Runner2;
38  using ucam::fsttools::SingleThreadedApplyLanguageModelTask;
39  using ucam::fsttools::MultiThreadedApplyLanguageModelTask;
40  using ucam::fsttools::ApplyLMData;
41  ucam::util::initLogger ( argc, argv );
42  FORCELINFO ( argv[0] << " starts!" );
43  ucam::util::RegistryPO rg ( argc, argv );
44  FORCELINFO ( rg.dump ( "CONFIG parameters:\n====================="
45  , "=====================" ) );
46  // Detect here kenlm binary type
47  // it's a bit ugly this way of initializing the correct kenlm handler
48  lm::ngram::ModelType kenmt = ucam::util::detectkenlm (rg.getVectorString (
49  HifstConstants::kLmLoad, 0) );
50  if (rg.get<std::string> (HifstConstants::kHifstSemiring.c_str() ) ==
51  HifstConstants::kHifstSemiringStdArc )
52  switch (kenmt) {
53  case lm::ngram::PROBING:
54  ( Runner2<SingleThreadedApplyLanguageModelTask<ApplyLMData<lm::ngram::ProbingModel> >
55  , MultiThreadedApplyLanguageModelTask< ApplyLMData<lm::ngram::ProbingModel> > >
56  ( rg ) ) ();
57  break;
58  case lm::ngram::REST_PROBING:
59  ( Runner2<SingleThreadedApplyLanguageModelTask<ApplyLMData<lm::ngram::RestProbingModel>
60  , lm::ngram::RestProbingModel >
61  , MultiThreadedApplyLanguageModelTask< ApplyLMData<lm::ngram::RestProbingModel>
62  , lm::ngram::RestProbingModel > > ( rg ) ) ();
63  break;
64  case lm::ngram::TRIE:
65  ( Runner2<SingleThreadedApplyLanguageModelTask<ApplyLMData<lm::ngram::TrieModel>, lm::ngram::TrieModel >
66  , MultiThreadedApplyLanguageModelTask< ApplyLMData<lm::ngram::TrieModel>, lm::ngram::TrieModel > >
67  ( rg ) ) ();
68  break;
69  case lm::ngram::QUANT_TRIE:
70  ( Runner2<SingleThreadedApplyLanguageModelTask<ApplyLMData<lm::ngram::QuantTrieModel>, lm::ngram::QuantTrieModel >
71  , MultiThreadedApplyLanguageModelTask< ApplyLMData<lm::ngram::QuantTrieModel>, lm::ngram::QuantTrieModel > >
72  ( rg ) ) ();
73  break;
74  case lm::ngram::ARRAY_TRIE:
75  ( Runner2<SingleThreadedApplyLanguageModelTask<ApplyLMData<lm::ngram::ArrayTrieModel>, lm::ngram::ArrayTrieModel >
76  , MultiThreadedApplyLanguageModelTask< ApplyLMData<lm::ngram::ArrayTrieModel>, lm::ngram::ArrayTrieModel > >
77  ( rg ) ) ();
78  break;
79  case lm::ngram::QUANT_ARRAY_TRIE:
80  ( Runner2<SingleThreadedApplyLanguageModelTask<ApplyLMData<lm::ngram::QuantArrayTrieModel>, lm::ngram::QuantArrayTrieModel >
81  , MultiThreadedApplyLanguageModelTask< ApplyLMData<lm::ngram::QuantArrayTrieModel>, lm::ngram::QuantArrayTrieModel > >
82  ( rg ) ) ();
83  break;
84  }
85  else if (rg.get<std::string> (HifstConstants::kHifstSemiring) ==
86  HifstConstants::kHifstSemiringLexStdArc )
87  switch (kenmt) {
88  case lm::ngram::PROBING:
89  ( Runner2<SingleThreadedApplyLanguageModelTask<ApplyLMData<lm::ngram::ProbingModel>, lm::ngram::ProbingModel, fst::LexStdArc >
90  , MultiThreadedApplyLanguageModelTask< ApplyLMData<lm::ngram::ProbingModel>, lm::ngram::ProbingModel, fst::LexStdArc > >
91  ( rg ) ) ();
92  break;
93  case lm::ngram::REST_PROBING:
94  ( Runner2<SingleThreadedApplyLanguageModelTask<ApplyLMData<lm::ngram::RestProbingModel>, lm::ngram::RestProbingModel, fst::LexStdArc >
95  , MultiThreadedApplyLanguageModelTask< ApplyLMData<lm::ngram::RestProbingModel>, lm::ngram::RestProbingModel , fst::LexStdArc > >
96  ( rg ) ) ();
97  break;
98  case lm::ngram::TRIE:
99  ( Runner2<SingleThreadedApplyLanguageModelTask<ApplyLMData<lm::ngram::TrieModel>, lm::ngram::TrieModel, fst::LexStdArc >
100  , MultiThreadedApplyLanguageModelTask< ApplyLMData<lm::ngram::TrieModel>, lm::ngram::TrieModel, fst::LexStdArc > >
101  ( rg ) ) ();
102  break;
103  case lm::ngram::QUANT_TRIE:
104  ( Runner2<SingleThreadedApplyLanguageModelTask<ApplyLMData<lm::ngram::QuantTrieModel>, lm::ngram::QuantTrieModel, fst::LexStdArc >
105  , MultiThreadedApplyLanguageModelTask< ApplyLMData<lm::ngram::QuantTrieModel>, lm::ngram::QuantTrieModel, fst::LexStdArc > >
106  ( rg ) ) ();
107  break;
108  case lm::ngram::ARRAY_TRIE:
109  ( Runner2<SingleThreadedApplyLanguageModelTask<ApplyLMData<lm::ngram::ArrayTrieModel>, lm::ngram::ArrayTrieModel, fst::LexStdArc >
110  , MultiThreadedApplyLanguageModelTask< ApplyLMData<lm::ngram::ArrayTrieModel>, lm::ngram::ArrayTrieModel, fst::LexStdArc > >
111  ( rg ) ) ();
112  break;
113  case lm::ngram::QUANT_ARRAY_TRIE:
114  ( Runner2<SingleThreadedApplyLanguageModelTask<ApplyLMData<lm::ngram::QuantArrayTrieModel>, lm::ngram::QuantArrayTrieModel, fst::LexStdArc >
115  , MultiThreadedApplyLanguageModelTask< ApplyLMData<lm::ngram::QuantArrayTrieModel>, lm::ngram::QuantArrayTrieModel, fst::LexStdArc > >
116  ( rg ) ) ();
117  break;
118  }
119  else {
120  LERROR ("Semiring not supported (semiring => lexstdarc,stdarc)");
121  }
122  FORCELINFO ( argv[0] << " ends!" );
123 };
