89  inline const std::string getRHSSource ( std::size_t idx ) {
90  if ( extrarules.find ( idx ) == extrarules.end() )
91  return grammar->getRHSSource ( idx );

98  inline const std::string getRHSSource ( std::size_t idx, uint rulepos ) {
99  if ( extrarules.find ( idx ) == extrarules.end() )
100  return grammar->getRHSSource ( idx, rulepos );
101  std::size_t pos = extrarules[idx].find_first_of ( " " );
102  std::size_t j = pos , jold;
103  for ( uint k = 0; k <= rulepos; ++k ) {
104  jold = j;
105  j = extrarules[idx].find_first_of ( "_ ", jold + 1 );
106  if ( j == std::string::npos )
107  if ( rulepos ) return "";
108  }
109  return extrarules[idx].substr ( jold + 1, j - jold - 1 );
110  };

174  inline const bool isPhrase ( std::size_t idx ) {
175  if ( extrarules.find ( idx ) == extrarules.end() )
176  return grammar->isPhrase ( idx );
177  std::size_t pos0 = extrarules[idx].find_first_of ( " " ) + 1;
178  std::size_t pos = extrarules[idx].find_first_of ( " ", pos0 );
179  for ( const char *c = extrarules[idx].c_str() + pos0;
180  c <= extrarules[idx].c_str() + pos; ++c )
181  if ( *c >= 'A' && *c <= 'Z' ) return false; //has non-terminals.
182  return true; //pure phrase.
183  };

219 inline bool phraseIsTerminalWord ( const std::string& phrase ) {
220  for ( uint k = 0; k < phrase.size(); ++k ) {
221  if ( phrase[k] >= 'A' && phrase[k] <= 'Z' ) return false;
222  else if ( phrase[k] == '_' ) return false;
223  }
224  return true;
