66  for ( uint k = 2; k < fsts_.size(); ++k ) Union ( & ( *aux ), *fsts_[k] );
67  aux2 = new VectorFst<Arc> ( *aux );
68  } else aux2 = new VectorFst<Arc> ( *fsts_[0] );
