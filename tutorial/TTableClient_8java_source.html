85  private double[] query(Collection<List<Integer>> query) throws IOException {
86 
87  try (Socket clientSocket = new Socket(hostName, port);
88  DataInputStream in = new DataInputStream(
89  new BufferedInputStream(clientSocket.getInputStream(),

93  clientSocket.getOutputStream(),

96  double[] result = new double[query.size()];
97  out.writeInt(query.size());
98  for (List<Integer> queryKey : query) {
99  out.writeInt(queryKey.get(0));
100  out.writeInt(queryKey.get(1));
101  out.writeInt(queryKey.get(2));
102  }
103  out.flush();
104 
105  for (int i = 0; i < query.size(); ++i) {
106  double val = in.readDouble();
107  result[i] = val;
108  }
109  return result;
110  } catch (java.net.ConnectException e) {
111  String message = "Failed to connect to ttable server. Hostname: "
112  + hostName + " Port: " + port;

117  public void queryRules(
118  List<Pair<RuleWritable, AlignmentAndFeatureMap>> rules)
119  throws IOException {
120  for (Pair<RuleWritable, AlignmentAndFeatureMap> entry : rules) {
121  RuleWritable key = entry.getFirst();
122  prob.buildQuery(key, mapping.length, wordAlignments);
123  }
124  List<List<Integer>> keys = new ArrayList<>(wordAlignments.keySet());
125  double[] results = query(keys);
126  wordAlignments.clear();
127  for (int i = 0; i < keys.size(); ++i) {
128  if (results[i] != Double.MAX_VALUE) {
129  wordAlignments.put(keys.get(i), results[i]);
130  }
131  }
132  for (Pair<RuleWritable, AlignmentAndFeatureMap> entry : rules) {
133  RuleWritable key = entry.getFirst();
134  AlignmentAndFeatureMap features = entry.getSecond();
135  for (int j = 0; j < mapping.length; ++j) {
136  double lexProb = prob.value(key, (byte) j, wordAlignments);
137  if (features.getSecond().containsKey(mapping[j])) {
138  throw new RuntimeException(
139  "FeatureMap already contains entry for "
140  + mapping[j] + " " + key + " " + features);
141  }
142  features.getSecond().put(mapping[j], lexProb);
143  }
144  }
145  }
