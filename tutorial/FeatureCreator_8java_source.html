54  public FeatureCreator(Configuration conf) {
55  this.conf = conf;
56  features = new HashMap<String, Feature>();
57  features.put("source2target_probability",
58  new Source2TargetProbability());
59  features.put("target2source_probability",
60  new Target2SourceProbability());
61  features.put("word_insertion_penalty", new WordInsertionPenalty());
62  features.put("rule_insertion_penalty", new RuleInsertionPenalty());
63  features.put("glue_rule", new GlueRule());
64  features.put("insert_scale", new InsertScale());
65  features.put("rule_count_1", new RuleCount1());
66  features.put("rule_count_2", new RuleCount2());
67  features.put("rule_count_greater_than_2", new RuleCountGreaterThan2());
68  features.put("source2target_lexical_probability",
69  new Source2TargetLexicalProbability());
70  features.put("target2source_lexical_probability",
71  new Target2SourceLexicalProbability());
72  String provenance = conf.get("provenance");
73  if (provenance != null) {
74  String[] provenances = provenance.split(",");
75  for (String prov : provenances) {
76  features.put("provenance_source2target_probability-" + prov,
77  new ProvenanceSource2TargetProbability(prov));
78  features.put("provenance_target2source_probability-" + prov,
79  new ProvenanceTarget2SourceProbability(prov));
80  features.put("provenance_source2target_lexical_probability-"
81  + prov, new ProvenanceSource2TargetLexicalProbability(
82  prov));
83  features.put("provenance_target2source_lexical_probability-"
84  + prov, new ProvenanceTarget2SourceLexicalProbability(
85  prov));
86  }
87  }
88  String selectedFeaturesString = conf.get("features");
89  if (selectedFeaturesString == null) {
90  System.err
91  .println("Missing property " + "'features' in the config");
92  System.exit(1);
93  }
94  selectedFeatures = selectedFeaturesString.split(",");
95  // initial feature index is zero, then increments with the number of
96  // features of each feature type. nextFeatureIndex is used to prevent
97  // conf to be overwritten before being used.
98  int featureIndex = 0, nextFeatureIndex = 0;
99  for (String selectedFeature : selectedFeatures) {
100  if (selectedFeature
101  .equals("provenance_source2target_lexical_probability")
102  || selectedFeature
103  .equals("provenance_target2source_lexical_probability")
104  || selectedFeature
105  .equals("provenance_source2target_probability")
106  || selectedFeature
107  .equals("provenance_target2source_probability")) {
108  for (String prov : conf.get("provenance").split(",")) {
109  featureIndex = nextFeatureIndex;
110  nextFeatureIndex += features.get(
111  selectedFeature + "-" + prov).getNumberOfFeatures(
112  conf);
113  conf.setInt(selectedFeature + "-" + prov, featureIndex);
114  }
115  } else {
116  featureIndex = nextFeatureIndex;
117  nextFeatureIndex += features.get(selectedFeature)
118  .getNumberOfFeatures(conf);
119  // TODO change this (maybe something like
120  // conf.setInt(selectedFeature+"-"+"startindex", featureIndex))
121  conf.setInt(selectedFeature + "-nbfeats",
122  features.get(selectedFeature).getNumberOfFeatures(conf));
123  conf.setInt(selectedFeature, featureIndex);
124  }
125  }
126  }

128  private Map<Integer, Double> createFeatures(String featureName,
129  RuleWritable rule, FeatureMap mapReduceFeatures,
130  EnumRuleType ruleType) {
131  switch (ruleType) {
132  case GLUE:
133  return features.get(featureName).valueGlue(new Rule(rule),
134  mapReduceFeatures, conf);
135  case ASCII_OOV_DELETE:
136  return features.get(featureName).valueAsciiOovDeletion(
137  new Rule(rule), mapReduceFeatures, conf);
138  default:
139  return features.get(featureName).value(new Rule(rule),
140  mapReduceFeatures, conf);
141  }

160  public void writeRule(RuleWritable rule, AlignmentAndFeatureMap features,
161  EnumRuleType ruleType, BufferedWriter out) {
162  SortedMap<Integer, Double> processedFeatures = createFeatures(rule,
163  features.getSecond(), ruleType);
164  StringBuilder res = new StringBuilder();
165  res.append(rule);
166  for (int featureIndex : processedFeatures.keySet()) {
167  double featureValue = processedFeatures.get(featureIndex);
168  // one-based index
169  int index = featureIndex + 1;
170  if (Math.floor(featureValue) == featureValue) {
171  int featureValueInt = (int) featureValue;
172  res.append(String.format(" %d@%d", featureValueInt, index));
173  } else {
174  res.append(String.format(" %f@%d", featureValue, index));
175  }
176  }
177  res.append("\n");
178  synchronized (out) {
179  try {
180  out.write(res.toString());
181  } catch (IOException e) {
182  e.printStackTrace();
183  System.exit(1);
184  }
185 
186  }
187  }

189  private SortedMap<Integer, Double> createFeatures(RuleWritable rule,
190  FeatureMap map, EnumRuleType ruleType) {
191  SortedMap<Integer, Double> allFeatures = new TreeMap<>();
192  for (String featureName : selectedFeatures) {
193  if (featureName
194  .equals("provenance_source2target_lexical_probability")
195  || featureName
196  .equals("provenance_target2source_lexical_probability")
197  || featureName
198  .equals("provenance_source2target_probability")
199  || featureName
200  .equals("provenance_target2source_probability")) {
201  for (String provenance : conf.get("provenance").split(",")) {
202  Map<Integer, Double> features = createFeatures(featureName
203  + "-" + provenance, rule, map, ruleType);
204  for (Integer featureIndex : features.keySet()) {
205  if (allFeatures.containsKey(featureIndex)) {
206  System.err
207  .println("ERROR: feature index already exists: "
208  + featureIndex);
209  System.exit(1);
210  }
211  Number feature = features.get(featureIndex);
212  allFeatures.put(featureIndex, feature.doubleValue());
213  }
214  }
215  } else {
216  Map<Integer, Double> features = createFeatures(featureName,
217  rule, map, ruleType);
218  for (Integer featureIndex : features.keySet()) {
219  if (allFeatures.containsKey(featureIndex)) {
220  System.err
221  .println("ERROR: feature index already exists: "
222  + featureIndex);
223  System.exit(1);
224  }
225  Number feature = features.get(featureIndex);
226  allFeatures.put(featureIndex, feature.doubleValue());
227  }
228  }
229  }
230  return allFeatures;
231  }
