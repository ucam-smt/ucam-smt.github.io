81  private static Map<Byte, String> getProvenanceMap(Configuration config) {

95  public static int[] getFeatureIndex(String featureName, String suffix,
96  Configuration conf) {
97  Map<Byte, String> id2Prov = ProvenanceCountMap.getProvenanceMap(conf);
98  int[] mapping = new int[id2Prov.size() + 1];
99  // If there is a problem with the mappings then we should know about it
100  Arrays.fill(mapping, Integer.MIN_VALUE);
101  String featureIndex = conf.get(featureName + suffix);
102  try {
103  mapping[0] = Integer.parseInt(featureIndex);
104  } catch (NumberFormatException e) {
105  throw new RuntimeException("Unable to parse feature: "
106  + featureIndex, e);
107  }
108  for (Entry<Byte, String> entry : id2Prov.entrySet()) {
109  String provenance = entry.getValue();
110  String configKey = "provenance_" + featureName + "-" + provenance
111  + suffix;
112  featureIndex = conf.get(configKey);
113  if (featureIndex == null) {
114  throw new RuntimeException(configKey);
115  }
116  try {
117  mapping[entry.getKey().intValue()] = Integer
118  .parseInt(featureIndex);
119  } catch (NumberFormatException e) {
120  throw new RuntimeException("Unable to parse feature: "
121  + configKey, e);
122  }
123  }
124  return mapping;
125  }
