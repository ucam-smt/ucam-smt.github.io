86  void write ( ucam::util::oszfstream& o ) {
87  for ( unordered_map<std::string, std::vector<timeb> >::iterator itx =
88  time2.begin(); itx != time2.end(); itx++ ) {
89  USER_CHECK ( time1[itx->first].size() == itx->second.size(),
90  "Mismatch with SpeedStats (each setTimeStart needs a setTimeEnd" );
91  o << std::setw ( 30 ) << setiosflags ( std::ios::right ) << itx->first << ":";
92  int64 fst_time = 0;
93  for ( unsigned k = 0; k < itx->second.size() - 1 ; ++k ) {
94  fst_time += ( itx->second[k].time - time1[itx->first][k].time ) * 1000;
95  fst_time += itx->second[k].millitm - time1[itx->first][k].millitm;
96  }
97  int64 last_time = ( itx->second[itx->second.size() - 1].time -
98  time1[itx->first][itx->second.size() - 1].time ) * 1000
99  + itx->second[itx->second.size() - 1].millitm -
100  time1[itx->first][itx->second.size() - 1].millitm;
101  fst_time += last_time;
102  o << std::setw ( 10 ) << last_time;
103  o << std::setw ( 10 ) << fst_time ;
104  o << " (" << itx->second.size() << " times )" << endl;
105  }
106  }
