142  void prune() {
143  fst::VectorFst<Arc> pruned, dweight;
144  if ( useshortestpath_ ) {
145  LINFO ( "Using shortestpath with reference lattice n=" << shortestpath_ );
146  fst::ShortestPath<Arc> ( *referencesubstringlattice_, &pruned, shortestpath_,
147  true );
148  }
149  if ( useweight_ ) {
150  LINFO ( "Pruning reference lattice with weight=" << weight_ );
151  fst::MakeWeight<Arc> mw;
152  fst::Prune<Arc> ( referencesubstringlattice_, mw ( weight_ ) );
153  LINFO ( "Weighted determinization with weight=" << weight_ );
154  fst::DeterminizeOptions<Arc> dopts;
155  dopts.weight_threshold = mw ( weight_ );
156  fst::Determinize<Arc> ( *referencesubstringlattice_, &dweight, dopts );
157  }
158  if ( useshortestpath_ || useweight_ )
159  *referencesubstringlattice_ = ( fst::UnionFst<Arc> ( pruned,
160  dweight ) ); //should work either way
161  };
162 
167  void reduce() {
168  fst::Map<Arc> ( referencesubstringlattice_,
169  fst::RmWeightMapper<Arc>() ); //finally take weights away, so composition scores not affected.
170  fst::Determinize<Arc> ( fst::RmEpsilonFst<Arc> ( *referencesubstringlattice_ ),
171  referencesubstringlattice_ );
172  fst::Minimize<Arc> ( referencesubstringlattice_ );
173  }
174 
180  void build ( const std::string& file ) {
181  if ( file == "" ) return;
182  if ( built_ && oldfile_ == file ) return;
183  oldfile_ = file;
184  unload();
185  vocabulary_.clear();
186  referencesubstringlattice_ = fst::VectorFstRead<Arc> ( file );
187  prune();
188  reduce();
189  referencelattice_ = new fst::VectorFst<Arc> ( *referencesubstringlattice_ );
190  if ( !disablesubstring_ ) {
191  LINFO ( "building substring reference" );
192  fst::buildSubstringTransducer<Arc>
193  ( referencesubstringlattice_ ); //now we build a sstransducer
194  } else {
195  LWARN ( "Using lattice as-is... substring will not be implemented!!!" );
196  }
197  fst::ArcSort<Arc> ( referencesubstringlattice_, fst::ILabelCompare<Arc>() );
198  fst::extractTargetVocabulary<Arc> ( *referencesubstringlattice_, &vocabulary_ );
199  built_ = true;
200  };
201 
203  void unload ( void ) {
204  if ( referencesubstringlattice_ ) delete referencesubstringlattice_;
205  referencesubstringlattice_ = NULL;
206  built_ = false;
207  if ( referencelattice_ ) delete referencelattice_;
208  referencelattice_ = NULL;
209  }
210 
212  void write ( Data& d ) {
213  if ( writereferencelatticefile_ ( d.sidx ) != "" )
214  fst::FstWrite ( *referencesubstringlattice_,
215  writereferencelatticefile_ ( d.sidx ) );
216  };
217 
219  bool run ( Data& d ) {
220  LINFO ( "build reference filter from lattice=" << translationlatticefile_.get (
221  d.sidx ) );
222  build ( translationlatticefile_.get ( d.sidx ) );
223  if ( referencesubstringlattice_ ) {
224  d.filters.push_back ( referencesubstringlattice_ );
225  d.tvcb = vocabulary_;
226  d.fsts[referencelatticekey_] = referencelattice_;
227  LINFO ( "Done! Full lattice stored with key="
228  << referencelatticekey_
229  << ", NS=" << static_cast<fst::VectorFst<Arc> *>
230  ( d.fsts[referencelatticekey_])->NumStates() );
231  }
232  write ( d );
233  return false;
234  };
