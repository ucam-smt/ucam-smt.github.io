149  void prune() {
150  fst::VectorFst<Arc> pruned, dweight;
151  if ( useshortestpath_ ) {
152  LINFO ( "Using shortestpath with reference lattice n=" << shortestpath_ );
153  fst::ShortestPath<Arc> ( *referencesubstringlattice_, &pruned, shortestpath_,
154  true );
155  }
156  if ( useweight_ ) {
157  LINFO ( "Pruning reference lattice with weight=" << weight_ );
158  fst::MakeWeight<Arc> mw;
159  fst::Prune<Arc> ( referencesubstringlattice_, mw ( weight_ ) );
160  LINFO ( "Weighted determinization with weight=" << weight_ );
161  fst::DeterminizeOptions<Arc> dopts;
162  dopts.weight_threshold = mw ( weight_ );
163  fst::Determinize<Arc> ( *referencesubstringlattice_, &dweight, dopts );
164  }
165  if ( useshortestpath_ || useweight_ )
166  *referencesubstringlattice_ = ( fst::UnionFst<Arc> ( pruned,
167  dweight ) ); //should work either way
168  };
169 
174  void reduce() {
175  fst::Map<Arc> ( referencesubstringlattice_,
176  fst::RmWeightMapper<Arc>() ); //finally take weights away, so composition scores not affected.
177  fst::Determinize<Arc> ( fst::RmEpsilonFst<Arc> ( *referencesubstringlattice_ ),
178  referencesubstringlattice_ );
179  fst::Minimize<Arc> ( referencesubstringlattice_ );
180  }

187  void build ( const std::string& file ) {
188  if ( file == "" ) return;
189  if ( built_ && oldfile_ == file ) return;
190  oldfile_ = file;
191  unload();
192  vocabulary_.clear();
193  loadLattice(file);
194  prune();
195  reduce();
196 
197  referencelattice_ = new fst::VectorFst<Arc> ( *referencesubstringlattice_ );
198 
199  if ( !disablesubstring_ ) {
200  LINFO ( "building substring reference" );
201  fst::buildSubstringTransducer<Arc>
202  ( referencesubstringlattice_ ); //now we build a sstransducer
203  } else {
204  LWARN ( "Using lattice as-is... substring will not be implemented!!!" );
205  }
206  fst::ArcSort<Arc> ( referencesubstringlattice_, fst::ILabelCompare<Arc>() );
207  fst::extractTargetVocabulary<Arc> ( *referencesubstringlattice_, &vocabulary_ );
208  built_ = true;
209  };

247  void loadLattice(std::string const &file) {
248  using namespace fst;
249  if (translationlatticefilesemiring_ == "" ) { // use default arc
250  referencesubstringlattice_ = VectorFstRead<Arc> ( file );
251  return;
252  }
253  if (semiring_ != "tuplearc") {
254  LERROR("Conversions currently allowed only from lexstdarc,tropical TO tuplearc)");
255  exit(EXIT_FAILURE);
256  }
257  referencesubstringlattice_ = new VectorFst<Arc>;
258 
259  if (translationlatticefilesemiring_ == "lexstdarc") {
260  VectorFst<LexStdArc> *aux= VectorFstRead<LexStdArc> ( file );
261  VectorFst<TupleArc32> *vwfst = new VectorFst<TupleArc32>;
262 
263  LINFO ( "Mapping Arc Target Lattice to TupleArc32" );
264  MakeSparseVectorWeight<LexStdArc> mwcopy ( 1 );
265  typedef GenericWeightMapper<LexStdArc, TupleArc32, MakeSparseVectorWeight<LexStdArc> > WeightMapper;
266  Map ( *aux, vwfst, WeightMapper(mwcopy));
267  // bypassing template instance conversions with a reinterpret cast
268  // but this code only happens from tuplearc32 to tuplearc32.
269  referencesubstringlattice_ = reinterpret_cast<VectorFst<Arc> *>(vwfst);
270 
271  delete aux;
272  return;
273  }
