68  public Alignment(String wordAlign, SentencePair sp) {
69  s2t = new ArrayList<List<Integer>>(sp.getSource().getWords().length);
70  for (int i = 0; i < sp.getSource().getWords().length; i++) {
71  s2t.add(null);
72  }
73  s2tmax = new ArrayList<Integer>(sp.getSource().getWords().length);
74  for (int i = 0; i < sp.getSource().getWords().length; i++) {
75  s2tmax.add(Integer.MIN_VALUE);
76  }
77  s2tmin = new ArrayList<Integer>(sp.getSource().getWords().length);
78  for (int i = 0; i < sp.getSource().getWords().length; i++) {
79  s2tmin.add(Integer.MAX_VALUE);
80  }
81  t2s = new ArrayList<List<Integer>>(sp.getTarget().getWords().length);
82  for (int i = 0; i < sp.getTarget().getWords().length; i++) {
83  t2s.add(null);
84  }
85  t2smax = new ArrayList<Integer>(sp.getTarget().getWords().length);
86  for (int i = 0; i < sp.getTarget().getWords().length; i++) {
87  t2smax.add(Integer.MIN_VALUE);
88  }
89  t2smin = new ArrayList<Integer>(sp.getTarget().getWords().length);
90  for (int i = 0; i < sp.getTarget().getWords().length; i++) {
91  t2smin.add(Integer.MAX_VALUE);
92  }
93  // handle empty alignment case
94  String[] links = new String[0]; // empty array, not null though
95  if (!wordAlign.isEmpty()) {
96  links = wordAlign.split("\\s+");
97  }
98  for (String link : links) {
99  String[] linkSrcposTrgpos = link.split("-");
100  if (linkSrcposTrgpos.length != 2) {
101  logger.log(Level.SEVERE, "Alignment link bad format: " + link
102  + "\n" + "word alignment: " + wordAlign + "\n"
103  + "sentencepair: " + sp);
104  System.exit(1);
105  }
106  int sourcePosition = -1, targetPosition = -1;
107  sourcePosition = Integer.parseInt(linkSrcposTrgpos[0]);
108  targetPosition = Integer.parseInt(linkSrcposTrgpos[1]);
109  if (s2t.get(sourcePosition) != null) {
110  s2t.get(sourcePosition).add(targetPosition);
111  } else {
112  List<Integer> newEntry = new ArrayList<Integer>();
113  newEntry.add(targetPosition);
114  s2t.set(sourcePosition, newEntry);
115  }
116  if (targetPosition > s2tmax.get(sourcePosition)) {
117  s2tmax.set(sourcePosition, targetPosition);
118  }
119  if (targetPosition < s2tmin.get(sourcePosition)) {
120  s2tmin.set(sourcePosition, targetPosition);
121  }
122  if (t2s.get(targetPosition) != null) {
123  t2s.get(targetPosition).add(sourcePosition);
124  } else {
125  List<Integer> newEntry = new ArrayList<Integer>();
126  newEntry.add(sourcePosition);
127  t2s.set(targetPosition, newEntry);
128  }
129  if (sourcePosition > t2smax.get(targetPosition)) {
130  t2smax.set(targetPosition, sourcePosition);
131  }
132  if (sourcePosition < t2smin.get(targetPosition)) {
133  t2smin.set(targetPosition, sourcePosition);
134  }
135  }
136  }
