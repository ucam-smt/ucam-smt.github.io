89  public Set<Rule> createSourcePatternInstances(String line) {
90  Set<Rule> res = new HashSet<>();
91  String[] parts;
92  parts = line.split(" ");
93  List<Integer> sourceSentence = new ArrayList<Integer>();
94  for (int i = 0; i < parts.length; i++) {
95  sourceSentence.add(Integer.parseInt(parts[i]));
96  List<Integer> sourcePhrase = new ArrayList<Integer>();
97  for (int j = 0; j < MAX_SOURCE_PHRASE && j < parts.length - i; j++) {
98  sourcePhrase.add(Integer.parseInt(parts[i + j]));
99  // add source phrase
100  Rule r = new Rule(sourcePhrase, new ArrayList<Integer>());
101  res.add(r);
102  }
103  }
104  Set<Rule> sourcePatternInstances = getPatternInstancesFromSourceSentence(
105  sourceSentence, sidePatterns);
106  res.addAll(sourcePatternInstances);
107  return res;
108  }

110  private Set<Rule> getPatternInstancesFromSourceSentence(
111  List<Integer> sourceSentence, List<SidePattern> sidePatterns) {
112  Set<Rule> res = new HashSet<Rule>();
113  for (SidePattern sidePattern : sidePatterns) {
114  for (int i = 0; i < sourceSentence.size(); i++) {

122  private Set<Rule> merge(Rule partialLeft, Set<Rule> partialRight) {
123  Set<Rule> res = new HashSet<Rule>();
124  List<Integer> sourceLeft = partialLeft.getSource();
125  if (partialRight.isEmpty()) {
126  res.add(new Rule(sourceLeft, new ArrayList<Integer>()));
127  return res;
128  }
129  for (Rule r : partialRight) {
130  List<Integer> merged = new ArrayList<Integer>();
131  List<Integer> sourceRight = r.getSource();
132  merged.addAll(sourceLeft);
133  merged.addAll(sourceRight);
134  res.add(new Rule(merged, new ArrayList<Integer>()));
135  }
136  return res;
137  }

139  private Set<Rule> getPatternInstancesFromSourceAndPattern2(
140  List<Integer> sourceSentence, SidePattern sidePattern,
141  int startSentenceIndex, int startPatternIndex, int nbSrcElt,
142  int nbCoveredWords) {
143  Set<Rule> res = new HashSet<Rule>();
144  if (startSentenceIndex >= sourceSentence.size()) {
145  return res;
146  }
147  if (startPatternIndex >= sidePattern.size()) {
148  return res;
149  }
150  // pattern is too big for the (rest of the) sentence, e.g. pattern wXw
151  // for the phrase 2_3
152  if (sourceSentence.size() - startSentenceIndex < sidePattern.size()
153  - startPatternIndex) {
154  return res;
155  }
156  // we already have MAX_SOURCE_ELEMENTS source elements
157  if (nbSrcElt >= MAX_SOURCE_ELEMENTS) {
158  return res;
159  }
160  // we already cover HR_MAX_HEIGHT
161  if (nbCoveredWords >= HR_MAX_HEIGHT) {
162  return res;
163  }
164  if (sourceSentence.size() - startSentenceIndex == sidePattern.size()
165  - startPatternIndex) {
166  if (nbSrcElt + sidePattern.size() - startPatternIndex > MAX_SOURCE_ELEMENTS) {
167  return res;
168  }
169  if (nbCoveredWords + sourceSentence.size() - startSentenceIndex > HR_MAX_HEIGHT) {
170  return res;
171  }
172  List<Integer> patternInstance = new ArrayList<Integer>();
173  for (int i = 0; i < sourceSentence.size() - startSentenceIndex; i++) {
174  if (sidePattern.get(startPatternIndex + i).equals("w")) {
175  patternInstance.add(sourceSentence.get(startSentenceIndex
176  + i));
177  } else {
178  patternInstance.add(Integer.parseInt(sidePattern
179  .get(startPatternIndex + i)));
180  }
181  }
182  Rule r = new Rule(patternInstance, new ArrayList<Integer>());
183  res.add(r);
184  return res;
185  }
186  List<Integer> partialPattern = new ArrayList<Integer>();
187  if (sidePattern.get(startPatternIndex).equals("w")) {
188  for (int i = startSentenceIndex; i < sourceSentence.size()
189  - (sidePattern.size() - startPatternIndex - 1)
190  && i < startSentenceIndex + MAX_TERMINAL_LENGTH
191  && i < startSentenceIndex + MAX_SOURCE_ELEMENTS - nbSrcElt
192  && i < startSentenceIndex + HR_MAX_HEIGHT - nbCoveredWords; i++) {
193  partialPattern.add(sourceSentence.get(i));
194  Rule r = new Rule(partialPattern, new ArrayList<Integer>());
195  Set<Rule> right = getPatternInstancesFromSourceAndPattern2(
196  sourceSentence, sidePattern, i + 1,
197  startPatternIndex + 1, nbSrcElt + i
198  - startSentenceIndex + 1, nbCoveredWords + i
199  - startSentenceIndex + 1);
200  Set<Rule> merged = merge(r, right);
201  res.addAll(merged);
202  }
203  } else {
204  partialPattern.add(Integer.parseInt(sidePattern
205  .get(startPatternIndex)));
206  Rule r = new Rule(partialPattern, new ArrayList<Integer>());
207  for (int i = startSentenceIndex; i < sourceSentence.size()
208  - (sidePattern.size() - startPatternIndex - 1)
209  && i < startSentenceIndex + MAX_NONTERMINAL_SPAN
210  && i < startSentenceIndex + HR_MAX_HEIGHT - nbCoveredWords; i++) {
211  Set<Rule> merged = merge(
212  r,
213  getPatternInstancesFromSourceAndPattern2(
214  sourceSentence, sidePattern, i + 1,
215  startPatternIndex + 1, nbSrcElt + 1,
216  nbCoveredWords + i - startSentenceIndex + 1));
217  // System.err.println("merged: " + merged);
218  res.addAll(merged);
219  }
220  }
221  return res;
222  }
