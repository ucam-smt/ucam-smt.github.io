117 template<typename T>
118 T DotProduct ( const TropicalSparseTupleWeight<T>& w,
119  const std::vector<T>& vw ) {
120  T result = w.DefaultValue().Value();
121  for ( SparseTupleWeightIterator<TropicalWeightTpl<T>, int> it ( w ); !it.Done();
122  it.Next() ) {
123  T param;
124  // Special case for flat params
125  if ( vw.empty() ) {
126  param = 1;
127  } else if ( it.Value().first > int ( vw.size() ) ) {
128  cerr
129  << "feature vector has a larger dimensionality than the parameters. "
130  << "Params: " << vw.size() << " Features: "
131  << it.Value().first << endl;
132  exit ( 1 );
133  } else {
134  param = vw[it.Value().first - 1];
135  }
136  result += it.Value().second.Value() * param;
137  }
138  return result;
139 }

141 template<class T>
142 inline TropicalSparseTupleWeight<T> Plus (
143  const TropicalSparseTupleWeight<T>& vw1,
144  const TropicalSparseTupleWeight<T>& vw2 ) {
145  T w1 = DotProduct ( vw1, TropicalSparseTupleWeight<T>::Params() );
146  T w2 = DotProduct ( vw2, TropicalSparseTupleWeight<T>::Params() );
147  return w1 < w2 ? vw1 : vw2;
148 }
