210  bool run ( Data& d ) {
211  cykfstresult_.DeleteStates();
212  this->d_ = &d;
213  hieroindexexistence_.clear();
214  LINFO ( "Running HiFST" );
215  //Reset one-time warnings for inexistent language models.
216  warned_ = false;
217  pdtparens_.clear();
218  cykdata_ = d.cykdata;
219  if ( !USER_CHECK ( cykdata_, "cyk parse has not been executed previously?" ) ) {
220  resetExternalData (d);
221  return true;
222  }
223  if ( d.cykdata->success == CYK_RETURN_FAILURE ) {
225  fst::VectorFst<Arc> aux;
226  d.fsts[outputkey_] = &cykfstresult_;
227  d.vcat = cykdata_->vcat;
228  resetExternalData (d);
229  return false;
230  }
232  initLocalConditions();
233  rtn_ = new RTNT;
234  if ( localprune_ )
235  rtnnumstates_ = new ExpandedNumStatesRTNT;
236  rfba_ = new ReplaceFstByArcT ( cykdata_->vcat, replacefstbyarc_,
237  replacefstbyarcexceptions_, aligner_, replacefstbynumstates_ );
238  piscount_ = 0; //reset pruning-in-search count to 0
239  LINFO ( "Second Pass: FST-building!" );
240  d.stats->setTimeStart ( "lattice-construction" );
241  //Owned by rtn_;
242  fst::Fst<Arc> *sfst = buildRTN ( cykdata_->categories["S"], 0,
243  cykdata_->sentence.size() - 1 );
244  d.stats->setTimeEnd ( "lattice-construction" );
245  cykfstresult_ = (*sfst);
246  LINFO ( "Final - RTN head optimizations !" );
247  optimize ( &cykfstresult_ ,
248  std::numeric_limits<unsigned>::max() ,
249  !hipdtmode_ && optimize_
250  );
251  FORCELINFO ("Stats for Sentence " << d.sidx <<
252  ": local pruning, number of times=" << piscount_);
253  d.stats->lpcount = piscount_; //store local pruning counts in stats
254  LINFO ("RTN expansion starts now!");
255  //Expand...
256  {
258  Label hieroindex = APBASETAG + 1 * APCCTAG + 0 * APXTAG +
259  ( cykdata_->sentence.size() - 1 ) * APYTAG;
260  if ( hieroindexexistence_.find ( hieroindex ) == hieroindexexistence_.end() )
261  pairlabelfsts_.push_back ( pair< Label, const fst::Fst<Arc> * > ( hieroindex,
262  &cykfstresult_ ) );
264  fst::ReplaceUtil<Arc> replace_util (pairlabelfsts_, hieroindex,
265  !aligner_); //has ownership of modified rtn fsts
266  if (rtnopt_) {
267  LINFO ("rtn optimizations...");
268  d_->stats->setTimeStart ("replace-opts");
269  replace_util.ReplaceTrivial();
270  replace_util.ReplaceUnique();
271  replace_util.Connect();
272  pairlabelfsts_.clear();
273  replace_util.GetFstPairs (&pairlabelfsts_);
274  d_->stats->setTimeEnd ("replace-opts");
275  }
276  //After optimizations, we can write RTN if required by user
277  writeRTN();
278  boost::scoped_ptr< fst::VectorFst<Arc> > efst (new fst::VectorFst<Arc>);
279  if (!hipdtmode_ ) {
280  LINFO ("Final Replace (RTN->FSA), main index=" << hieroindex);
281  d_->stats->setTimeStart ("replace-rtn-final");
282  Replace (pairlabelfsts_, &*efst, hieroindex, !aligner_);
283  d_->stats->setTimeEnd ("replace-rtn-final");
284  } else {
285  LINFO ("Final Replace (RTN->PDA)");
286  d_->stats->setTimeStart ("replace-pdt-final");
287  Replace (pairlabelfsts_, &*efst, &pdtparens_, hieroindex);
288  d_->stats->setTimeEnd ("replace-pdt-final");
289  LINFO ("Number of pdtparens=" << pdtparens_.size() );
290  }
291  LINFO ("Removing Epsilons...");
292  fst::RmEpsilon<Arc> ( &*efst );
293  LINFO ("Done! NS=" << efst->NumStates() );
294  //Apply filters
295  applyFilters ( &*efst );
296  //Compose with full reference lattice to ensure that final lattice is correct.
297  if ( d.fsts.find ( fullreferencelatticekey_ ) != d.fsts.end() ) {
298  if ( static_cast< fst::VectorFst<Arc> * >
299  (d.fsts[fullreferencelatticekey_])->NumStates() > 0 ) {
300  LINFO ( "Composing with full reference lattice, NS=" <<
301  static_cast< fst::VectorFst<Arc> * >
302  (d.fsts[fullreferencelatticekey_])->NumStates() );
303  fst::Compose<Arc> ( *efst,
304  * ( static_cast<fst::VectorFst<Arc> * > (d.fsts[fullreferencelatticekey_]) ),
305  &*efst );
306  LINFO ( "After composition: NS=" << efst->NumStates() );
307  } else {
308  LINFO ( "No composition with full ref lattice" );
309  };
310  } else {
311  LINFO ( "No composition with full ref lattice" );
312  };
313  //Apply language model
314  fst::VectorFst<Arc> *res = NULL;
315  if (efst->NumStates() )
316  res = applyLanguageModel ( *efst );
317  else {
318  LWARN ("Empty lattice -- skipping LM application");
319  }
320  if ( res != NULL ) {
321  boost::shared_ptr<fst::VectorFst<Arc> >latlm ( res );
322  if ( latlm.get() == efst.get() ) {
323  LWARN ( "Yikes! Unexpected situation! Will it crash? (muhahaha) " );
324  }
325  //Todo: union with shortest path...
326  if ( pruneweight_ < std::numeric_limits<float>::max() ) {
327  if (!hipdtmode_ || pdtparens_.empty() ) {
328  LINFO ("Pruning, weight=" << pruneweight_);
329  fst::Prune<Arc> (*latlm, &cykfstresult_, mw_ ( pruneweight_ ) );
330  } else {
331  LINFO ("Expanding, weight=" << pruneweight_);
332  fst::ExpandOptions<Arc> eopts (true, false, mw_ ( pruneweight_ ) );
333  Expand ( *latlm, pdtparens_, &cykfstresult_, eopts);
334  pdtparens_.clear();
335  }
336  } else {
337  LINFO ("Copying through full lattice with lm scores");
338  cykfstresult_ = *latlm;
339  }
340  } else {
341  LINFO ("Copying through full lattice (no lm)");
342  cykfstresult_ = *efst;
343  }
344  if ( hieroindexexistence_.find ( hieroindex ) == hieroindexexistence_.end() )
345  pairlabelfsts_.pop_back();
346  }
347  pairlabelfsts_.clear();
348  LINFO ( "Reps" );
349  fst::RmEpsilon ( &cykfstresult_ );
350  LINFO ( "NS=" << cykfstresult_.NumStates() );
351  //This should delete all pertinent fsas...
352  LINFO ( "deleting data stuff..." );
353  delete rtn_;
354  if ( localprune_ )
355  delete rtnnumstates_;
356  delete rfba_;
357  d.vcat = cykdata_->vcat;
358  resetExternalData (d);
359  d.fsts[outputkey_] = &cykfstresult_;
360  if (hipdtmode_ && pdtparens_.size() )
361  d.fsts[outputkey_ + ".parens" ] = &pdtparens_;
362  LINFO ( "done..." );
363  FORCELINFO ( "End Sentence ******************************************************" );
364  d.stats->setTimeEnd ( "sent-dec" );
365  d.stats->message += "[" + ucam::util::getTimestamp() + "] End Sentence\n";
366  return false;
367  };

425  fst::Fst<Arc>* buildRTN ( unsigned int cc, unsigned int x, unsigned int y ) {
426  fst::Fst<Arc> *ptr = ( *rtn_ ) ( cc, x, y );
427  if ( ptr != NULL ) return ptr;
428 #ifdef PRINTDEBUG
429  std::ostringstream o;
430  o << cc << "." << x << "." << y;
431 #endif
432  unsigned& nnt = cykdata_->nnt;
433  grammar_inversecategories_t& vcat = cykdata_->vcat;
434  std::stringstream ostr;
435  ostr << vcat[cc] << "." << x << "." << y;
436  std::string filename;
437  ostr >> filename;
438  SentenceSpecificGrammarData& g = *d_->ssgd;
439  MultiUnionT mur;
440  Label hieroindex = APBASETAG + cc * APCCTAG + x * APXTAG + y * APYTAG;
441  LDEBUG ( "bp> " << cc << "," << x << "," << y << ":" <<
442  ( unsigned ) cykdata_->bp ( cc, x, y ).size() );
443  for ( unsigned i = 0; i < cykdata_->bp ( cc, x, y ).size(); i++ ) {
444  unsigned idx = cykdata_->cykgrid ( cc, x, y, i );
445  std::vector<fst::Fst<Arc>*> requiredfsts;
446  if ( g.isPhrase ( idx ) ) {
447  mur.Add ( addRule ( idx, requiredfsts ) ) ;
448  LDEBUG ( "AT " << cc << "," << x << "," << y <<
449  ":adding phrase-based rule index " << idx );
450  continue;
451  }
452  const cykparser_ruledependencies_t& mybp = cykdata_->bp ( cc, x, y );
453  for ( unsigned j = 0; j < mybp[i].size(); j += 3 ) {
454  if ( mybp[i][j] > nnt ) {
455  continue;
456  }
457  requiredfsts.push_back ( buildRTN ( mybp[i][j], mybp[i][j + 1],
458  mybp[i][j + 2] ) );
459  LDEBUG ( "back to bp> " << cc << "," << x << "," << y << ":" <<
460  ( unsigned ) cykdata_->bp ( cc, x, y ).size() );
461  }
462  mapfsts ( idx, requiredfsts );
463  LDEBUG ( "AT " << cc << "," << x << "," << y << ": adding hiero rule index " <<
464  idx );
465  mur.Add ( addRule ( idx, requiredfsts ) );
466  }
467  boost::shared_ptr< fst::VectorFst<Arc> > mdfst ( mur() );
468  LDBG_EXECUTE ( mdfst->Write ( "fsts/" + o.str() + ".fst" ) );
469  //Optimize
470  optimize ( &*mdfst ,
471  std::numeric_limits<unsigned>::max(),
472  optimize_ );
473  LDEBUG ( "AT " << cc << "," << x << "," << y << ": FST built!" );
474  LDBG_EXECUTE ( mdfst->Write ( "fsts/" + o.str() + "redm.fst" ) );
475  d_->stats->numstates[ cc * 1000000 + y * 1000 + x ] =
476  mdfst->NumStates(); //Just store the number of states of the not-expanded FSA.
477  //Calculate expanded number of states of the partial rtn.
478  if ( localprune_ )
479  rtnnumstates_->update ( cc, x, y, &*mdfst );
480  boost::scoped_ptr< fst::VectorFst<Arc> > pruned ( localPruning ( *mdfst, cc, x,
481  y ) );
482  //We now might have a pruned lattice!
483  if ( pruned.get() != NULL ) {
484  optimize (&*pruned , numstatesthreshold_ , !hipdtmode_ && optimize_ );
485  *mdfst = *pruned;
486  //Only if we prune, we add to stats total number of states of full and pruned lattice
487  d_->stats->numstates[ cc * 1000000 + y * 1000 + x ] = ( *rtnnumstates_ ) ( cc,
488  x,
489  y );
490  rtnnumstates_->update ( cc, x, y,
491  &*mdfst ); //Update rtnnumstates again...
492  d_->stats->numprunedstates[ cc * 1000000 + y * 1000 + x ] = ( *rtnnumstates_ ) (
493  cc, x, y );
494  } else {
495  LDEBUG ( "AT " << cc << "," << x << "," << y << ":No pruning" );
496  }
497  boost::shared_ptr< fst::VectorFst<Arc> > outfst ( ( *rfba_ ) ( *mdfst,
498  hieroindex ) );
499  if ( outfst.get() != NULL ) {
500  LDEBUG ( "AT " << cc << "," << x << "," << y << ": replacefstbyarcfor cat= " <<
501  vcat[cc] << ",NS=" << mdfst->NumStates() );
502  rtn_->Add ( cc, x, y, outfst , mdfst );
503  hieroindexexistence_.insert ( hieroindex );
504  pairlabelfsts_.push_back ( pair< Label, const fst::Fst<Arc> * > ( hieroindex,
505  &*mdfst ) );
506  } else {
507  rtn_->Add ( cc, x, y, mdfst , outfst );
508  LDEBUG ( "AT: " << cc << "," << x << "," << y << ":" <<
509  "Delaying not applied. Stored, NS=" << ( unsigned ) mdfst->NumStates() );
510  }
511  return ( *rtn_ ) ( cc, x, y );
512  };

521  fst::VectorFst<Arc> *addRule ( unsigned rule_idx,
522  std::vector<fst::Fst<Arc>*>& lowerfsts ) {
523  SentenceSpecificGrammarData& gd = *d_->ssgd;
524  std::vector<std::string> translation = gd.getRHSSplitTranslation ( rule_idx );
525  if ( !translation.size() ) {
526  LERROR ( gd.getRule ( rule_idx ) );
527  translation.push_back ( 0 );
528  }
529  for (unsigned k = 0; k < translation.size(); ++k) {
530  if ( translation[k] == "<s>" ) {
531  translation[k] = "1";
532  } else if ( translation[k] == "</s>" ) translation[k] = "2";
533  else if ( translation[k] == "<dr>" ) {
534  std::stringstream dr;
535  dr << DR;
536  translation[k] = dr.str();
537  LDEBUG ( "Deletion rule: " << gd.getRule ( rule_idx ) << "," <<
538  translation[k] );
539  } else if ( translation[k] == "<oov>" ) {
540  std::stringstream oov;
541  oov << OOV;
542  translation[k] = oov.str();
543  LDEBUG ( "oov rule: " << gd.getRule ( rule_idx ) << "," << translation[k] );
544  } else if ( translation[k] == "<sep>" ) {
545  std::stringstream sep;
546  sep << SEP;
547  translation[k] = sep.str();
548  LDEBUG ( "separator rule: " << gd.getRule ( rule_idx ) << "," <<
549  translation[k] );
550  }
551  }
552  LDEBUG ( "Starting to build!" );
553  fst::VectorFst<Arc> *rulefst = new fst::VectorFst<Arc>;
554  rulefst->AddState();
555  rulefst->SetStart ( 0 );
556  rulefst->AddState();
557  Label iw2 = gd.getIdx ( rule_idx ) + 1;
558  Label iw;
559  if ( !aligner_ ) iw = 0;
560  else iw = iw2;
561  LINFO ("Building FST for rule " << gd.getRule ( rule_idx ) );
562  unsigned kmax = translation.size();
563  unsigned nonterminal = 0;
564  std::vector< pair< Label, const fst::Fst<Arc> * > > pairlabelfsts;
565  for ( unsigned k = 0; k < kmax; k++ ) {
566  //if non-terminal... just place special arc and expand later...
567  Label ow;
568  if ( !isTerminal ( translation[k] ) ) {
569  ow = APRULETAG + nonterminal;
570  USER_CHECK ( lowerfsts.size() > nonterminal,
571  "Missing fsts to build the rule..." );
572  pairlabelfsts.push_back ( pair< Label, const fst::Fst<Arc> * > ( ow,
573  lowerfsts[nonterminal++] ) );
574  } else {
575  std::istringstream buffer ( translation[k] );
576  buffer >> ow;
577  }
578  rulefst->AddState();
579  Label iw;
580  if ( !aligner_ ) iw = ow;
581  else iw = NORULE;
582  rulefst->AddArc ( k, Arc ( iw, ow, Weight::One(), k + 1 ) );
583  }
584  float w = gd.getWeight ( rule_idx );
585  Weight weight = mw_ ( w , iw2 );
586  rulefst->AddArc ( kmax, Arc ( iw, 0, weight, kmax + 1 ) );
587  rulefst->SetFinal ( kmax + 1, Weight::One() );
588  fst::VectorFst<Arc>* auxi;
589  if ( nonterminal > 0 ) {
590  pairlabelfsts.push_back ( pair< Label, const fst::Fst<Arc> * >
591  ( APRULETAG + nonterminal, rulefst ) );
592  fst::VectorFst<Arc> *aux = new fst::VectorFst<Arc>;
593  Replace (pairlabelfsts, aux, APRULETAG + nonterminal, !aligner_);
594  delete rulefst;
595  rulefst = aux;
596  }
597  fst::RmEpsilon<Arc> ( rulefst );
598  return rulefst;
599  }

601  //Note that local conditions can be sentence-specific
602  void initLocalConditions() {
603  if ( !localprune_ ) return;
604  if ( !lpctuples_.size() ) return;
605  lpc_.clear();
606  LINFO ( "Set up conditions for local cell pruning" );
