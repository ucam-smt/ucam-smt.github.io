139 #else

185  VectorFst<Arc> * operator() () {
186  if (!fst_.NumStates() ) {
187  LWARN ("Empty lattice. ... Skipping LM application!");
188  return NULL;
189  }
190  while ( qc_.size() ) {
191  StateId s = qc_.front();
192  qc_.pop();
193  pair<StateId, const typename KenLMModelT::State> p = get ( s );
194  StateId& s1 = p.first;
195  const typename KenLMModelT::State s2 = p.second;
196  for ( ArcIterator< VectorFst<Arc> > arc1 ( fst_, s1 ); !arc1.Done();
197  arc1.Next() ) {
198  const Arc& a1 = arc1.Value();
199  float w = 0;
200  float wp = wp_;
201  typename KenLMModelT::State nextlmstate;
202  if ( epsilons_.find ( a1.olabel ) == epsilons_.end() ) {
203  w = lmmodel_.Score ( s2, idbridge_.map (a1.olabel), nextlmstate ) * natlog10_;
204  //silly hack
205  if ( a1.olabel <= 2 ) {
206  wp = 0;
207  if (a1.olabel == 1 ) w = 0; //get same result as srilm
208  }
209  } else {
210  nextlmstate = s2;
211  wp = 0; //We don't count epsilon labels
212  }
213  pair<StateId, bool> nextp = add ( nextlmstate
214  , a1.nextstate
215  , fst_.Final ( a1.nextstate ) );
216  StateId& newstate = nextp.first;
217  bool visited = nextp.second;
218  composed_->AddArc ( s
219  , Arc ( a1.ilabel, a1.olabel
220  , Times ( a1.weight, Times (mw_ ( w ) , mw_ (wp) ) )
221  , newstate ) );
222  //Finally, only add newstate to the queue if it hasn't been visited previously
223  if ( !visited ) {
224  qc_.push ( newstate );
225  }
226  }
227  }
228  LINFO ( "Done! Number of states=" << composed_->NumStates() );
229  return composed_;
230  };

238  inline pair <StateId, bool> add ( typename KenLMModelT::State& m2nextstate,
239  StateId m1nextstate, Weight m1stateweight ) {
240  static StateId lm = 0;
241  getIdx ( m2nextstate );
243  if ( seenlmstates_.find ( history ) == seenlmstates_.end() ) {
244  seenlmstates_[history] = ++lm;
245  }
246  uint64_t compound = m1nextstate * sid + seenlmstates_[history];
247  LDEBUG ( "compound id=" << compound );
248  if ( stateexistence_.find ( compound ) == stateexistence_.end() ) {
249  LDEBUG ( "New State!" );
250  statemap_[composed_->NumStates()] =
251  pair<StateId, const typename KenLMModelT::State > ( m1nextstate, m2nextstate );
252  composed_->AddState();
253  if ( m1stateweight != mw_ ( ZPosInfinity() ) ) composed_->SetFinal (
254  composed_->NumStates() - 1, m1stateweight );
255  stateexistence_[compound] = composed_->NumStates() - 1;
256  return pair<StateId, bool> ( composed_->NumStates() - 1, false );
257  }
258  return pair<StateId, bool> ( stateexistence_[compound], true );
259  };
