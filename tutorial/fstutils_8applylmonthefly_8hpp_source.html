168  virtual VectorFst<Arc> *run(const VectorFst<Arc>& fst, unordered_set<Label> const &epsilons) {
169  epsilons_ = epsilons; // this may be necessary e.g. for pdts

173  VectorFst<Arc> * operator() (const VectorFst<Arc>& fst) {
174  const VectorFst<Arc> &fst_ = fst;
175  if (!fst_.NumStates() ) {
176  LWARN ("Empty lattice. ... Skipping LM application!");
177  return NULL;
178  }
179 
180  VectorFst<Arc> *composed;
181  composed = new VectorFst<Arc>;
183  typename KenLMModelT::State bs = lmmodel_.NullContextState();
184  pair<StateId, bool> nextp = add ( composed, bs, fst_.Start(), fst_.Final ( fst_.Start() ) );
185  qc_.push ( nextp.first );
186  composed->SetStart ( nextp.first );
187  while ( qc_.size() ) {
188  StateId s = qc_.front();
189  qc_.pop();
190  pair<StateId, const typename KenLMModelT::State> p = get ( s );
191  StateId& s1 = p.first;
192  const typename KenLMModelT::State s2 = p.second;
193  for ( ArcIterator< VectorFst<Arc> > arc1 ( fst_, s1 ); !arc1.Done();
194  arc1.Next() ) {
195  const Arc& a1 = arc1.Value();
196  float w = 0;
197  float wp = wp_;
198  typename KenLMModelT::State nextlmstate;
199  if ( epsilons_.find ( a1.olabel ) == epsilons_.end() ) {
200  w = lmmodel_.Score ( s2, idbridge_.map (a1.olabel), nextlmstate ) * natlog10_;
201  //silly hack
202  if ( a1.olabel <= 2 ) {
203  wp = 0;
204  if (a1.olabel == 1 ) w = 0; //get same result as srilm
205  }
206  } else {
207  nextlmstate = s2;
208  wp = 0; //We don't count epsilon labels
209  }
210  pair<StateId, bool> nextp = add ( composed, nextlmstate
211  , a1.nextstate
212  , fst_.Final ( a1.nextstate ) );
213  StateId& newstate = nextp.first;
214  bool visited = nextp.second;
215  composed->AddArc ( s
216  , Arc ( a1.ilabel, a1.olabel
217  , Times ( a1.weight, Times (mw_ ( w ) , mw_ (wp) ) )
218  , newstate ) );
219  //Finally, only add newstate to the queue if it hasn't been visited previously
220  if ( !visited ) {
221  qc_.push ( newstate );
222  }
223  }
224  }
225  LINFO ( "Done! Number of states=" << composed->NumStates() );
226  stateexistence_.clear();
227  statemap_.clear();
228  seenlmstates_.clear();
229  history.resize( lmmodel_.Order(), 0);
230  return composed;
231  };

239  inline pair <StateId, bool> add ( fst::VectorFst<Arc> *composed, typename KenLMModelT::State& m2nextstate,
240  StateId m1nextstate, Weight m1stateweight ) {
241  static StateId lm = 0;
242  getIdx ( m2nextstate );
244  if ( seenlmstates_.find ( history ) == seenlmstates_.end() ) {
245  seenlmstates_[history] = ++lm;
246  }
247  uint64_t compound = m1nextstate * sid + seenlmstates_[history];
248  LDEBUG ( "compound id=" << compound );
249  if ( stateexistence_.find ( compound ) == stateexistence_.end() ) {
250  LDEBUG ( "New State!" );
251  statemap_[composed->NumStates()] =
252  pair<StateId, const typename KenLMModelT::State > ( m1nextstate, m2nextstate );
253  composed->AddState();
254  if ( m1stateweight != mw_ ( ZPosInfinity() ) ) composed->SetFinal (
255  composed->NumStates() - 1, m1stateweight );
256  stateexistence_[compound] = composed->NumStates() - 1;
257  return pair<StateId, bool> ( composed->NumStates() - 1, false );
258  }
259  return pair<StateId, bool> ( stateexistence_[compound], true );
260  };
