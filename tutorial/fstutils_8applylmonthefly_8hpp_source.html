121  ApplyLanguageModelOnTheFly ( const Fst<Arc>& fst,
122  KenLMModelT& model,
123 #ifndef USE_GOOGLE_SPARSE_HASH
124  unordered_set<Label>& epsilons,
125 #else
126  google::dense_hash_set<Label>& epsilons,
127 #endif
128  bool natlog,
129  float lmscale ,
130  float lmwp,
131  const IdBridgeT& idbridge
132  ) :
133  composed_ ( NULL ) ,
134  natlog10_ ( natlog ? -lmscale* ::log ( 10.0 ) : -lmscale ),
135  fst_ ( fst ),
136  lmmodel_ ( model ),
137  vocab_ ( model.GetVocabulary() ),
138  wp_ ( lmwp ) ,
139  epsilons_ ( epsilons ) ,
140  history ( model.Order(), 0),
141  idbridge_ (idbridge) {
142 #ifdef USE_GOOGLE_SPARSE_HASH
143  stateexistence_.set_empty_key ( numeric_limits<ull>::max() );
144  statemap_.set_empty_key ( numeric_limits<uint64_t>::max() );
145  basic_string<unsigned> aux (KENLM_MAX_ORDER, numeric_limits<unsigned>::max() );
146  seenlmstates_.set_empty_key ( aux );
147 #endif
148  buffersize = ( model.Order() - 1 ) * sizeof ( unsigned int );
149  buffer = const_cast<unsigned *> ( history.c_str() );
150  if (!fst_.NumStates() ) {
151  LWARN ("Empty lattice");
152  return;
153  }
154  composed_ = new VectorFst<Arc>;
155  typename KenLMModelT::State bs = model.NullContextState();
157  pair<StateId, bool> nextp = add ( bs, fst_.Start(),
158  fst_.Final ( fst_.Start() ) );
159  qc_.push ( nextp.first );
160  composed_->SetStart ( nextp.first );
161  };
162 
164  ~ApplyLanguageModelOnTheFly() {
165  if (composed_ ) delete composed_;
166  };

169  VectorFst<Arc> * operator() () {
170  if (!fst_.NumStates() ) {
171  LWARN ("Empty lattice. ... Skipping LM application!");
172  return NULL;
173  }
174  while ( qc_.size() ) {
175  StateId s = qc_.front();
176  qc_.pop();
177  pair<StateId, const typename KenLMModelT::State> p = get ( s );
178  StateId& s1 = p.first;
179  const typename KenLMModelT::State s2 = p.second;
180  for ( ArcIterator< VectorFst<Arc> > arc1 ( fst_, s1 ); !arc1.Done();
181  arc1.Next() ) {
182  const Arc& a1 = arc1.Value();
183  float w = 0;
184  float wp = wp_;
185  typename KenLMModelT::State nextlmstate;
186  if ( epsilons_.find ( a1.olabel ) == epsilons_.end() ) {
187  w = lmmodel_.Score ( s2, idbridge_.map (a1.olabel), nextlmstate ) * natlog10_;
188  //silly hack
189  if ( a1.olabel <= 2 ) {
190  wp = 0;
191  if (a1.olabel == 1 ) w = 0; //get same result as srilm
192  }
193  } else {
194  nextlmstate = s2;
195  wp = 0; //We don't count epsilon labels
196  }
197  pair<StateId, bool> nextp = add ( nextlmstate
198  , a1.nextstate
199  , fst_.Final ( a1.nextstate ) );
200  StateId& newstate = nextp.first;
201  bool visited = nextp.second;
202  composed_->AddArc ( s
203  , Arc ( a1.ilabel, a1.olabel
204  , Times ( a1.weight, Times (mw_ ( w ) , mw_ (wp) ) )
205  , newstate ) );
206  //Finally, only add newstate to the queue if it hasn't been visited previously
207  if ( !visited ) {
208  qc_.push ( newstate );
209  }
210  }
211  }
212  LINFO ( "Done! Number of states=" << composed_->NumStates() );
213  return composed_;
214  };

222  inline pair <StateId, bool> add ( typename KenLMModelT::State& m2nextstate,
223  StateId m1nextstate, Weight m1stateweight ) {
224  static StateId lm = 0;
225  getIdx ( m2nextstate );
227  if ( seenlmstates_.find ( history ) == seenlmstates_.end() ) {
228  seenlmstates_[history] = ++lm;
229  }
230  uint64_t compound = m1nextstate * sid + seenlmstates_[history];
231  LDEBUG ( "compound id=" << compound );
232  if ( stateexistence_.find ( compound ) == stateexistence_.end() ) {
233  LDEBUG ( "New State!" );
234  statemap_[composed_->NumStates()] =
235  pair<StateId, const typename KenLMModelT::State > ( m1nextstate, m2nextstate );
236  composed_->AddState();
237  if ( m1stateweight != mw_ ( ZPosInfinity() ) ) composed_->SetFinal (
238  composed_->NumStates() - 1, m1stateweight );
239  stateexistence_[compound] = composed_->NumStates() - 1;
240  return pair<StateId, bool> ( composed_->NumStates() - 1, false );
241  }
242  return pair<StateId, bool> ( stateexistence_[compound], true );
243  };

249  inline void getIdx ( const typename KenLMModelT::State& state,
250  uint order = 4 ) {
251  memcpy ( buffer, state.words, buffersize );
252  for ( uint k = state.length; k < history.size(); ++k ) history[k] = 0;
253  };
254 
256  inline pair<StateId, typename KenLMModelT::State > get ( StateId state ) {
257  return statemap_[state];
258  };
