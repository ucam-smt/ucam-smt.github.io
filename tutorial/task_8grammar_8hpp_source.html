116  bool run ( Data& d ) {

239  __always_inline void parse ( std::string& line ) {
240  boost::algorithm::trim ( line );
241  if ( line == "" ) return;
242  std::size_t pos1 = line.find_first_of ( " " );
243  std::size_t pos2 = line.find_first_of ( " ", pos1 + 1 );
244  std::size_t pos3 = line.find_first_of ( " \n\0", pos2 + 1 );
245  // vector<float> weights = ParseParamString<float> ( line, pos3 + 1 );
246  std::vector<float> weights;
247  ucam::util::ParseParamString<float> ( line, weights, pos3 + 1 );
248  std::string sweight = ucam::util::toString<float> ( ucam::util::dotproduct (
249  weights, grammarscales_ )
250  , std::numeric_limits<uint>::max() );
251  ucam::util::trim_trailing_zeros ( sweight );
252  line = line.substr ( 0, pos3 + 1 ) + sweight;
253  gd_.filecontents += line + '\n';
254  posindex pi;
255  bool waitingfornextfield = false;
256  uint cf = 2; //Second field
257  char previous = ' ';
258  for ( uint k = 0; k < line.size(); ++k ) {
259  if ( previous == ' ' && line[k] != ' ' ) --cf;
260  if ( !cf ) {
261  pi.o = k;
262  break;
263  }
264  previous = line[k];
265  }
266  pi.p = pos_ + pi.o;
267  std::string pattern;
268  bool word = false;
269  bool nt = false;
270  for ( uint k = pi.o; k < line.size(); ++k ) {
271  if ( line[k] == ' ' ) break;
272  if ( line[k] >= '0' && line[k] <= '9' ) {
273  if ( !word && !nt ) {
274  pattern += 'w';
275  word = true;
276  nt = false;
277  }
278  } else if ( line[k] >= 'A' && line[k] <= 'Z' ) {
279  if ( !nt ) {
280  pattern += 'X';
281  nt = true;
282  word = false;
283  }
284  } else {
285  pattern += line[k];
286  nt = word = false;
287  }
288  }
289  if ( gd_.patterns.find ( pattern ) == gd_.patterns.end() ) {
290  gd_.patterns.insert ( pattern );
291  }
292  pi.order = vpq_->size();
293  vpq_->push ( pi );
294  pos_ += line.size() + 1;
295  LDEBUG2 ( "reading rule " << line << ", at line " << pi.order << ", pattern=" <<
296  pattern );
297  if ( pattern == "X" ) {
298  LINFO ( "Identity rule detected:" << line << "===" );
299  nth_.insertIdentityRule ( line );
300  } else {
301  nth_.insertLHS ( line.substr ( 0, pi.o - 1 ) );
302  }
303  };
